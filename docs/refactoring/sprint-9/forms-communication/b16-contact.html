Usage Guidelines

  • Icons: Lucide icons in teal glow circles (40px)

STEP 2 TypeScript Type Definitions

Define TypeScript interfaces for type-safe development with the Contact block.

TypeScript src/types/blocks.ts 
// src/types/blocks.ts
export interface ContactAddress {
  street?: string
  postalCode?: string
  city?: string
}

export interface OpeningHour {
  day: string
  hours: string
  id?: string
}

export interface ContactBlock {
  blockType: 'contact'
  id?: string
  title: string
  subtitle?: string
  address?: ContactAddress
  phone?: string
  email?: string
  openingHours?: OpeningHour[]
  showMap: boolean
  mapUrl?: string
}

// Type Guards
export function isContactBlock(block: any): block is ContactBlock {
  return block?.blockType === 'contact'
}

export function hasAddress(block: ContactBlock): boolean {
  return !!(block.address?.street || block.address?.city)
}

export function hasOpeningHours(block: ContactBlock): boolean {
  return !!(block.openingHours && block.openingHours.length > 0)
}

// Helper Functions
export function formatAddress(address: ContactAddress): string {
  const parts = [address.street, `${address.postalCode} ${address.city}`.trim()]
  return parts.filter(Boolean).join(', ')
}

export function formatPhoneLink(phone: string): string {
  // Remove all non-numeric characters except +
  return `tel:${phone.replace(/[^0-9+]/g, '')}`
}
STEP 3 React Component Implementation

Create the Next.js React component with Tailwind CSS styling and responsive layout.

TypeScript React src/branches/blocks/forms-communication/Contact/Component.tsx 
// src/branches/blocks/forms-communication/Contact/Component.tsx
'use client'

import React from 'react'
import { MapPin, Phone, Mail, Clock } from 'lucide-react'
import type { ContactBlock } from '@/types/blocks'

interface ContactProps {
  block: ContactBlock
}

export const Contact: React.FC<ContactProps> = ({ block }) => {
  const {
    title,
    subtitle,
    address,
    phone,
    email,
    openingHours,
    showMap,
    mapUrl
  } = block

  const hasAddress = address?.street || address?.city
  const hasOpeningHours = openingHours && openingHours.length > 0

  return (
    <section className="py-16 md:py-20">
      <div className="container mx-auto px-6">
        {/* Header */}
        {(title || subtitle) && (
          <div className="text-center mb-12">
            {title && <h2 className="font-serif text-3xl md:text-4xl text-navy mb-3">{title}</h2>}
            {subtitle && <p className="text-base text-grey-dark max-w-2xl mx-auto">{subtitle}</p>}
          </div>
        )}

        {/* Contact Grid */}
        <div className="grid md:grid-cols-2 gap-12 md:gap-16 max-w-5xl mx-auto">
          {/* Contact Information */}
          <div className="flex flex-col gap-8">
            {/* Address */}
            {hasAddress && (
              <div className="flex gap-4">
                <div className="w-10 h-10 bg-teal-glow rounded-lg flex items-center justify-center flex-shrink-0">
                  <MapPin className="w-5 h-5 text-teal" />
                </div>
                <div>
                  <h3 className="font-bold text-sm text-navy mb-2">Adres</h3>
                  <p className="text-sm text-grey-dark leading-relaxed">
                    {address.street && (
                      <>
                        {address.street}
                        <br />
                      </>
                    )}
                    {(address.postalCode || address.city) && (
                      <>
                        {address.postalCode} {address.city}
                      </>
                    )}
                  </p>
                </div>
              </div>
            )}

            {/* Phone */}
            {phone && (
              <div className="flex gap-4">
                <div className="w-10 h-10 bg-teal-glow rounded-lg flex items-center justify-center flex-shrink-0">
                  <Phone className="w-5 h-5 text-teal" />
                </div>
                <div>
                  <h3 className="font-bold text-sm text-navy mb-2">Telefoon</h3>
                  <a href={`tel:${phone.replace(/[^0-9+]/g, '')}`} className="text-sm text-teal hover:text-teal-dark transition">
                    {phone}
                  </a>
                </div>
              </div>
            )}

            {/* Email */}
            {email && (
              <div className="flex gap-4">
                <div className="w-10 h-10 bg-teal-glow rounded-lg flex items-center justify-center flex-shrink-0">
                  <Mail className="w-5 h-5 text-teal" />
                </div>
                <div>
                  <h3 className="font-bold text-sm text-navy mb-2">Email</h3>
                  <a href={`mailto:${email}`} className="text-sm text-teal hover:text-teal-dark transition">
                    {email}
                  </a>
                </div>
              </div>
            )}

            {/* Opening Hours */}
            {hasOpeningHours && (
              <div className="flex gap-4">
                <div className="w-10 h-10 bg-teal-glow rounded-lg flex items-center justify-center flex-shrink-0">
                  <Clock className="w-5 h-5 text-teal" />
                </div>
                <div className="flex-1">
                  <h3 className="font-bold text-sm text-navy mb-2">Openingstijden</h3>
                  <div className="flex flex-col gap-2">
                    {openingHours.map((hour, index) => (
                      <div key={hour.id || index} className="flex justify-between text-sm py-2 border-b border-grey last:border-b-0">
                        <span className="font-semibold text-navy">{hour.day}</span>
                        <span className="text-grey-dark">{hour.hours}</span>
                      </div>
                    ))}
                  </div>
                </div>
              </div>
            )}
          </div>

          {/* Map */}
          {showMap && mapUrl && (
            <div className="bg-grey-light rounded-xl overflow-hidden h-96 md:h-auto min-h-96">
              <iframe
                src={mapUrl}
                width="100%"
                height="100%"
                style={{ border: 0 }}
                allowFullScreen
                loading="lazy"
                referrerPolicy="no-referrer-when-downgrade"
                title="Google Maps Location"
              ></iframe>
            </div>
          )}
        </div>
      </div>
    </section>
  )
}
Component Features
  • Responsive: 2-column grid on desktop, stacks on mobile
  • Icons: Lucide icons with teal glow background
  • Clickable Links: Phone (tel:) and email (mailto:) are automatically linked
  • Optional Map: Google Maps iframe only shown when showMap is true
  • Conditional Rendering: Only shows sections that have data

STEP 7 Troubleshooting Guide
⚠️ Common Issues & Solutions
❌ Map iframe not loading
Solution: Ensure you're using the embed URL from Google Maps, not the regular sharing link. Go to Google Maps → Share → "Embed a map" → Copy the iframe src URL. Should start with https://www.google.com/maps/embed?pb=...
❌ Map shows "This page can't load Google Maps correctly"
Solution: You may need a Google Maps API key for production. Add API key restriction to allow your domain. Alternatively, use the public embed URL without restrictions for simple embeds.
❌ Phone link not working on mobile
Solution: Ensure phone number is formatted correctly in tel: link. Remove all spaces, dashes, parentheses. Example: tel:+31201234567 instead of tel:020 - 123 45 67. The formatPhoneLink helper does this automatically.
❌ Opening hours not displaying
Solution: Check that openingHours array is populated and each entry has both day and hours fields. Verify conditional rendering: {hasOpeningHours && (...)} is working correctly.
❌ Address not showing line breaks
Solution: Ensure you're using <br /> JSX elements between address lines, not string concatenation. Check component code for proper conditional rendering of street and city on separate lines.
❌ Icons not rendering
Solution: Verify lucide-react is installed: npm install lucide-react. Check imports: import { MapPin, Phone, Mail, Clock } from 'lucide-react'.
❌ Map column empty when showMap is true
Solution: Check condition logic: {showMap && mapUrl && (...)} requires BOTH fields to be true/populated. If showMap is checked but mapUrl is empty, nothing will render. Add admin-side validation or show placeholder.
❌ Grid not stacking on mobile
Solution: Verify Tailwind responsive classes: grid md:grid-cols-2 defaults to 1 column, switches to 2 columns at md breakpoint (768px). Test with browser dev tools mobile viewport.
