STEP 1: Component Overview

Component Name: BB03: BundleItemRow

Category: Product Types › Bundle Builder

Version: 1.0.0

Last Updated: 25 Februari 2026

Purpose

The BundleItemRow component displays individual products that have been added to a custom bundle. It appears in the bundle sidebar/summary, showing product details, pricing, and a remove action. This component is the key visual representation of the user's bundle selection.

Key Features

  • Horizontal Layout: Compact row displaying emoji/icon, product info, price, and remove button
  • Text Truncation: Smart text overflow with ellipsis for long product names
  • Monospace Price: JetBrains Mono font for clear pricing display
  • Quick Remove: One-click removal with hover feedback (coral highlight)
  • Empty State: Visual placeholder when bundle is empty
  • Size Variants: Compact, default, and large sizes for different contexts
  • Accessibility: Full ARIA support with descriptive labels for screen readers

Visual Hierarchy

  1. Product Emoji: 36×36px icon (primary visual anchor)
  2. Product Name: 13px bold navy text (most prominent)
  3. Category Label: 11px grey text (secondary info)
  4. Price: 13px monospace (right-aligned, easy to scan)
  5. Remove Button: 24×24px button (subtle until hover)

STEP 2: When to Use

Primary Use Cases

  • Bundle Builder Sidebar: Main use case — displays selected products in the bundle summary panel
  • Shopping Cart Bundles: Shows bundle composition in cart/checkout (read-only variant)
  • Order Confirmation: Lists bundled products in order receipt
  • Account › My Bundles: Shows saved bundle configurations
  • Product Configurator: Displays selected components/modules

When NOT to Use

  • Full Product Cards: Use ProductCard (EC01) for grid/list layouts with images and full details
  • Simple Line Items: Use CartLineItem (EC06) for standard cart items without bundle context
  • Product Comparison: Use comparison table components for side-by-side product comparison
  • Horizontal Carousels: Use RecentlyViewed (C13) for horizontal scrolling product lists

Recommended Pairings

  • BB05: BundleTotalCalculator — Price breakdown below bundle items
  • BB06: BundleProgressBar — Progress indicator above bundle items
  • BB04: BundleDiscountTiers — Discount tiers hint between items and total
  • BB02: BundleProductCard — Selectable product cards that add to this list

Flexbox Layout Pattern

  • Container: display: flex; flex-direction: column; gap: 6px;
  • Row: display: flex; align-items: center; gap: 10px;
  • Emoji: flex-shrink: 0; (fixed width, doesn't shrink)
  • Info Section: flex: 1; min-width: 0; (grows to fill space, allows text overflow)
  • Price: flex-shrink: 0; white-space: nowrap; (fixed width, never wraps)
  • Remove Button: flex-shrink: 0; (fixed width)

Text Overflow Handling

Product names are truncated with ellipsis when too long:

.bi-name {
  white-space: nowrap;
  overflow: hidden;
  text-overflow: ellipsis;
}

Example: "IconPack Premium - 4.000+ vector iconen" → "IconPack Premium - 4.000..."

STEP 7: Implementation Notes

React/Next.js Component Example

'use client';

import { X, PackageOpen } from 'lucide-react';
import { useState } from 'react';

interface BundleItem {
  id: string;
  name: string;
  category: string;
  price: number;
  emoji?: string;
  image?: string;
}

interface BundleItemRowProps {
  item: BundleItem;
  onRemove: (id: string) => void;
  size?: 'compact' | 'default' | 'large';
  showCategory?: boolean;
  disabled?: boolean;
}

export function BundleItemRow({
  item,
  onRemove,
  size = 'default',
  showCategory = true,
  disabled = false,
}: BundleItemRowProps) {
  const handleRemove = () => {
    if (!disabled) {
      onRemove(item.id);
    }
  };

  return (
    <div className={`bi-item ${size !== 'default' ? size : ''}`}>
      <div className="bi-emoji">
        {item.emoji || '📦'}
      </div>
      <div className="bi-info">
        <div className="bi-name">{item.name}</div>
        {showCategory && (
          <div className="bi-cat">{item.category}</div>
        )}
      </div>
      <div className="bi-price">
        €{item.price.toFixed(2).replace('.', ',')}
      </div>
      <button
        className="bi-remove"
        onClick={handleRemove}
        disabled={disabled}
        aria-label={`Remove ${item.name} from bundle`}
        type="button"
      >
        <X size={14} aria-hidden="true" />
      </button>
    </div>
  );
}

interface BundleItemListProps {
  items: BundleItem[];
  onRemove: (id: string) => void;
  size?: 'compact' | 'default' | 'large';
  emptyStateText?: string;
}

export function BundleItemList({
  items,
  onRemove,
  size = 'default',
  emptyStateText = 'Voeg producten toe aan je bundel',
}: BundleItemListProps) {
  if (items.length === 0) {
    return (
      <div className="bi-empty" role="status" aria-live="polite">
        <PackageOpen size={24} aria-hidden="true" />
        {emptyStateText}
      </div>
    );
  }

  return (
    <div className="bi-list" role="list" aria-label="Bundle items">
      {items.map((item) => (
        <BundleItemRow
          key={item.id}
          item={item}
          onRemove={onRemove}
          size={size}
        />
      ))}
    </div>
  );
}

State Management with React

// Parent component managing bundle state
function BundleBuilder() {
  const [bundleItems, setBundleItems] = useState<BundleItem[]>([]);

  const handleRemoveItem = (itemId: string) => {
    setBundleItems(prev => prev.filter(item => item.id !== itemId));

    // Optional: Announce to screen readers
    const item = bundleItems.find(i => i.id === itemId);
    if (item) {
      announceToScreenReader(
        `${item.name} removed from bundle. ${bundleItems.length - 1} items remaining.`
      );
    }
  };

  const handleAddItem = (product: Product) => {
    const newItem: BundleItem = {
      id: crypto.randomUUID(),
      name: product.name,
      category: product.category,
      price: product.bundlePrice || product.price,
      emoji: product.emoji,
    };
    setBundleItems(prev => [...prev, newItem]);
  };

  return (
    <div className="bundle-sidebar">
      <BundleItemList
        items={bundleItems}
        onRemove={handleRemoveItem}
      />
    </div>
  );
}

Integration with Payload CMS

// Fetch bundle items from Payload CMS
async function getBundleItems(bundleId: string) {
  const bundle = await payload.findByID({
    collection: 'bundles',
    id: bundleId,
    depth: 2, // Populate product relationships
  });

  return bundle.bundleItems.map(item => ({
    id: item.id,
    name: item.product.name,
    category: item.category || item.product.category,
    price: item.price,
    emoji: item.emoji || item.product.emoji,
  }));
}

Price Formatting

// Utility function for consistent price formatting
function formatPrice(price: number, currency: string = 'EUR'): string {
  return new Intl.NumberFormat('nl-NL', {
    style: 'currency',
    currency,
  }).format(price);
}

// Usage in component
<div className="bi-price">

</div>
