📋 PC05: ConfiguratorPriceBreakdown

Component Type: Product Type Component — Configurator Pricing Summary
Category: Pricing Display / Call-to-Action
Complexity: High

💡 Purpose: Sticky pricing sidebar that shows a real-time breakdown of the configured product's price. Updates dynamically as users select options, displays financing info, and provides primary CTA to add to cart. Critical for transparency and conversion in high-value configurators.

STEP 1: Component Overview

What is ConfiguratorPriceBreakdown?

ConfiguratorPriceBreakdown is a sticky sidebar component that displays the current configuration's pricing in real-time. It shows a summary of selected options, itemized pricing breakdown, total cost with financing options, primary "Add to Cart" CTA, alternative actions (save/share/PDF), and trust signals to reduce purchase anxiety.

Key Features

  • Sticky positioning: Stays visible while scrolling (desktop only, top: 88px)
  • Configuration summary: Visual display of selected options with icons + price deltas
  • Itemized pricing: Base price + upgrades + extras → total
  • Financing calculator: Monthly payment option (e.g., €112/mnd for 24 months)
  • Primary CTA: Teal gradient "Add to Cart" button with shopping cart icon
  • Alternative actions: Save configuration, share with others, download PDF
  • Trust signals: Free shipping, returns, warranty, payment options
  • Dynamic updates: Real-time recalculation as user selects options
  • Color indicator: Visual color dot for selected color option

Component Anatomy

  1. Header: Dark gradient bg, product name (emoji + title), subtitle
  2. Configuration Summary: Rows of selected options (icon + label + value + price)
  3. Pricing Breakdown: Line items (base + upgrades + extras)
  4. Total: Large bold total with optional monthly financing
  5. Primary CTA: Teal gradient button (52px height, full-width)
  6. Alternative Actions: 3 outline buttons (save, share, PDF)
  7. Trust Signals: 4 green-checkmark badges (shipping, returns, warranty, payment)

STEP 2: When to Use This Component

Primary Use Cases

  • Product configurators: Multi-step customization with pricing impact
  • High-value products: €500+ where transparency builds trust
  • Complex pricing: Base price + multiple upgrades + optional extras
  • Financing offers: Products with monthly payment options
  • B2B configurators: Quote generation, save/share functionality

When NOT to Use

  • Simple products: Single price, no options (use basic product page)
  • Low-value items: < €50 where itemized breakdown is overkill
  • Mobile-first flows: Limited screen space (use simplified version)
  • Services: Non-physical products without clear pricing tiers

Component Pairing

ConfiguratorPriceBreakdown works best when combined with:

  • PC03: ConfiguratorOptionSelector — User selects options → pricing updates
  • PC04: ConfiguratorPreview — Visual preview (left) + pricing (right) sticky layout
  • PC02: ConfiguratorStepNav — Shows progress, pricing shows cumulative cost

// utils/formatPrice.ts
export function formatPrice(cents: number): string {
  const euros = cents / 100;
  return new Intl.NumberFormat('nl-NL', {
    style: 'currency',
    currency: 'EUR',
  }).format(euros);
}

// Usage
formatPrice(269800);  // "€ 2.698,00"
formatPrice(60000);   // "€ 600,00"

// components/ConfiguratorPriceBreakdown.tsx
import React, { useMemo } from 'react';
import * as Icons from 'lucide-react';
import { formatPrice } from '@/utils/formatPrice';

interface ConfigOption {
  id: string;
  label: string;
  icon: string;
  value: string;
  price?: string;
  colorValue?: string;
}

interface PriceLine {
  id: string;
  label: string;
  amount: string;
  type?: 'base' | 'upgrade' | 'extra' | 'discount';
}

interface TrustSignal {
  icon: string;
  text: string;
}

interface ConfiguratorPriceBreakdownProps {
  productName: string;
  productEmoji?: string;
  subtitle?: string;
  selectedOptions: ConfigOption[];
  priceLines: PriceLine[];
  total: number;
  monthlyPayment?: { amount: number; months: number };
  onAddToCart: () => void;
  onSave?: () => void;
  onShare?: () => void;
  onDownloadPDF?: () => void;
  trustSignals?: TrustSignal[];
  sticky?: boolean;
  stickyTop?: string;
  showAlternativeActions?: boolean;
  showTrustSignals?: boolean;
  showMonthlyPayment?: boolean;
}

export function ConfiguratorPriceBreakdown({
  productName,
  productEmoji,
  subtitle = 'Je huidige configuratie',
  selectedOptions,
  priceLines,
  total,
  monthlyPayment,
  onAddToCart,
  onSave,
  onShare,
  onDownloadPDF,
  trustSignals = [],
  sticky = true,
  stickyTop = '88px',
  showAlternativeActions = true,
  showTrustSignals = true,
  showMonthlyPayment = true,
}: ConfiguratorPriceBreakdownProps) {

  const formattedTotal = useMemo(() => formatPrice(total), [total]);
  const formattedMonthly = useMemo(() => {
    if (!monthlyPayment) return null;
    return {
      amount: formatPrice(monthlyPayment.amount),
      months: monthlyPayment.months,
    };
  }, [monthlyPayment]);

  return (
    <aside
      className="summary-card"
      style={{ position: sticky ? 'sticky' : 'static', top: stickyTop }}
      role="complementary"
      aria-label="Prijs en configuratie samenvatting"
    >
      <div className="sum-header">
        <div className="sum-h-title">
          {productEmoji && <span>{productEmoji} </span>}
          {productName}
        </div>
        <div className="sum-h-sub">{subtitle}</div>
      </div>

      <div className="sum-body">
        {/* Configuration summary */}
        <div className="sum-config" aria-label="Geselecteerde opties">
          {selectedOptions.map((option) => {
            const IconComponent = Icons[option.icon as keyof typeof Icons] as React.ElementType;

            return (
              <div key={option.id} className="sc-row">
                <div className="sc-label">
                  <IconComponent size={14} aria-hidden="true" /> {option.label}
                </div>
                <div className="sc-value">
                  {option.colorValue && (
                    <div
                      className="sc-dot"
                      style={{ background: option.colorValue }}
                      aria-hidden="true"
                    />
                  )}
                  <span>{option.value}</span>
                  {option.price && <span className="sc-price">{option.price}</span>}
                </div>
              </div>
            );
          })}
        </div>

        {/* Pricing breakdown */}
        <div className="sum-pricing" role="region" aria-label="Prijsoverzicht" aria-live="polite">
          {priceLines.map((line) => (
            <div
              key={line.id}
              className={`sp-line ${line.type === 'extra' || line.type === 'discount' ? 'extra' : ''}`}
              style={line.type === 'discount' ? { color: 'var(--green)' } : {}}
            >
              <span>{line.label}</span>
              <span>{line.amount}</span>
            </div>
          ))}

          <div className="sp-total">
            <span>Totaal</span>
            <div>
              <div>{formattedTotal}</div>
              {showMonthlyPayment && formattedMonthly && (
                <div className="sp-monthly">
                  of {formattedMonthly.amount}/mnd ({formattedMonthly.months}×)
                </div>
              )}
            </div>
          </div>
        </div>

        {/* Primary CTA */}
        <button className="sum-btn" onClick={onAddToCart}>
          <Icons.ShoppingCart size={18} />
          Toevoegen aan winkelwagen
        </button>

        {/* Alternative actions */}
        {showAlternativeActions && (
          <div className="sum-alt" role="toolbar" aria-label="Configuratie acties">
            {onSave && (
              <button className="sum-alt-btn" onClick={onSave}>
                <Icons.Heart size={13} /> Opslaan
              </button>
            )}
            {onShare && (
              <button className="sum-alt-btn" onClick={onShare}>
                <Icons.Share2 size={13} /> Delen
              </button>
            )}
            {onDownloadPDF && (
              <button className="sum-alt-btn" onClick={onDownloadPDF}>
                <Icons.FileText size={13} /> PDF
              </button>
            )}
          </div>
        )}

        {/* Trust signals */}
        {showTrustSignals && trustSignals.length > 0 && (
          <div className="sum-trust" role="list" aria-label="Voordelen">
            {trustSignals.map((signal, index) => {
              const IconComponent = Icons[signal.icon as keyof typeof Icons] as React.ElementType;

              return (
                <div key={index} className="str" role="listitem">
                  <IconComponent size={13} aria-hidden="true" />
                  {signal.text}
                </div>
              );
            })}
          </div>
        )}
      </div>
    </aside>
  );
}

Dynamic Price Calculation

// Parent component managing configurator state
function ProductConfigurator() {
  const [config, setConfig] = useState({
    model: { id: 'probook-16', basePrice: 159900 },
    color: 'space-grey',
    processor: { id: 'm4-pro', price: 60000 },
    ram: { id: '36gb', price: 20000 },
    storage: { id: '512gb', price: 0 },
    extras: [
      { id: 'applecare', label: 'AppleCare+ (3 jaar)', price: 29900 },
    ],
  });

  const total = useMemo(() => {
    return (
      config.model.basePrice +
      config.processor.price +
      config.ram.price +
      config.storage.price +
      config.extras.reduce((sum, extra) => sum + extra.price, 0)
    );
  }, [config]);

  const monthlyPayment = useMemo(() => {
    const monthlyAmount = Math.round(total / 24);
    return { amount: monthlyAmount, months: 24 };
  }, [total]);

  const priceLines: PriceLine[] = [
    { id: 'base', label: `Basisprijs ${config.model.id}`, amount: formatPrice(config.model.basePrice), type: 'base' },
    { id: 'cpu', label: 'M4 Pro upgrade', amount: formatPrice(config.processor.price), type: 'upgrade' },
    { id: 'ram', label: '36 GB RAM upgrade', amount: formatPrice(config.ram.price), type: 'upgrade' },
    ...config.extras.map(extra => ({
      id: extra.id,
      label: extra.label,
      amount: formatPrice(extra.price),
      type: 'extra' as const,
    })),
  ];

  return (
    <ConfiguratorPriceBreakdown
      productName="ProBook Ultra 16\""
      productEmoji="💻"
      selectedOptions={[...]}
      priceLines={priceLines}
      total={total}
      monthlyPayment={monthlyPayment}
      onAddToCart={handleAddToCart}
    />
  );
}
