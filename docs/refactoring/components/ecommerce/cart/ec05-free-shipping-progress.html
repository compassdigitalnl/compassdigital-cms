Description: Visual progress bar that shows customers how much more they need to spend to qualify for free shipping. Encourages larger cart values through clear visual feedback.

Key Features:

  • Real-time progress calculation based on cart total
  • Teal gradient progress bar with smooth animation
  • Truck icon indicator (changes to check-circle when achieved)
  • Remaining amount highlighted in teal
  • Success state with green color scheme
  • Threshold amount display
  • Responsive layout

🎯 STEP 2: When to Use

Use this component when:

  • You offer free shipping above a certain order value
  • You want to encourage customers to add more items to their cart
  • Displaying cart summary or cart page
  • You need to show progress toward a shipping threshold
  • Implementing upselling strategies

Common Design Patterns:

  • Cart Page: Prominent display at top of cart items
  • Mini Cart Flyout: Condensed version in cart dropdown
  • Checkout Page: Reminder in order summary sidebar
  • Product Page: Preview in sticky add-to-cart bar

Do NOT use when:

  • Free shipping is always offered (no threshold)
  • Cart is empty (nothing to progress toward)
  • Shipping is not applicable (digital products only)
  • Threshold has been exceeded (show success state instead)

💻 STEP 7: Implementation Notes

React/Next.js Component

'use client'

import { useMemo } from 'react'
import { Truck, CheckCircle } from 'lucide-react'

interface FreeShippingProgressProps {
  cartTotal: number
  threshold: number
  currencySymbol?: string
  locale?: string
  showEmoji?: boolean
}

export function FreeShippingProgress({
  cartTotal,
  threshold,
  currencySymbol = '€',
  locale = 'nl-NL',
  showEmoji = true
}: FreeShippingProgressProps) {

  const { remaining, percentage, achieved } = useMemo(() => {
    const remaining = Math.max(threshold - cartTotal, 0)
    const percentage = Math.min((cartTotal / threshold) * 100, 100)
    const achieved = cartTotal >= threshold

    return { remaining, percentage, achieved }
  }, [cartTotal, threshold])

  const formatCurrency = (amount: number) => {
    return new Intl.NumberFormat(locale, {
      style: 'currency',
      currency: locale === 'nl-NL' ? 'EUR' : 'USD',
      minimumFractionDigits: 2
    }).format(amount)
  }

  const Icon = achieved ? CheckCircle : Truck

  return (
    <div
      className={`free-shipping-progress ${
        achieved ? 'free-shipping-progress--achieved' : ''
      }`}
      role="status"
      aria-live="polite"
      aria-atomic="true"
    >
      <Icon
        className="free-shipping-progress__icon"
        aria-hidden="true"
      />
      <div className="free-shipping-progress__content">
        <div className="free-shipping-progress__text">
          {achieved ? (
            <>
              Je bestelling komt in aanmerking voor gratis verzending!{' '}
              {showEmoji && '🎉'}
            </>
          ) : (
            <>
              Nog{' '}
              <span className="free-shipping-progress__amount">
                {formatCurrency(remaining)}
              </span>{' '}
              tot gratis verzending!
            </>
          )}
        </div>
        <div
          className="free-shipping-progress__bar-wrapper"
          role="progressbar"
          aria-valuenow={percentage}
          aria-valuemin={0}
          aria-valuemax={100}
          aria-label={`Voortgang gratis verzending: ${percentage.toFixed(1)}%`}
        >
          <div
            className="free-shipping-progress__bar-fill"
            style={{ width: `${percentage}%` }}
          />
        </div>
        <div className="free-shipping-progress__threshold">
          {achieved ? (
            'Gratis verzending toegepast bij checkout'
          ) : (
            `Bij bestellingen vanaf ${formatCurrency(threshold)} gratis verzending`
          )}
        </div>
      </div>
    </div>
  )
}

Important Implementation Details

  • Calculation Logic: Always calculate remaining = threshold - cartTotal, clamped to 0 minimum
  • Percentage: (cartTotal / threshold) * 100, clamped to 100 maximum
  • Smooth Animation: Use transition: width 0.5s ease on progress bar fill
  • Currency Formatting: Use Intl.NumberFormat for proper locale formatting
  • Icon Swap: Change from truck to check-circle when achieved
  • State Persistence: Progress is derived from cart total, no separate state needed
  • Server Components: Can be server component if cart data is available server-side

Cart Context Integration

// In CartContext or cart hook
export function useCartProgress(threshold: number = 60) {
  const { items } = useCart()

  const cartTotal = useMemo(() =>
    items.reduce((sum, item) => sum + (item.price * item.quantity), 0),
    [items]
  )

  const remaining = Math.max(threshold - cartTotal, 0)
  const percentage = Math.min((cartTotal / threshold) * 100, 100)
  const achieved = cartTotal >= threshold

  return { cartTotal, remaining, percentage, achieved }
}

// Usage
const { cartTotal, threshold } = useCartProgress()
<FreeShippingProgress cartTotal={cartTotal} threshold={threshold} />
