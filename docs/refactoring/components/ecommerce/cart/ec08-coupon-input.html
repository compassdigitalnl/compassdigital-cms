Description: Input field with submit button for applying discount/coupon codes. Includes validation, loading states, error/success feedback, and applied coupon display.

Key Features:

  • Text input with placeholder
  • Submit button ("Toepassen")
  • Loading state during validation
  • Error feedback (invalid/expired code)
  • Success feedback with applied discount amount
  • Applied coupon display with remove button
  • Uppercase auto-formatting (optional)

🎯 STEP 2: When to Use

Use this component when:

  • Embedding in EC07: OrderSummary sidebar
  • On cart page for discount application
  • In checkout flow for last-minute discounts
  • In mini cart flyout (condensed version)

Common Design Patterns:

  • Cart Sidebar: Above checkout CTA buttons
  • Checkout: In payment step sidebar
  • Mini Cart: Collapsible "Apply discount" section

Do NOT use when:

  • Cart is empty (hide component)
  • On product pages (use direct "Apply at checkout" messaging)
  • For gift cards (use separate gift card input)

💻 STEP 7: Implementation Notes

React/Next.js Component

'use client'

import { useState } from 'react'
import { CheckCircle, XCircle, X } from 'lucide-react'

export function CouponInput({
  onApply,
  onRemove,
  appliedCoupon,
  placeholder = 'Kortingscode',
  buttonText = 'Toepassen',
  autoUppercase = true
}: CouponInputProps) {

  const [code, setCode] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!code.trim() || loading) return

    setLoading(true)
    setError(null)

    try {
      const result = await onApply?.(code.trim())

      if (result?.success) {
        setCode('')
        setError(null)
      } else {
        setError(result?.message || 'Deze kortingscode is ongeldig')
      }
    } catch (err) {
      setError('Er is iets misgegaan. Probeer het opnieuw.')
    } finally {
      setLoading(false)
    }
  }

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    let value = e.target.value
    if (autoUppercase) {
      value = value.toUpperCase()
    }
    setCode(value)
    setError(null) // Clear error on input
  }

  // If coupon is applied, show applied state
  if (appliedCoupon) {
    return (
      <div className="coupon-applied">
        <div className="coupon-applied__left">
          <CheckCircle />
          <div className="coupon-applied__text">
            <span className="coupon-applied__code">
              {appliedCoupon.code}
            </span>
            <span className="coupon-applied__discount">
              −€ {appliedCoupon.discountAmount.toFixed(2)} korting toegepast
            </span>
          </div>
        </div>
        <button
          className="coupon-applied__remove"
          onClick={onRemove}
          aria-label="Verwijder kortingscode"
        >
          <X />
        </button>
      </div>
    )
  }

  return (
    <div>
      <form onSubmit={handleSubmit} className="coupon-input">
        <input
          type="text"
          className={`coupon-input__field ${
            error ? 'coupon-input__field--error' : ''
          }`}
          placeholder={placeholder}
          value={code}
          onChange={handleChange}
          disabled={loading}
          aria-label={placeholder}
          aria-invalid={!!error}
          aria-describedby={error ? 'coupon-error' : undefined}
        />
        <button
          type="submit"
          className={`coupon-input__btn ${
            loading ? 'coupon-input__btn--loading' : ''
          }`}
          disabled={loading || !code.trim()}
        >
          {buttonText}
        </button>
      </form>

      {error && (
        <div
          id="coupon-error"
          className="coupon-feedback coupon-feedback--error"
          role="alert"
        >
          <XCircle size={14} />
          {error}
        </div>
      )}
    </div>
  )
}

API Endpoint Example

// app/api/cart/apply-coupon/route.ts
export async function POST(req: Request) {
  const { code, cartTotal } = await req.json()

  const coupon = await payload.find({
    collection: 'coupons',
    where: {
      code: { equals: code.toUpperCase() },
      isActive: { equals: true }
    }
  })

  if (!coupon.docs.length) {
    return Response.json({
      success: false,
      message: 'Deze kortingscode is ongeldig'
    }, { status: 400 })
  }

  const c = coupon.docs[0]

  // Check expiry
  if (c.validUntil && new Date(c.validUntil) < new Date()) {
    return Response.json({
      success: false,
      message: 'Deze kortingscode is verlopen'
    }, { status: 400 })
  }

  // Check min order value
  if (c.minOrderValue && cartTotal < c.minOrderValue) {
    return Response.json({
      success: false,
      message: `Minimum bestelbedrag voor deze code is € ${c.minOrderValue.toFixed(2)}`
    }, { status: 400 })
  }

  // Calculate discount
  const discount = c.discountType === 'percentage'
    ? cartTotal * (c.discountValue / 100)
    : c.discountValue

  // Increment usage count
  await payload.update({
    collection: 'coupons',
    id: c.id,
    data: { usedCount: (c.usedCount || 0) + 1 }
  })

  return Response.json({
    success: true,
    discount,
    code: c.code
  })
}

Important Implementation Details

  • Auto-Uppercase: Convert input to uppercase on change
  • Debouncing: Not needed (submit-based, not live search)
  • Loading State: Disable input/button, show spinner
  • Error Clearing: Clear error on new input
  • Form Submission: Use <form> for Enter key support
  • Validation: Server-side validation is critical (check expiry, usage limits, min order)
