📋 STEP 1: COMPONENT OVERVIEW

Component: C1 - Instant Search Overlay

Category: E-commerce / Navigation

Complexiteit: High (overlay, keyboard nav, live search, result grouping)

Purpose:

Full-screen search overlay with live results grouped by type (products, categories, brands, knowledge base). Provides instant feedback as users type, with keyboard navigation support.

Key Features:

  • Full-screen overlay with backdrop blur
  • Live search with instant results (debounced API calls)
  • Grouped results: Products, Categories, Brands, Knowledge
  • Keyboard shortcuts (⌘K to open, ESC to close, ↑↓ to navigate)
  • Product result cards with image, brand, name, SKU, price, stock
  • Highlight matching text in results
  • Footer with keyboard hints and "View all" link
  • Smooth animations (slide up, fade in)

🎯 STEP 2: WHEN TO USE

Ideaal voor:

  • E-commerce sites met grote productcatalogi (500+ producten)
  • B2B platforms met SKU/artikel nummer zoeken
  • Sites met multiple content types (producten + kennisbank + merken)
  • Power users die keyboard shortcuts waarderen
  • Modern, app-like interfaces (SaaS style)

Niet geschikt voor:

  • Simpele sites met <50 producten (gebruik dropdown search)
  • Mobile-first sites (overlay is desktop-oriented)
  • Sites waar search niet primary navigation is
  • Older audiences niet gewend aan keyboard shortcuts

Best Practices:

  • Open with ⌘K (Mac) / Ctrl+K (Windows) keyboard shortcut
  • Show minimum 3 characters before searching (performance)
  • Debounce API calls (300ms wait after typing stops)
  • Group results by relevance: Products first, then categories, brands, content
  • Show max 4-5 results per group, link to "View all" for more
  • Highlight matching search terms in results

import { useState, useEffect, useRef } from 'react';
import { Search, X, Package, Folder, Award, BookOpen } from 'lucide-react';

interface SearchResult {
  products: Product[];
  categories: Category[];
  brands: Brand[];
  articles: Article[];
}

export function InstantSearch() {
  const [isOpen, setIsOpen] = useState(false);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<SearchResult | null>(null);
  const [activeIndex, setActiveIndex] = useState(0);
  const inputRef = useRef<HTMLInputElement>(null);

  // Global keyboard shortcut
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        setIsOpen(true);
      }
      if (e.key === 'Escape' && isOpen) {
        setIsOpen(false);
      }
    };

    document.addEventListener('keydown', handleKeyDown);
    return () => document.removeEventListener('keydown', handleKeyDown);
  }, [isOpen]);

  // Auto-focus input
  useEffect(() => {
    if (isOpen && inputRef.current) {
      setTimeout(() => inputRef.current?.focus(), 100);
    }
  }, [isOpen]);

  // Debounced search
  useEffect(() => {
    if (query.length < 3) {
      setResults(null);
      return;
    }

    const timer = setTimeout(async () => {
      const res = await fetch(`/api/search?q=${encodeURIComponent(query)}`);
      const data = await res.json();
      setResults(data.results);
    }, 300);

    return () => clearTimeout(timer);
  }, [query]);

  // Arrow navigation
  const handleKeyDown = (e: KeyboardEvent) => {
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      setActiveIndex(i => i + 1);
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      setActiveIndex(i => Math.max(0, i - 1));
    }
    if (e.key === 'Enter') {
      // Navigate to active result
    }
  };

  return (
    <div className={`is-overlay ${isOpen ? 'open' : ''}`}>
      <div className="is-backdrop" onClick={() => setIsOpen(false)} />
      <div className="is-panel">
        <div className="is-input-wrap">
          <Search size={20} />
          <input
            ref={inputRef}
            className="is-input"
            type="search"
            placeholder="Zoek producten, merken of artikelen…"
            value={query}
            onChange={e => setQuery(e.target.value)}
            onKeyDown={handleKeyDown}
          />
          <span className="is-esc">ESC</span>
          <button className="is-close" onClick={() => setIsOpen(false)}>
            <X size={16} />
          </button>
        </div>

        <div className="is-body">
          {results?.products && (
            <ProductResults products={results.products} query={query} />
          )}
          {/* Categories, Brands, Articles... */}
        </div>

        <div className="is-footer">
          <div className="is-footer-left">
            <kbd>↑</kbd> <kbd>↓</kbd> navigeren · <kbd>↵</kbd> openen
          </div>
          <a href={`/zoeken?q=${query}`} className="is-footer-all">
            Alle resultaten
          </a>
        </div>
      </div>
    </div>
  );
}

Highlight Matches:

function highlightMatch(text: string, query: string) {
  const regex = new RegExp(`(${query})`, 'gi');
  return text.replace(regex, '<mark>$1</mark>');
}

// Example: Next.js API Route
export default async function handler(req, res) {
  const { q } = req.query;

  if (!q || q.length < 3) {
    return res.json({ results: null });
  }

  // Search products (fuzzy match on name, SKU, brand)
  const products = await db.product.findMany({
    where: {
      OR: [
        { name: { contains: q, mode: 'insensitive' } },
        { sku: { contains: q, mode: 'insensitive' } },
        { brand: { name: { contains: q, mode: 'insensitive' } } }
      ]
    },
    take: 5,
    include: { brand: true }
  });

  // Search categories, brands, articles...
  const categories = await db.category.findMany({ ... });
  const brands = await db.brand.findMany({ ... });
  const articles = await db.article.findMany({ ... });

  res.json({
    query: q,
    total: products.length + categories.length + brands.length + articles.length,
    results: { products, categories, brands, articles }
  });
}

// Global search endpoint
import { Payload } from 'payload';

export async function globalSearch(query: string) {
  const payload = await getPayload();

  // Search across multiple collections
  const [products, posts, pages] = await Promise.all([
    payload.find({
      collection: 'products',
      where: {
        or: [
          { name: { contains: query } },
          { sku: { contains: query } }
        ]
      },
      limit: 5
    }),
    payload.find({
      collection: 'posts',
      where: { title: { contains: query } },
      limit: 3
    }),
    payload.find({
      collection: 'pages',
      where: { title: { contains: query } },
      limit: 3
    })
  ]);

  return {
    products: products.docs,
    articles: [...posts.docs, ...pages.docs]
  };
}
