Purpose

Shipping address form for checkout flow (step 1: Gegevens). Collects delivery address with validation, postcode autocomplete (NL), and saved addresses dropdown for B2B customers. Provides clear visual feedback for required fields, validation errors, and successful input.

Visual Pattern

  • White card with border + rounded corners (14px radius)
  • Section header with icon (map-pin) + title
  • 2-column grid layout (stacks to 1-column on mobile)
  • Form fields with labels + required indicators (red asterisks)
  • Input focus state: teal border + glow ring
  • Error state: coral border + error message below
  • Success state: green border (after validation)

Key Features

  • ✅ 7 address fields (firstName, lastName, street, addition, postalCode, city, country)
  • ✅ Required field indicators (red asterisk)
  • ✅ Validation states (error, success)
  • ✅ Error messages below invalid fields
  • ✅ Postcode autocomplete (NL format: "1234 AB")
  • ✅ Saved addresses dropdown (B2B only)
  • ✅ Helper text for optional fields
  • ✅ Country dropdown (NL, BE, DE, FR, UK)
  • ✅ Responsive layout (2-col → 1-col on mobile)
  • ✅ Fully keyboard accessible

STEP 7: Implementation Notes

React/Next.js Component

'use client';

import { useState } from 'react';
import { MapPin, Bookmark, Sparkles } from 'lucide-react';

export function AddressForm({ onSubmit, savedAddresses = [], showSavedAddresses = false }: AddressFormProps) {
  const [formData, setFormData] = useState<Address>({
    firstName: '',
    lastName: '',
    street: '',
    addition: '',
    postalCode: '',
    city: '',
    country: 'NL',
  });

  const [errors, setErrors] = useState<Record<string, string>>({});
  const [autocompleteLoading, setAutocompleteLoading] = useState(false);

  // Postcode autocomplete (NL only)
  const handlePostalCodeChange = async (postalCode: string) => {
    setFormData(prev => ({ ...prev, postalCode }));

    if (formData.country === 'NL' && /^\d{4}\s?[A-Z]{2}$/i.test(postalCode)) {
      setAutocompleteLoading(true);
      try {
        const response = await fetch(`/api/postcode-lookup?postal=${postalCode}`);
        const data = await response.json();
        setFormData(prev => ({
          ...prev,
          city: data.city,
          street: data.street || prev.street,
        }));
      } catch (error) {
        console.error('Postcode lookup failed:', error);
      } finally {
        setAutocompleteLoading(false);
      }
    }
  };

  // Load saved address
  const loadSavedAddress = (addressId: string) => {
    const saved = savedAddresses.find(a => a.id === addressId);
    if (saved) {
      setFormData(saved.address);
    }
  };

  // Validate form
  const validate = (): boolean => {
    const newErrors: Record<string, string> = {};

    if (!formData.firstName) newErrors.firstName = 'Voornaam is verplicht';
    if (!formData.lastName) newErrors.lastName = 'Achternaam is verplicht';
    if (!formData.street) newErrors.street = 'Straat + huisnummer is verplicht';
    if (!formData.postalCode) {
      newErrors.postalCode = 'Postcode is verplicht';
    } else if (formData.country === 'NL' && !/^\d{4}\s?[A-Z]{2}$/i.test(formData.postalCode)) {
      newErrors.postalCode = 'Voer een geldige Nederlandse postcode in (bijv. 1234 AB)';
    }
    if (!formData.city) newErrors.city = 'Plaats is verplicht';

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  // Submit form
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (validate()) {
      onSubmit(formData);
    }
  };

  return (
    <div className="form-section">
      <h2 className="form-section-title">
        <MapPin size={20} />
        Afleveradres
      </h2>

      {showSavedAddresses && savedAddresses.length > 0 && (
        <div className="saved-addresses">
          <Bookmark size={20} />
          <select onChange={(e) => loadSavedAddress(e.target.value)}>
            <option value="">Selecteer opgeslagen adres...</option>
            {savedAddresses.map((addr) => (
              <option key={addr.id} value={addr.id}>{addr.label}</option>
            ))}
            <option value="new">➕ Nieuw adres toevoegen</option>
          </select>
        </div>
      )}

      <form onSubmit={handleSubmit} className="form-grid" noValidate>
        {/* Fields here... */}
      </form>
    </div>
  );
}

Postcode Autocomplete API

// /api/postcode-lookup/route.ts

export async function GET(req: Request) {
  const { searchParams } = new URL(req.url);
  const postalCode = searchParams.get('postal');

  if (!postalCode) {
    return Response.json({ error: 'Postcode is required' }, { status: 400 });
  }

  try {
    // Use Pro6PP, Postcode.nl, or similar service
    const response = await fetch(
      `https://api.pro6pp.nl/v2/autocomplete/nl/${postalCode}`,
      {
        headers: { 'Authorization': `Bearer ${process.env.POSTCODE_API_KEY}` },
      }
    );

    const data = await response.json();

    return Response.json({
      city: data.city,
      street: data.street,
      municipality: data.municipality,
    });
  } catch (error) {
    return Response.json({ error: 'Lookup failed' }, { status: 500 });
  }
}
