Purpose

Payment method selection cards for the checkout flow. Each card represents a different payment option (iDEAL, Credit Card, PayPal, Invoice) with radio button selection. Provides clear visual feedback for the selected payment method and displays relevant payment logos and descriptions.

Visual Pattern

  • 2-column grid layout (stacks to 1-column on mobile)
  • White card with 2px border (teal when selected, grey when inactive)
  • Horizontal flex layout: radio (left) → info (center) → logo (right)
  • Active selection indicated by teal border
  • Payment logo badge (48×32px) with grey background
  • Optional B2B badge for invoice payment method

Key Features

  • ✅ Single selection (radio buttons)
  • ✅ Clear visual feedback (border color change)
  • ✅ 4 payment methods: iDEAL, Credit Card, PayPal, Invoice
  • ✅ Payment logos (emoji placeholders, swap with real logos in production)
  • ✅ B2B badge for invoice option
  • ✅ Hover state (border color + subtle shadow)
  • ✅ Focus state (outline ring)
  • ✅ Fully keyboard accessible
  • ✅ Mobile responsive (stacks to single column)

STEP 2: When to Use

✅ USE this component when:

  • ✅ Checkout flow — step 3: payment method selection
  • ✅ Subscription signup — recurring payment method
  • ✅ Account settings — update payment method
  • ✅ B2B quote requests — payment preference selection
  • ✅ Order edit — change payment method before confirmation

❌ DON'T use this component when:

  • ❌ Quick order forms — use simpler dropdown instead
  • ❌ Guest checkout with single payment option — no selection needed
  • ❌ Mobile apps — use native payment UI
  • ❌ Admin panel — use standard form inputs

STEP 7: Implementation Notes

React/Next.js Component

'use client';

import { useState } from 'react';

interface PaymentMethod {
  method: string;
  name: string;
  description: string;
  logo: string | React.ReactNode;
  isB2B?: boolean;
  disabled?: boolean;
}

export function PaymentMethodCard({
  method,
  name,
  description,
  logo,
  isB2B = false,
  disabled = false,
  selectedMethod,
  onMethodChange,
}: PaymentMethod & {
  selectedMethod: string;
  onMethodChange: (method: string) => void;
}) {
  return (
    <label
      className={`payment-method-card ${disabled ? 'disabled' : ''}`}
    >
      <input
        type="radio"
        name="payment"
        value={method}
        checked={selectedMethod === method}
        onChange={(e) => onMethodChange(e.target.value)}
        disabled={disabled}
      />
      <div className="payment-info">
        <div className="payment-name">
          {name}
          {isB2B && <span className="b2b-badge">B2B</span>}
        </div>
        <div className="payment-description">{description}</div>
      </div>
      <div className="payment-logo" aria-hidden="true">
        {logo}
      </div>
    </label>
  );
}

export function PaymentMethodsGrid() {
  const [selectedMethod, setSelectedMethod] = useState('ideal');

  const methods: PaymentMethod[] = [
    {
      method: 'ideal',
      name: 'iDEAL',
      description: 'Direct betalen via je bank',
      logo: '🏦',
    },
    {
      method: 'creditcard',
      name: 'Credit Card',
      description: 'Visa, Mastercard, Amex',
      logo: '💳',
    },
    {
      method: 'paypal',
      name: 'PayPal',
      description: 'Betaal via PayPal',
      logo: <span className="paypal-logo">P</span>,
    },
    {
      method: 'invoice',
      name: 'Factuur',
      description: 'Betaal binnen 14 dagen',
      logo: '📄',
      isB2B: true,
    },
  ];

  return (
    <div className="payment-methods-grid">
      {methods.map((method) => (
        <PaymentMethodCard
          key={method.method}
          {...method}
          selectedMethod={selectedMethod}
          onMethodChange={setSelectedMethod}
        />
      ))}
    </div>
  );
}

Form Validation

// Ensure a payment method is selected before proceeding

const validateCheckout = (formData: CheckoutFormData) => {
  const errors: Record<string, string> = {};

  if (!formData.paymentMethod) {
    errors.paymentMethod = 'Selecteer een betaalmethode';
  }

  // B2B validation: invoice only for verified business accounts
  if (formData.paymentMethod === 'invoice' && !user.isB2B) {
    errors.paymentMethod = 'Factuur betaling is alleen beschikbaar voor zakelijke klanten';
  }

  return errors;
};

Payment Gateway Integration

// Stripe/Mollie integration based on selected method

const createPaymentIntent = async (order: Order) => {
  switch (order.paymentMethod) {
    case 'ideal':
      return await mollie.payments.create({
        amount: { value: order.total, currency: 'EUR' },
        method: 'ideal',
        description: `Order #${order.orderNumber}`,
        redirectUrl: `${siteUrl}/order-confirmation/${order.id}`,
      });

    case 'creditcard':
      return await stripe.paymentIntents.create({
        amount: order.total * 100, // cents
        currency: 'eur',
        payment_method_types: ['card'],
        metadata: { orderId: order.id },
      });

    case 'paypal':
      return await paypal.orders.create({
        purchase_units: [{ amount: { value: order.total } }],
        application_context: {
          return_url: `${siteUrl}/order-confirmation/${order.id}`,
        },
      });

    case 'invoice':
      // Create invoice, send email, set payment status to 'pending'
      return await createInvoice(order);
  }
};
