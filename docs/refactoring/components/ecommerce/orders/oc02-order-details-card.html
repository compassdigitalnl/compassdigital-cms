STEP 1: Component Overview

Purpose

The OrderDetailsCard component displays three key order metrics in a visually balanced grid layout on the order confirmation page. Each card shows a specific aspect of the order: delivery timeline, payment status, and total amount. The cards use color-coded icons and prominent values to create quick visual scanning.

Component Identity

  • Name: OrderDetailsCard
  • ID: OC02
  • Category: UI Elements / Stats Cards
  • Type: Informational display (read-only)
  • Priority: High (Phase 2)

Usage Context

Appears immediately below the success hero (OC01) on the order confirmation page. Provides at-a-glance order information before the detailed order items summary (OC03). Essential for customer reassurance after completing a purchase.

Key Features

  • 3-column responsive grid (stacks to 1-column on mobile)
  • Color-coded icon backgrounds (teal, green, navy)
  • Large bold values (20px, weight 800) for quick scanning
  • Subtle hover animation (lift + shadow)
  • Flexible content (delivery time, payment method, total amount)
  • Status badge variant for detailed payment states

STEP 2: When to Use This Component

Primary Use Cases

  1. Order Confirmation Page: Display key order details after successful checkout
  2. Order Status Dashboard: Show current order state in account area
  3. Email Confirmation Template: Inline version for order confirmation emails
  4. Admin Order View: Quick overview panel in admin order detail page

When NOT to Use

  • Checkout flow: Use EC07 OrderSummary instead (vertical layout, more detailed)
  • Invoice pages: Use dedicated invoice layout with full line items
  • Product pages: Not suitable for product-level information
  • Single metric display: Overkill for showing just one value — use simple stat card

Content Guidelines

Delivery Card: Show realistic delivery timeframe ("Morgen", "3-5 werkdagen", "Vandaag"). For pickup orders, use "Afhalen" with availability time. Include "Voor X uur" for same-day/next-day delivery.

Payment Card: Status must be clear and accurate ("Betaald", "In behandeling", "Factuur"). Show payment method in subtitle ("Via iDEAL", "Creditcard", etc.).

Total Card: Always include currency symbol and BTW/VAT clarification. Format as "€ XX,XX" with space after €.

  2. Label: 14px, weight 700, navy, 4px margin-bottom
  3. Value: 20px, weight 800, color-coded, 4px margin-bottom
  4. Subtitle: 12px, grey-mid, no margin
  5. Badge (optional): 8px margin-top, centered

Spacing Hierarchy

Card padding:     24px (all sides)
Icon size:        56×56px container, 28×28px icon
Icon → Label:     16px gap
Label → Value:    4px gap
Value → Subtitle: 4px gap
Grid gap:         20px (desktop), 16px (mobile)

Alignment

  • Text alignment: Center for all elements
  • Icon alignment: Center (horizontal and vertical)
  • Grid alignment: Equal-width columns, aligned baseline

White Space Management

Critical: Maintain consistent 4px spacing increments. Icon container has 16px margin-bottom to create breathing room between icon and label. Labels/values/subtitles are tightly grouped (4px gaps) to form a cohesive unit. Cards have generous 24px padding to prevent crowding.

Visual Hierarchy

  1. Primary focus: Value (20px, weight 800, color-coded) — largest and boldest
  2. Secondary focus: Icon (28px, colored) — visual anchor
  3. Tertiary: Label (14px, weight 700) — context
  4. Supporting: Subtitle (12px, grey) — additional details

STEP 7: Implementation Notes

React/Next.js Component

import { Euro, Truck, CheckCircle, Package, Clock, XCircle, FileText } from 'lucide-react'

interface OrderDetailsProps {
  delivery: {
    value: string
    subtitle: string
    icon?: 'truck' | 'package' | 'plane'
  }
  payment: {
    value: string
    subtitle: string
    status: 'paid' | 'pending' | 'failed' | 'invoice'
  }
  total: {
    value: number  // In cents
    currency?: string
  }
}

const iconMap = {
  truck: Truck,
  package: Package,
  plane: Plane,
}

const paymentIconMap = {
  paid: CheckCircle,
  pending: Clock,
  failed: XCircle,
  invoice: FileText,
}

export default function OrderDetailsCard({ delivery, payment, total }: OrderDetailsProps) {
  const DeliveryIcon = iconMap[delivery.icon || 'truck']
  const PaymentIcon = paymentIconMap[payment.status]

  const formatCurrency = (cents: number, currency = '€') => {
    return `${currency} ${(cents / 100).toFixed(2).replace('.', ',')}`
  }

  return (
    <section aria-label="Order details" className="order-details-grid">
      {/* Delivery Card */}
      <article className="order-details-card delivery">
        <div className="order-details-icon" aria-hidden="true">
          <DeliveryIcon size={28} />
        </div>
        <div className="order-details-label">Verwachte levering</div>
        <div className="order-details-value">{delivery.value}</div>
        <div className="order-details-subtitle">{delivery.subtitle}</div>
      </article>

      {/* Payment Card */}
      <article className={`order-details-card payment ${payment.status}`}>
        <div className="order-details-icon" aria-hidden="true">
          <PaymentIcon size={28} />
        </div>
        <div className="order-details-label">Betaalstatus</div>
        <div className="order-details-value">{payment.value}</div>
        <div className="order-details-subtitle">{payment.subtitle}</div>
      </article>

      {/* Total Card */}
      <article className="order-details-card total">
        <div className="order-details-icon" aria-hidden="true">
          <Euro size={28} />
        </div>
        <div className="order-details-label">Totaalbedrag</div>
        <div className="order-details-value">
          {formatCurrency(total.value, total.currency)}
        </div>
        <div className="order-details-subtitle">Incl. BTW</div>
      </article>
    </section>
  )
}

Dynamic Delivery Estimation

// Utility function to calculate delivery estimate
function calculateDeliveryEstimate(
  shippingMethod: 'standard' | 'express' | 'next-day' | 'pickup',
  orderTime: Date
): { value: string; subtitle: string } {
  const now = orderTime
  const cutoffTime = new Date(now)
  cutoffTime.setHours(23, 59, 59)

  if (shippingMethod === 'pickup') {
    return {
      value: 'Vandaag',
      subtitle: 'Vanaf 14:00 uur',
    }
  }

  if (shippingMethod === 'next-day') {
    if (now < cutoffTime) {
      return {
        value: 'Morgen',
        subtitle: 'Voor 23:59 uur',
      }
    }
    return {
      value: 'Overmorgen',
      subtitle: 'Standaard levering',
    }
  }

  if (shippingMethod === 'express') {
    return {
      value: '1-2 werkdagen',
      subtitle: 'Express verzending',
    }
  }

  // Standard shipping
  return {
    value: '3-5 werkdagen',
    subtitle: 'Standaard verzending',
  }
}

API Response Transformation

// Transform Payload CMS order data to component props
function transformOrderData(order: PayloadOrder): OrderDetailsProps {
  return {
    delivery: {
      value: order.deliveryEstimate,
      subtitle: order.deliveryTime,
      icon: order.shippingMethod === 'pickup' ? 'package' : 'truck',
    },
    payment: {
      value: order.paymentStatus === 'paid' ? 'Betaald' :
             order.paymentStatus === 'pending' ? 'In behandeling' :
             order.paymentStatus === 'invoice' ? 'Factuur' : 'Mislukt',
      subtitle: `Via ${order.paymentMethod}`,
      status: order.paymentStatus,
    },
    total: {
      value: order.total,  // Total in cents
      currency: '€',
    },
  }
}
