Component Documentation

STEP 1: Component Overview

VP10: VariantSelectionSidebar is a sticky sidebar that displays selected variants with a live price summary. It provides users with a persistent overview of their multi-variant selections as they browse.

Key Features:

  • Sticky positioning (340px width, top: 24px)
  • Header with cart icon + item count badge + "Wis alles" (clear all) button
  • Scrollable items list (max-height: 400px)
  • Empty state when no items selected
  • Remove buttons per item (X icon)
  • Price breakdown (subtotal, discount, total)
  • Disabled/enabled CTA button based on selection
  • Optional trust signals footer
  • Real-time updates on variant selection/deselection

File Size: ~820 lines

Component ID: VP10

Related: VP08 (grid card), VP09 (list row), VP11 (toolbar)

STEP 2: When to Use

✅ Use VP10 (Sidebar) When:

  • Desktop experience with 1024px+ viewport width
  • Multi-variant selection (VP08/VP09) on archive/product pages
  • Need persistent visibility of selected items while scrolling
  • Complex pricing (staffel discounts, bulk pricing)
  • B2B context where users select 5-20+ variants at once
  • Need to show running total before adding to cart

❌ Don't Use VP10 When:

  • Mobile/tablet-only experience (use inline summary instead)
  • Single-variant selection (VP01-VP05) — no sidebar needed
  • Simple pricing (no discounts) — summary below grid is enough
  • Tight horizontal space (sidebars require 340px + 32px gap)

Responsive Behavior

Desktop (≥1024px):

  • Sticky positioning: position: sticky; top: 24px;
  • 340px fixed width
  • Stays visible while scrolling page

Mobile/Tablet (<1024px):

  • Non-sticky: position: static;
  • Full-width (no fixed 340px)
  • Appears below variant grid (not beside)
  • Alternative: Use inline summary (like VP08 demo 1)

STEP 7: Implementation Notes

JavaScript Behavior

// State management
const selectedVariants = {};

// Update sidebar when variant selected/deselected
function updateSidebar() {
  const selected = Object.values(selectedVariants).filter(v => v.quantity > 0);
  const itemCount = selected.length;
  const totalQty = selected.reduce((sum, v) => sum + v.quantity, 0);
  const subtotal = selected.reduce((sum, v) => sum + (v.price * v.quantity), 0);

  // Update count badge
  document.querySelector('.sidebar-count').textContent =
    itemCount === 0 ? '0' : `${itemCount} item${itemCount > 1 ? 's' : ''}`;

  // Show/hide clear all button
  document.querySelector('.clear-all-btn').style.display =
    itemCount > 0 ? 'block' : 'none';

  // Render items or empty state
  const itemsList = document.querySelector('.selected-items-list');
  if (itemCount === 0) {
    itemsList.innerHTML = `
      <div class="empty-state">
        <i data-lucide="package-open"></i>
        <p>Geen varianten geselecteerd</p>
      </div>
    `;
  } else {
    itemsList.innerHTML = selected.map(v => `
      <div class="selected-item" data-item-id="${v.id}">
        <div class="selected-item-size">${v.name}</div>
        <div class="selected-item-info">
          <div class="selected-item-name">Maat ${v.name}</div>
          <div class="selected-item-qty">${v.quantity}× stuks</div>
        </div>
        <div class="selected-item-price">€${(v.price * v.quantity).toFixed(2).replace('.', ',')}</div>
        <button class="selected-item-remove" data-variant-id="${v.id}" aria-label="Verwijder ${v.name}">
          <i data-lucide="x"></i>
        </button>
      </div>
    `).join('');

    // Attach remove handlers
    itemsList.querySelectorAll('.selected-item-remove').forEach(btn => {
      btn.addEventListener('click', () => {
        const variantId = btn.dataset.variantId;
        // Remove from state and update UI
        delete selectedVariants[variantId];
        updateSidebar();
      });
    });
  }

  // Update price breakdown
  document.querySelector('[data-price="subtotal"]').textContent =
    `€${subtotal.toFixed(2).replace('.', ',')}`;
  document.querySelector('[data-price="total-qty"]').textContent =
    `${totalQty} stuks`;

  // Calculate discount (example: 10% for 25+ items)
  const discount = totalQty >= 25 ? subtotal * 0.1 : 0;
  const discountRow = document.querySelector('.price-line.discount');
  if (discount > 0) {
    discountRow.classList.remove('hidden');
    discountRow.querySelector('.price-line-value').textContent =
      `−€${discount.toFixed(2).replace('.', ',')}`;
  } else {
    discountRow.classList.add('hidden');
  }

  const total = subtotal - discount;
  document.querySelector('[data-price="total"]').textContent =
    `€${total.toFixed(2).replace('.', ',')}`;

  // Update CTA button
  const ctaBtn = document.querySelector('.sidebar-cta');
  ctaBtn.disabled = itemCount === 0;
  ctaBtn.innerHTML = `
    <i data-lucide="shopping-cart"></i>
    ${itemCount === 0 ? 'In Winkelwand' : `${itemCount} item${itemCount > 1 ? 's' : ''} toevoegen`}
  `;

  lucide.createIcons();
}

// Clear all button
document.querySelector('.clear-all-btn').addEventListener('click', () => {
  selectedVariants = {};
  updateSidebar();
});

React/Next.js Example

import { ShoppingCart, PackageOpen, X, Truck, ShieldCheck, RotateCcw } from 'lucide-react';

interface VariantSelectionSidebarProps {
  selectedVariants: Array<{
    id: string;
    name: string;
    quantity: number;
    price: number;
  }>;
  onRemoveItem: (id: string) => void;
  onClearAll: () => void;
  onAddToCart: () => void;
  showTrustSignals?: boolean;
}

export function VariantSelectionSidebar({
  selectedVariants,
  onRemoveItem,
  onClearAll,
  onAddToCart,
  showTrustSignals = false
}: VariantSelectionSidebarProps) {
  const itemCount = selectedVariants.length;
  const totalQty = selectedVariants.reduce((sum, v) => sum + v.quantity, 0);
  const subtotal = selectedVariants.reduce((sum, v) => sum + (v.price * v.quantity), 0);
  const discount = totalQty >= 25 ? subtotal * 0.1 : 0;
  const total = subtotal - discount;

  return (
    <aside className="selection-sidebar">
      <div className="sidebar-header">
        <div className="sidebar-title">
          <ShoppingCart className="w-4 h-4" />
          Geselecteerd
        </div>
        <span className="sidebar-count">
          {itemCount === 0 ? '0' : `${itemCount} item${itemCount > 1 ? 's' : ''}`}
        </span>
        {itemCount > 0 && (
          <button className="clear-all-btn" onClick={onClearAll}>
            Wis alles
          </button>
        )}
      </div>

      <div className="selected-items-list">
        {itemCount === 0 ? (
          <div className="empty-state">
            <PackageOpen className="w-12 h-12" />
            <p>Geen varianten geselecteerd</p>
          </div>
        ) : (
          selectedVariants.map(v => (
            <div key={v.id} className="selected-item">
              <div className="selected-item-size">{v.name}</div>
              <div className="selected-item-info">
                <div className="selected-item-name">Maat {v.name}</div>
                <div className="selected-item-qty">{v.quantity}× stuks</div>
              </div>
              <div className="selected-item-price">
                €{(v.price * v.quantity).toFixed(2).replace('.', ',')}
              </div>
              <button
                className="selected-item-remove"
                onClick={() => onRemoveItem(v.id)}
                aria-label={`Verwijder ${v.name}`}
              >
                <X className="w-3.5 h-3.5" />
              </button>
            </div>
          ))
        )}
      </div>

      <div className="price-breakdown">
        <div className="price-line">
          <span className="price-line-label">Subtotaal</span>
          <span className="price-line-value">€{subtotal.toFixed(2).replace('.', ',')}</span>
        </div>
        {discount > 0 && (
          <div className="price-line discount">
            <span className="price-line-label">Staffelkorting (−10%)</span>
            <span className="price-line-value">−€{discount.toFixed(2).replace('.', ',')}</span>
          </div>
        )}
        <div className="price-line">
          <span className="price-line-label">Totaal aantal</span>
          <span className="price-line-value">{totalQty} stuks</span>
        </div>
        <div className="price-line total">
          <span className="price-line-label">Totaal</span>
          <span className="price-line-value">€{total.toFixed(2).replace('.', ',')}</span>
        </div>
      </div>

      <button className="sidebar-cta" onClick={onAddToCart} disabled={itemCount === 0}>
        <ShoppingCart className="w-4 h-4" />
        {itemCount === 0 ? 'In Winkelwand' : `${itemCount} item${itemCount > 1 ? 's' : ''} toevoegen`}
      </button>

      {showTrustSignals && (
        <div className="trust-signals">
          <div className="trust-item">
            <Truck className="w-3 h-3" />
            <span>Gratis verzending vanaf €60</span>
          </div>
          <div className="trust-item">
            <ShieldCheck className="w-3 h-3" />
            <span>100% veilig bestellen</span>
          </div>
          <div className="trust-item">
            <RotateCcw className="w-3 h-3" />
            <span>14 dagen bedenktijd</span>
          </div>
        </div>
      )}
    </aside>
  );
}
