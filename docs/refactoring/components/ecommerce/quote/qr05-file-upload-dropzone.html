📋 QR05: FileUploadDropzone Documentation

STEP 1: Component Overview

Component Name: FileUploadDropzone (QR05)

Category: Forms / File Upload

Usage Context: Quote requests, project uploads, document submission, support tickets

Purpose:

A drag-and-drop file upload component for B2B quote requests. Supports multiple file types (PDF, Word, Excel, CAD drawings) with visual feedback for upload progress, success, and errors. Allows users to attach technical specifications, drawings, or reference documents to their quote request.

Key Features:

  • ✅ Drag-and-drop file upload
  • ✅ Click to browse file picker
  • ✅ Multiple file selection
  • ✅ File type validation (PDF, Word, Excel, CAD, images)
  • ✅ File size validation (10MB max per file)
  • ✅ Upload progress bar (0-100%)
  • ✅ File type icons (color-coded)
  • ✅ Success/error status badges
  • ✅ Delete/cancel buttons per file
  • ✅ Hover states with teal highlight
  • ✅ Accessible keyboard navigation

STEP 2: When to Use

Use this component when:

  • ✅ B2B users need to attach specifications/drawings to quotes
  • ✅ Supporting documents are required (certifications, licenses)
  • ✅ Multiple file uploads are expected (project files)
  • ✅ File types need to be restricted (security)
  • ✅ Upload progress needs to be visible (large files)

Do NOT use when:

  • ❌ Single image upload (use image uploader instead)
  • ❌ Avatar/profile photo upload (use cropper component)
  • ❌ CSV data import (use dedicated CSV uploader)
  • ❌ Files > 100MB (use chunked upload instead)

Component Guidelines:

  • Always validate file type and size before upload
  • Show progress bar for files > 1MB
  • Color-code file icons by type (PDF red, Excel green, Word blue)
  • Allow canceling uploads in progress
  • Show clear error messages for rejected files
  • Limit total files (e.g., max 10 files per request)
  • Truncate long file names with ellipsis

STEP 7: Implementation Notes

React/Next.js Component with Drag & Drop:

'use client'

import { useState, useRef } from 'react'
import { UploadCloud, FileText, FileSpreadsheet, File, Image, Check, AlertCircle, Trash2, X } from 'lucide-react'
import { FileUploadDropzoneProps, UploadedFile } from '@/types/components'

export function FileUploadDropzone({
  files = [],
  onFilesChange,
  onUpload,
  acceptedTypes = ['.pdf', '.doc', '.docx', '.xls', '.xlsx', '.dwg', '.png', '.jpg', '.jpeg'],
  maxFileSize = 10 * 1024 * 1024, // 10MB
  maxFiles = 10,
  allowMultiple = true,
  disabled = false,
  className = '',
}: FileUploadDropzoneProps) {
  const [dragOver, setDragOver] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const inputRef = useRef<HTMLInputElement>(null)

  const validateFile = (file: File): string | null => {
    // Check file type
    const extension = '.' + file.name.split('.').pop()?.toLowerCase()
    if (!acceptedTypes.includes(extension)) {
      return `Bestandstype niet toegestaan. Alleen ${acceptedTypes.join(', ')} bestanden.`
    }

    // Check file size
    if (file.size > maxFileSize) {
      const maxMB = Math.round(maxFileSize / (1024 * 1024))
      return `Bestand te groot. Maximaal ${maxMB}MB toegestaan.`
    }

    // Check file count
    if (files.length >= maxFiles) {
      return `Maximaal ${maxFiles} bestanden toegestaan.`
    }

    return null
  }

  const handleFiles = async (fileList: FileList) => {
    setError(null)
    const newFiles: UploadedFile[] = []

    for (let i = 0; i < fileList.length; i++) {
      const file = fileList[i]
      const validationError = validateFile(file)

      if (validationError) {
        setError(validationError)
        continue
      }

      const uploadedFile: UploadedFile = {
        id: `${Date.now()}-${i}`,
        name: file.name,
        size: file.size,
        type: file.type,
        status: 'uploading',
        progress: 0,
      }

      newFiles.push(uploadedFile)
    }

    if (newFiles.length > 0) {
      const updatedFiles = [...files, ...newFiles]
      onFilesChange(updatedFiles)

      // Trigger upload
      const filesToUpload = Array.from(fileList).slice(0, newFiles.length)
      await onUpload(filesToUpload)
    }
  }

  const handleDrop = (e: React.DragEvent) => {
    e.preventDefault()
    setDragOver(false)
    if (!disabled && e.dataTransfer.files.length > 0) {
      handleFiles(e.dataTransfer.files)
    }
  }

  const handleDragOver = (e: React.DragEvent) => {
    e.preventDefault()
    if (!disabled) setDragOver(true)
  }

  const handleDragLeave = () => {
    setDragOver(false)
  }

  const handleInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files && e.target.files.length > 0) {
      handleFiles(e.target.files)
    }
  }

  const handleDelete = (id: string) => {
    onFilesChange(files.filter(f => f.id !== id))
  }

  const getFileIcon = (type: string) => {
    if (type.includes('pdf')) return <FileText />
    if (type.includes('sheet') || type.includes('excel')) return <FileSpreadsheet />
    if (type.includes('word') || type.includes('document')) return <File />
    if (type.includes('image')) return <Image />
    return <File />
  }

  const getFileIconClass = (type: string): string => {
    if (type.includes('pdf')) return 'pdf'
    if (type.includes('sheet') || type.includes('excel')) return 'excel'
    if (type.includes('word') || type.includes('document')) return 'word'
    if (type.includes('image')) return 'image'
    return ''
  }

  const formatFileSize = (bytes: number): string => {
    if (bytes < 1024) return `${bytes} B`
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`
    return `${(bytes / (1024 * 1024)).toFixed(1)} MB`
  }

  return (
    <div className={`file-upload-dropzone ${className}`}>
      <input
        ref={inputRef}
        type="file"
        className="file-upload-input"
        multiple={allowMultiple}
        accept={acceptedTypes.join(',')}
        onChange={handleInputChange}
        disabled={disabled}
      />

      <label
        className={`file-upload-area ${dragOver ? 'drag-over' : ''}`}
        onDrop={handleDrop}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onClick={() => inputRef.current?.click()}
      >
        <UploadCloud className="file-upload-icon" />
        <div className="file-upload-text">
          Sleep bestanden hierheen of klik om te uploaden
        </div>
        <div className="file-upload-hint">
          PDF, Word, Excel, AutoCAD (max {Math.round(maxFileSize / (1024 * 1024))}MB per bestand)
        </div>
      </label>

      {files.length > 0 && (
        <div className="file-upload-list">
          {files.map((file) => (
            <div key={file.id} className="file-upload-item">
              <div className={`file-upload-item-icon ${getFileIconClass(file.type)}`}>
                {getFileIcon(file.type)}
              </div>
              <div className="file-upload-item-info">
                <div className="file-upload-item-name">{file.name}</div>
                <div className="file-upload-item-meta">{formatFileSize(file.size)}</div>
                {file.status === 'uploading' && file.progress !== undefined && (
                  <div className="file-upload-item-progress">
                    <div
                      className="file-upload-item-progress-bar"
                      style={{ width: `${file.progress}%` }}
                    />
                  </div>
                )}
              </div>
              {file.status === 'uploading' && (
                <span className="file-upload-item-status uploading">Uploading...</span>
              )}
              {file.status === 'success' && (
                <span className="file-upload-item-status success">
                  <Check size={10} /> Geüpload
                </span>
              )}
              {file.status === 'error' && (
                <span className="file-upload-item-status error">
                  <AlertCircle size={10} /> {file.errorMessage || 'Fout'}
                </span>
              )}
              <button
                className="file-upload-item-delete"
                onClick={() => handleDelete(file.id)}
                aria-label={`Verwijder ${file.name}`}
              >
                {file.status === 'uploading' ? <X /> : <Trash2 />}
              </button>
            </div>
          ))}
        </div>
      )}

      {error && (
        <div className="file-upload-error">
          <AlertCircle className="file-upload-error-icon" />
          <div><strong>Upload mislukt:</strong> {error}</div>
        </div>
      )}
    </div>
  )
}

Upload Progress Tracking:

// Upload function with progress tracking
async function uploadFile(
  file: File,
  onProgress: (progress: number) => void
): Promise<string> {
  const formData = new FormData()
  formData.append('file', file)

  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest()

    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const progress = Math.round((e.loaded / e.total) * 100)
        onProgress(progress)
      }
    })

    xhr.addEventListener('load', () => {
      if (xhr.status === 200) {
        const response = JSON.parse(xhr.responseText)
        resolve(response.url)
      } else {
        reject(new Error('Upload failed'))
      }
    })

    xhr.addEventListener('error', () => reject(new Error('Network error')))
    xhr.open('POST', '/api/upload')
    xhr.send(formData)
  })
}

Common Pitfalls:

  • ❌ Not preventing default on drag events (causes browser to open file)
  • ❌ Validating file type on frontend only (security risk — always validate on backend)
  • ❌ Not showing progress for large files (poor UX)
  • ❌ Not allowing cancel during upload (frustrating for users)
  • ❌ Not truncating long file names (breaks layout)
  • ❌ Missing aria-labels on delete buttons (accessibility)
  • ❌ Not clearing error messages when new files are added
