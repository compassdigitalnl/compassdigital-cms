📋 QR04: ProjectInfoForm Documentation

STEP 1: Component Overview

Component Name: ProjectInfoForm (QR04)

Category: Forms / B2B

Usage Context: Quote request flow, bulk orders, project-based procurement

Purpose:

Captures project-specific information for B2B quote requests. Includes project name (for invoicing/delivery docs), delivery date (with validation), and optional notes for special requirements or questions.

Key Features:

  • ✅ Project name input with helper text
  • ✅ Delivery date picker with min date validation
  • ✅ Optional notes textarea with character counter (500 max)
  • ✅ Required field indicators (*)
  • ✅ Real-time validation with error messages
  • ✅ Helper icons (info, calendar, alert-circle)
  • ✅ Responsive single-column layout

STEP 2: When to Use

Use this component when:

  • ✅ B2B user is requesting a custom quote (offerte)
  • ✅ Order is project-based (construction, medical, events)
  • ✅ Delivery date needs to be specified upfront
  • ✅ Customer may have special requirements/questions
  • ✅ Project name needed for invoicing/tracking

Do NOT use when:

  • ❌ Standard checkout flow (use shipping date selector instead)
  • ❌ B2C orders (too complex for consumers)
  • ❌ Quick reorders (no project context needed)

Component Guidelines:

  • Always validate delivery date (min 3 business days in future)
  • Show helper text explaining lead times
  • Make notes optional but encourage usage
  • Character counter turns warning (amber) at 90%, error (coral) at 100%
  • Preserve form data in localStorage (auto-save draft)

STEP 7: Implementation Notes

React/Next.js Component:

'use client'

import { useState } from 'react'
import { FolderOpen, Info, Calendar, AlertCircle } from 'lucide-react'
import { ProjectInfoFormProps, ValidationErrors } from '@/types/components'

export function ProjectInfoForm({
  projectInfo,
  onSubmit,
  minDeliveryDate,
  notesMaxLength = 500,
  showNotesCounter = true,
  autoSave = true,
  className = '',
}: ProjectInfoFormProps) {
  const [formData, setFormData] = useState({
    projectName: projectInfo?.projectName || '',
    deliveryDate: projectInfo?.deliveryDate || '',
    notes: projectInfo?.notes || '',
  })

  const [errors, setErrors] = useState<ValidationErrors>({})
  const [charCount, setCharCount] = useState(formData.notes.length)

  // Calculate min date (3 business days from today)
  const getMinDate = () => {
    if (minDeliveryDate) return minDeliveryDate
    const date = new Date()
    let daysAdded = 0
    while (daysAdded < 3) {
      date.setDate(date.getDate() + 1)
      if (date.getDay() !== 0 && date.getDay() !== 6) daysAdded++
    }
    return date.toISOString().split('T')[0]
  }

  const handleNotesChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    const value = e.target.value
    setFormData(prev => ({ ...prev, notes: value }))
    setCharCount(value.length)
  }

  const validate = () => {
    const newErrors: ValidationErrors = {}

    if (!formData.projectName.trim()) {
      newErrors.projectName = 'Projectnaam is verplicht'
    }

    if (!formData.deliveryDate) {
      newErrors.deliveryDate = 'Leverdatum is verplicht'
    } else if (formData.deliveryDate < getMinDate()) {
      newErrors.deliveryDate = 'Leverdatum moet minimaal 3 werkdagen in de toekomst liggen'
    }

    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (validate()) {
      onSubmit(formData)
    }
  }

  // Character counter color
  const getCounterClass = () => {
    const percentage = (charCount / notesMaxLength) * 100
    if (percentage >= 100) return 'error'
    if (percentage >= 90) return 'warning'
    return ''
  }

  return (
    <form className={`project-info-form ${className}`} onSubmit={handleSubmit}>
      <div className="project-info-header">
        <h3 className="project-info-title">
          <FolderOpen className="project-info-title-icon" />
          Projectinformatie
        </h3>
      </div>

      <div className="project-info-fields">
        {/* Project Name */}
        <div className={`project-info-field ${errors.projectName ? 'error' : ''}`}>
          <label htmlFor="project-name" className="project-info-label required">
            Projectnaam
          </label>
          <input
            type="text"
            id="project-name"
            className="project-info-input"
            value={formData.projectName}
            onChange={(e) => setFormData(prev => ({ ...prev, projectName: e.target.value }))}
            placeholder="Bijv. Q2 2024 Stock Replenishment"
            required
          />
          {errors.projectName ? (
            <p className="project-info-error-message">
              <AlertCircle className="project-info-error-icon" />
              {errors.projectName}
            </p>
          ) : (
            <p className="project-info-helper">
              <Info className="project-info-helper-icon" />
              Deze naam wordt gebruikt op facturen en leveringsdocumenten.
            </p>
          )}
        </div>

        {/* Delivery Date */}
        <div className={`project-info-field ${errors.deliveryDate ? 'error' : ''}`}>
          <label htmlFor="delivery-date" className="project-info-label required">
            Gewenste leverdatum
          </label>
          <input
            type="date"
            id="delivery-date"
            className="project-info-input"
            value={formData.deliveryDate}
            onChange={(e) => setFormData(prev => ({ ...prev, deliveryDate: e.target.value }))}
            min={getMinDate()}
            required
          />
          {errors.deliveryDate ? (
            <p className="project-info-error-message">
              <AlertCircle className="project-info-error-icon" />
              {errors.deliveryDate}
            </p>
          ) : (
            <p className="project-info-helper">
              <Calendar className="project-info-helper-icon" />
              Standaard levertijd is 3-5 werkdagen. Voor spoedlevering neem contact op.
            </p>
          )}
        </div>

        {/* Notes */}
        <div className="project-info-field">
          <label htmlFor="notes" className="project-info-label">
            Opmerkingen / Speciale wensen
          </label>
          <textarea
            id="notes"
            className="project-info-textarea"
            value={formData.notes}
            onChange={handleNotesChange}
            placeholder="Bijv. speciale levertijden, verpakkingseisen, of vragen over producten..."
            maxLength={notesMaxLength}
          />
          {showNotesCounter && (
            <div className={`project-info-counter ${getCounterClass()}`}>
              {charCount} / {notesMaxLength}
            </div>
          )}
        </div>
      </div>
    </form>
  )
}

Date Validation (Business Days):

// utils/dateValidation.ts
export function addBusinessDays(date: Date, days: number): Date {
  const result = new Date(date)
  let daysAdded = 0

  while (daysAdded < days) {
    result.setDate(result.getDate() + 1)
    // Skip weekends (0 = Sunday, 6 = Saturday)
    if (result.getDay() !== 0 && result.getDay() !== 6) {
      daysAdded++
    }
  }

  return result
}

export function getMinDeliveryDate(leadTimeDays: number = 3): string {
  const minDate = addBusinessDays(new Date(), leadTimeDays)
  return minDate.toISOString().split('T')[0]
}

Common Pitfalls:

  • ❌ Not validating business days (allowing weekend delivery)
  • ❌ Allowing past dates (min date validation missing)
  • ❌ Character counter not updating in real-time
  • ❌ Not clearing errors after user corrects input
  • ❌ Missing aria-live on character counter (accessibility)
  • ❌ Textarea not allowing vertical resize (poor UX)
