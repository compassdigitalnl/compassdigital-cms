📋 STEP 1: COMPONENT OVERVIEW

Component: C10 - Review Widget

Category: E-commerce / Social Proof

Complexiteit: High (summary stats, rating bars, review list, helpful voting)

Purpose:

Complete review section for product pages. Shows aggregate rating, star distribution, individual reviews with verified badges, and helpful voting. Builds trust and influences purchase decisions.

Key Features:

  • Split layout: Summary (left) + Rating bars (right)
  • Large average score (48px, 800 weight)
  • Star distribution with visual bars (amber)
  • Review count
  • "Review schrijven" CTA button
  • Individual review cards with avatar (initials)
  • Verified purchase badge (green)
  • 5-star rating display
  • Review text
  • Helpful buttons (thumbs up/down with count)

🎯 STEP 2: WHEN TO USE

Ideaal voor:

  • Product detail pages (all products)
  • Service pages (for service-based businesses)
  • Company reviews (testimonials page)

Best Practices:

  • Show only verified purchases (requires login + order check)
  • Sort by: Most helpful (default), Newest, Highest/Lowest rating
  • Pagination or "Load more" for 10+ reviews
  • Allow filtering by star rating (e.g., "Show only 5-star")
  • Include review images/photos (optional)
  • Reply to reviews (vendor/admin responses)
  • Report inappropriate reviews
  • Integrate with Google Reviews / Trustpilot for SEO

// React Component
export function ReviewWidget({ product, reviews, onWriteReview, onHelpful }) {
  const [sortBy, setSortBy] = useState('helpful');
  const [filterBy, setFilterBy] = useState(null);

  const sortedReviews = useMemo(() => {
    let filtered = reviews.items;
    if (filterBy) {
      filtered = filtered.filter(r => r.rating === filterBy);
    }

    return filtered.sort((a, b) => {
      if (sortBy === 'helpful') return b.helpful.yes - a.helpful.yes;
      if (sortBy === 'newest') return new Date(b.date) - new Date(a.date);
      if (sortBy === 'rating') return b.rating - a.rating;
    });
  }, [reviews, sortBy, filterBy]);

  const handleHelpful = async (reviewId, vote) => {
    await onHelpful(reviewId, vote);
    // Update local state or refetch
  };

  return (
    <div className="reviews-card">
      <div className="rv-header">
        <div className="rv-summary">
          <div className="rv-avg">{reviews.average}</div>
          <div className="rv-stars-big">{'★'.repeat(Math.round(reviews.average))}</div>
          <div className="rv-total">{reviews.total} reviews</div>
          <button className="rv-write-btn" onClick={onWriteReview}>
            <PenLine size={15} /> Review schrijven
          </button>
        </div>
        <div className="rv-bars">
          {[5, 4, 3, 2, 1].map(star => (
            <div key={star} className="rv-bar-row">
              <span className="rv-bar-label">{star}</span>
              <span className="rv-bar-star">★</span>
              <div className="rv-bar-track">
                <div className="rv-bar-fill" style={{ width: `${(reviews.distribution[star] / reviews.total) * 100}%` }} />
              </div>
              <span className="rv-bar-count">{reviews.distribution[star]}</span>
            </div>
          ))}
        </div>
      </div>
      <div className="rv-list">
        {sortedReviews.map(review => (
          <article key={review.id} className="rv-item">
            <div className="rv-item-header">
              <div className="rv-author">
                <div className="rv-avatar">{review.author.initials}</div>
                <div>
                  <div className="rv-name">{review.author.name}</div>
                  {review.author.verified && (
                    <div className="rv-verified">
                      <BadgeCheck size={12} /> Geverifieerde aankoop
                    </div>
                  )}
                </div>
              </div>
              <div className="rv-date">{formatDate(review.date)}</div>
            </div>
            <div className="rv-stars">{'★'.repeat(review.rating)}{'☆'.repeat(5 - review.rating)}</div>
            <div className="rv-text">{review.text}</div>
            <div className="rv-helpful">
              <span>Was dit nuttig?</span>
              <button className="rv-helpful-btn" onClick={() => handleHelpful(review.id, 'yes')}>
                <ThumbsUp size={12} /> Ja ({review.helpful.yes})
              </button>
              <button className="rv-helpful-btn" onClick={() => handleHelpful(review.id, 'no')}>
                <ThumbsDown size={12} /> Nee
              </button>
            </div>
          </article>
        ))}
      </div>
    </div>
  );
}
