📋 STEP 1: COMPONENT OVERVIEW

Component: C20 - Pagination

Category: Utility / Navigation

Complexiteit: Low (numbered buttons, active state, disabled state)

Purpose:

Page navigation component for long lists of content (product archives, search results, blog posts). Allows users to navigate between pages with numbered buttons, previous/next arrows, and ellipsis for skipped pages.

Key Features:

  • Numbered page buttons (42px × 42px)
  • Active page indicator (teal background)
  • Previous/next arrow buttons (chevron icons)
  • Disabled state for first/last page (opacity 0.3, no pointer events)
  • Ellipsis (…) for skipped pages
  • Hover state (teal border + background)
  • 3 variants: Default, With count text, Compact (mobile)
  • Responsive: Smaller on mobile (38px buttons)

🎯 STEP 2: WHEN TO USE

Ideaal voor:

  • Product category pages (>24 products)
  • Search results pages
  • Blog/article archives
  • Order history (account pages)
  • Admin tables with many rows

Best Practices:

  • Show 24-48 items per page (avoid infinite scroll for B2B)
  • Always show: first page, last page, current page, 2 pages before/after current
  • Use ellipsis (…) for gaps >1 page
  • On mobile: Use compact variant (Prev/Next only)
  • Preserve filters/sorting in pagination URLs (?page=2&sort=price)
  • Update URL on page change (SEO + back button support)
  • Scroll to top on page change (smooth scroll)
  • Show result count: "Resultaten 1-24 van 186"
  • Disable prev on page 1, disable next on last page

💡 STEP 7: IMPLEMENTATION NOTES

// React Component
import { ChevronLeft, ChevronRight } from 'lucide-react';

export function Pagination({
  currentPage,
  totalPages,
  totalItems,
  itemsPerPage = 24,
  onPageChange,
  variant = 'default'
}) {
  // Generate visible page numbers with smart ellipsis
  const getPageNumbers = () => {
    const pages = [];
    const maxVisible = 7;

    if (totalPages <= maxVisible) {
      // Show all pages
      for (let i = 1; i <= totalPages; i++) {
        pages.push(i);
      }
    } else {
      // Smart ellipsis logic
      if (currentPage <= 3) {
        // Near start: 1 2 3 … 8
        pages.push(1, 2, 3, '...', totalPages);
      } else if (currentPage >= totalPages - 2) {
        // Near end: 1 … 6 7 8
        pages.push(1, '...', totalPages - 2, totalPages - 1, totalPages);
      } else {
        // Middle: 1 … 4 5 6 … 8
        pages.push(1, '...', currentPage - 1, currentPage, currentPage + 1, '...', totalPages);
      }
    }

    return pages;
  };

  const handlePageChange = (page) => {
    if (page < 1 || page > totalPages || page === currentPage) return;
    onPageChange(page);
    window.scrollTo({ top: 0, behavior: 'smooth' });

    // Screen reader announcement
    const start = (page - 1) * itemsPerPage + 1;
    const end = Math.min(page * itemsPerPage, totalItems);
    announce(`Pagina ${page} van ${totalPages} geladen. Toont resultaten ${start} tot ${end}.`);
  };

  const pages = getPageNumbers();

  if (variant === 'compact') {
    return (
      <div className="pagination-compact">
        <button
          className="page-btn"
          onClick={() => handlePageChange(currentPage - 1)}
          disabled={currentPage === 1}
        >
          <ChevronLeft size={16} /> Vorige
        </button>
        <div className="page-info">
          Pagina {currentPage} van {totalPages}
        </div>
        <button
          className="page-btn"
          onClick={() => handlePageChange(currentPage + 1)}
          disabled={currentPage === totalPages}
        >
          Volgende <ChevronRight size={16} />
        </button>
      </div>
    );
  }

  return (
    <nav aria-label="Paginanummering" role="navigation">
      {variant === 'with-count' && (
        <div className="page-count">
          Resultaten <strong>{(currentPage - 1) * itemsPerPage + 1}-{Math.min(currentPage * itemsPerPage, totalItems)}</strong> van <strong>{totalItems}</strong>
        </div>
      )}

      <div className={variant === 'with-count' ? 'page-buttons' : 'pagination'}>
        <button
          className={`page-btn ${currentPage === 1 ? 'disabled' : ''}`}
          onClick={() => handlePageChange(currentPage - 1)}
          disabled={currentPage === 1}
          aria-label="Vorige pagina"
        >
          <ChevronLeft size={16} />
        </button>

        {pages.map((page, i) =>
          page === '...' ? (
            <span key={`ellipsis-${i}`} className="page-dots" aria-hidden="true">…</span>
          ) : (
            <button
              key={page}
              className={`page-btn ${page === currentPage ? 'active' : ''}`}
              onClick={() => handlePageChange(page)}
              aria-current={page === currentPage ? 'page' : undefined}
              aria-label={page === currentPage ? `Pagina ${page}, huidige pagina` : `Ga naar pagina ${page}`}
            >
              {page}
            </button>
          )
        )}

        <button
          className={`page-btn ${currentPage === totalPages ? 'disabled' : ''}`}
          onClick={() => handlePageChange(currentPage + 1)}
          disabled={currentPage === totalPages}
          aria-label="Volgende pagina"
        >
          <ChevronRight size={16} />
        </button>
      </div>
    </nav>
  );
}

// Next.js usage with URL sync
import { useRouter, useSearchParams } from 'next/navigation';

export function ProductArchive() {
  const router = useRouter();
  const searchParams = useSearchParams();
  const currentPage = parseInt(searchParams.get('page') || '1');

  const handlePageChange = (page) => {
    const params = new URLSearchParams(searchParams);
    params.set('page', page.toString());
    router.push(`?${params.toString()}`);
  };

  return (
    <Pagination
      currentPage={currentPage}
      totalPages={8}
      totalItems={186}
      onPageChange={handlePageChange}
      variant="with-count"
    />
  );
}
