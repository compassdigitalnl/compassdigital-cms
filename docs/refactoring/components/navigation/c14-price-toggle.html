📋 STEP 1: COMPONENT OVERVIEW

Component: C14 - Price Toggle

Parent: C14-MegaNavigation (embedded in SearchBar) / Standalone

Category: Navigation / Controls

Complexiteit: Low (simple toggle)

Purpose:

Switch tussen B2B (zakelijk) en B2C (particulier) prijsweergave op e-commerce platforms met dual pricing.

Variants:

  • Default: 11px font, 5px 10px padding (in search bar)
  • Large: 12px font, 8px 16px padding (on product pages, pricing pages)
  • With Label: Optional "Prijsweergave:" label left of toggle

🎯 STEP 2: WHEN TO USE

Ideaal voor:

  • E-commerce platforms die zowel B2B als B2C klanten bedienen
  • Groothandels met consumentendivisie
  • Sites met volumekortingen (B2B) en enkele aankopen (B2C)
  • Medische/industriële leveranciers die ook particulieren bedienen

Niet geschikt voor:

  • Pure B2C webshops (geen dual pricing)
  • Pure B2B platformen (één prijstype)
  • Sites waar prijsverschil irrelevant is

Common Placements:

  • Header Search Bar: Global price mode (meest gebruikelijk)
  • Product Pages: Local toggle per product
  • Pricing Pages: Switch tussen tarievenplannen
  • Account Settings: Default prijsweergave instelling

💡 STEP 7: IMPLEMENTATION NOTES

Basic Click Handler:

function setPriceMode(btn) {
  // Get container
  const container = btn.parentElement;

  // Remove active from all buttons
  container.querySelectorAll('.price-toggle-btn')
    .forEach(b => {
      b.classList.remove('active');
      b.setAttribute('aria-pressed', 'false');
    });

  // Activate clicked button
  btn.classList.add('active');
  btn.setAttribute('aria-pressed', 'true');

  // Get mode
  const mode = btn.textContent.trim();

  // Store in localStorage
  localStorage.setItem('priceMode', mode);

  // Dispatch event
  document.dispatchEvent(
    new CustomEvent('pricemodechange', {
      detail: { mode }
    })
  );
}

Initialize on Page Load:

// Restore saved mode
document.addEventListener('DOMContentLoaded', () => {
  const savedMode = localStorage.getItem('priceMode') || 'B2B';
  const button = document.querySelector(
    `.price-toggle-btn:contains("${savedMode}")`
  );

  if (button && !button.classList.contains('active')) {
    button.click();
  }
});

Keyboard Arrow Navigation:

document.querySelectorAll('.price-toggle').forEach(toggle => {
  toggle.addEventListener('keydown', (e) => {
    if (e.key === 'ArrowLeft' || e.key === 'ArrowRight') {
      e.preventDefault();

      const buttons = [...toggle.querySelectorAll('.price-toggle-btn')];
      const currentIndex = buttons.findIndex(b => b === document.activeElement);

      let nextIndex;
      if (e.key === 'ArrowRight') {
        nextIndex = (currentIndex + 1) % buttons.length;
      } else {
        nextIndex = (currentIndex - 1 + buttons.length) % buttons.length;
      }

      buttons[nextIndex].focus();
      buttons[nextIndex].click();
    }
  });
});

Global Price Updates:

function updatePrices(mode) {
  // Update all product prices on page
  document.querySelectorAll('[data-price-b2b][data-price-b2c]').forEach(el => {
    const b2bPrice = el.dataset.priceB2b;
    const b2cPrice = el.dataset.priceB2c;

    el.textContent = mode === 'B2B' ? b2bPrice : b2cPrice;
  });

  // Update staffel tables visibility
  document.querySelectorAll('.staffel-table').forEach(table => {
    table.style.display = mode === 'B2B' ? 'block' : 'none';
  });
}
