// src/types/blocks.ts import { SerializedEditorState } from '@payloadcms/richtext-lexical/lexical' import { Media } from '@/payload-types' export interface MediaButton { label: string url: string variant: 'primary' | 'secondary' | 'outline' newTab?: boolean } export interface MediaBlock { blockType: 'media' id?: string mediaType: 'image' | 'video' media: string | Media videoUrl?: string mediaPosition: 'left' | 'right' split: '50-50' | '60-40' | '40-60' subtitle?: string title: string content?: SerializedEditorState buttons?: MediaButton[] backgroundColor: 'white' | 'grey' | 'teal' } // Type guard to check if a block is a MediaBlock export function isMediaBlock(block: any): block is MediaBlock { return block?.blockType === 'media' } // Helper to extract media URL export function getMediaUrl(media: string | Media): string | undefined { if (typeof media === 'string') { return media } return media?.url }
Step 3: React Component Implementation

Create src/branches/blocks/shared/MediaBlock/Component.tsx to render the block:

// src/branches/blocks/shared/MediaBlock/Component.tsx import React from 'react' import { RichText } from '@payloadcms/richtext-lexical/react' import { MediaBlock as MediaBlockType, getMediaUrl } from '@/types/blocks' import { cn } from '@/lib/utils' import Image from 'next/image' interface MediaBlockProps { block: MediaBlockType } export const MediaBlock: React.FC<MediaBlockProps> = ({ block }) => { const { mediaType, media, videoUrl, mediaPosition, split, subtitle, title, content, buttons, backgroundColor, } = block const mediaUrl = getMediaUrl(media) // Map split values to grid classes const gridClass = { '50-50': 'md:grid-cols-2', '60-40': 'md:grid-cols-[60fr_40fr]', '40-60': 'md:grid-cols-[40fr_60fr]', }[split] // Map background color to Tailwind classes const bgClass = { white: 'bg-white', grey: 'bg-grey-light', teal: 'bg-teal-bg', }[backgroundColor] // Button variant styles const buttonStyles = { primary: 'bg-teal text-white hover:bg-teal-dark', secondary: 'bg-navy text-white hover:bg-navy-light', outline: 'bg-transparent border-2 border-teal text-teal hover:bg-teal hover:text-white', } return ( <section className={cn('py-12 md:py-16', bgClass)}> <div className="container"> <div className={cn( 'grid gap-0 items-center', gridClass, mediaPosition === 'right' && split === '60-40' && 'md:grid-cols-[60fr_40fr]', mediaPosition === 'right' && split === '40-60' && 'md:grid-cols-[40fr_60fr]', )} > {/* Media Column */} <div className={cn( 'relative min-h-[300px] md:min-h-[400px]', mediaPosition === 'right' && 'md:order-2', )} > {mediaType === 'image' && mediaUrl && ( <Image src={mediaUrl} alt={title} fill className="object-cover" /> )} {mediaType === 'video' && ( <div className="relative w-full h-full bg-grey-light flex items-center justify-center"> {videoUrl ? ( <iframe src={videoUrl} title={title} className="absolute inset-0 w-full h-full" allowFullScreen /> ) : mediaUrl ? ( <video src={mediaUrl} controls className="w-full h-full object-cover" > Your browser does not support the video tag. </video> ) : ( <div className="text-6xl">🎥</div> )} </div> )} </div> {/* Content Column */} <div className={cn( 'p-8 md:p-12 flex flex-col justify-center', mediaPosition === 'right' && 'md:order-1', )} > {subtitle && ( <div className="text-xs font-bold uppercase tracking-wider text-teal mb-3"> {subtitle} </div> )} <h2 className="font-display text-2xl md:text-3xl text-navy mb-4"> {title} </h2> {content && ( <div className="prose prose-sm md:prose-base prose-slate max-w-none mb-6"> <RichText data={content} /> </div> )} {buttons && buttons.length > 0 && ( <div className="flex flex-wrap gap-3"> {buttons.map((button, index) => ( <a key={index} href={button.url} target={button.newTab ? '_blank' : undefined} rel={button.newTab ? 'noopener noreferrer' : undefined} className={cn( 'inline-flex items-center justify-center px-6 py-3 rounded-lg font-bold text-sm transition-all duration-200', buttonStyles[button.variant] )} > {button.label} </a> ))} </div> )} </div> </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Media Block" is visible
2. Image upload works: Upload an image → preview appears in admin panel
3. Video upload works: Upload MP4 video → video player appears on frontend
4. Media position toggles: Test left/right positions → media and text swap correctly
5. Split ratios work: Test all 3 split options (50/50, 60/40, 40/60) → column widths adjust correctly
6. Buttons render correctly: Add 1-2 buttons → test all 3 variants (primary/secondary/outline)
7. TypeScript compilation: Run npm run build → no TypeScript errors
Step 6: Database Migration

⚠️ IMPORTANT: After creating the Media block, generate a database migration:

# Generate migration for Media block npx payload migrate:create add_media_block # Review generated migration file in src/migrations/ # It should contain ALTER TABLE statements for pages_blocks table # Expected schema changes: # - pages_blocks.media_type (type: varchar) — enum: 'image', 'video' # - pages_blocks.media_id (type: integer) — foreign key to media table # - pages_blocks.video_url (type: varchar) — optional YouTube/Vimeo URL # - pages_blocks.media_position (type: varchar) — enum: 'left', 'right' # - pages_blocks.split (type: varchar) — enum: '50-50', '60-40', '40-60' # - pages_blocks.subtitle (type: varchar) — optional overline text # - pages_blocks.title (type: varchar) — required heading # - pages_blocks.content (type: jsonb) — serialized Lexical state # - pages_blocks.buttons (type: jsonb) — array of button objects # - pages_blocks.background_color (type: varchar) — enum: 'white', 'grey', 'teal' # Apply migration to database npx payload migrate # Verify in database (PostgreSQL example) psql -d your_database -c "SELECT column_name, data_type FROM information_schema.columns WHERE table_name = 'pages_blocks' AND column_name IN ('media_type', 'media_id', 'media_position', 'split', 'title', 'content');" # Expected output: # column_name | data_type # -----------------+----------- # media_type | varchar # media_id | integer # media_position | varchar # split | varchar # title | varchar # content | jsonb
Step 7: Common Issues & Solutions
Issue Solution
Image not displaying on frontend Check next.config.js → add media domain to images.domains array. Verify media file uploaded correctly in /admin/collections/media
Video not playing For uploaded videos: use MP4 format with H.264 codec. For embeds: verify YouTube/Vimeo URL is embed format (e.g., https://www.youtube.com/embed/VIDEO_ID)
Media and text not swapping positions Check mediaPosition === 'right' conditional in Component → ensure md:order-1 and md:order-2 classes applied correctly
Split ratios not working Verify Tailwind grid classes: md:grid-cols-[60fr_40fr] syntax. May need to add to safelist in tailwind.config.ts if using JIT mode
Buttons not styled correctly Check button variant mapping in Component → ensure all 3 variants (primary/secondary/outline) have complete Tailwind classes
Database error on save Run migration: npx payload migrate → check pages_blocks table has all required columns. Verify foreign key to media table exists
TypeScript error: Media type Import Media type from @/payload-types (auto-generated). Run payload generate:types if missing
Conditional field not showing (videoUrl) Check admin condition: siblingData?.mediaType === 'video' → ensure mediaType field defined before videoUrl field
Step 8: Register Block in Payload Config

Add the Media block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { MediaBlock } from '@/branches/blocks/shared/MediaBlock/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ MediaBlock, // ← Add Media block here // Hero, // Features, // Content, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { MediaBlock } from '@/branches/blocks/shared/MediaBlock/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'media': return <MediaBlock key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'content': return <Content key={index} block={block} /> default: return null } })} </main> ) }
