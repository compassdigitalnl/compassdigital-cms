How it works: Set data-theme="vertical-name" on the <html> element. All components reference var(--teal), var(--navy), etc. The theme override changes the value of those variables globally. See Chapter 6 for complete theme configurations per vertical.

Generate CSS Variables in Next.js Layout

Create utility to convert theme to CSS variables and apply in app layout:

// src/lib/theme/generateThemeCSS.ts
import { ThemeConfig } from '@/types/theme'

export function generateThemeCSS(theme: ThemeConfig): string {
  const vars: string[] = []

  // Color overrides
  if (theme.primaryColor) vars.push(`--teal: ${theme.primaryColor};`)
  if (theme.primaryColorLight) vars.push(`--teal-light: ${theme.primaryColorLight};`)
  if (theme.primaryColorDark) vars.push(`--teal-dark: ${theme.primaryColorDark};`)
  if (theme.primaryColorGlow) vars.push(`--teal-glow: ${theme.primaryColorGlow};`)
  if (theme.primaryColorBg) vars.push(`--teal-bg: ${theme.primaryColorBg};`)

  // Dark surfaces
  if (theme.darkSurface) vars.push(`--navy: ${theme.darkSurface};`)
  if (theme.darkSurfaceLight) vars.push(`--navy-light: ${theme.darkSurfaceLight};`)

  // Custom colors
  if (theme.customColors) {
    theme.customColors.forEach(({ tokenName, tokenValue }) => {
      vars.push(`--${tokenName}: ${tokenValue};`)
    })
  }

  // Typography
  if (theme.bodyFont) vars.push(`--font: '${theme.bodyFont}', system-ui, sans-serif;`)
  if (theme.headingFont) vars.push(`--font-heading: '${theme.headingFont}', var(--font);`)

  // Visual overrides
  if (theme.borderRadiusSm) vars.push(`--r-sm: ${theme.borderRadiusSm}px;`)
  if (theme.borderRadiusMd) vars.push(`--r-md: ${theme.borderRadiusMd}px;`)

  // Custom shadows
  if (theme.customShadows) {
    theme.customShadows.forEach(({ tokenName, tokenValue }) => {
      vars.push(`--${tokenName}: ${tokenValue};`)
    })
  }

  return `[data-theme="${theme.slug}"] {\n  ${vars.join('\n  ')}\n}`
}

// ═══════════════════════════════════════════════════════════
// app/layout.tsx — Apply theme CSS variables
// ═══════════════════════════════════════════════════════════

import { getPayloadHMR } from '@payloadcms/next/utilities'
import configPromise from '@payload-config'
import { generateThemeCSS } from '@/lib/theme/generateThemeCSS'
import { ThemeConfig } from '@/types/theme'

export default async function RootLayout({ children }: { children: React.ReactNode }) {
  const payload = await getPayloadHMR({ config: configPromise })

  // Fetch all active themes
  const { docs: themes } = await payload.find({
    collection: 'themes',
    where: {
      status: { equals: 'active' },
    },
  })

  // Generate CSS for all themes
  const themeCSS = themes.map((theme: ThemeConfig) => generateThemeCSS(theme)).join('\n\n')

  // Find default theme
  const defaultTheme = themes.find((t: ThemeConfig) => t.isDefault)

  return (
    <html lang="en" data-theme={defaultTheme?.slug || 'medisch'}>
      <head>
        <style dangerouslySetInnerHTML={{ __html: themeCSS }} />
      </head>
      <body>{children}</body>
    </html>
  )
}

Create Theme Switcher Component

Allow users to switch themes dynamically (optional — useful for previewing verticals):

// src/components/ThemeSwitcher.tsx
'use client'

import { useState, useEffect } from 'react'

interface Theme {
  name: string
  slug: string
  icon: string
}

export function ThemeSwitcher({ themes }: { themes: Theme[] }) {
  const [currentTheme, setCurrentTheme] = useState('medisch')

  useEffect(() => {
    // Read current theme from HTML attribute
    const html = document.documentElement
    setCurrentTheme(html.getAttribute('data-theme') || 'medisch')
  }, [])

  const handleThemeChange = (slug: string) => {
    document.documentElement.setAttribute('data-theme', slug)
    setCurrentTheme(slug)
    localStorage.setItem('theme', slug) // Persist choice
  }

  return (
    <div className="theme-switcher">
      <label>Select Vertical:</label>
      <select value={currentTheme} onChange={(e) => handleThemeChange(e.target.value)}>
        {themes.map((theme) => (
          <option key={theme.slug} value={theme.slug}>
            {theme.icon} {theme.name}
          </option>
        ))}
      </select>
    </div>
  )
}

async function seedThemes() {
  const payload = await getPayloadHMR({ config: configPromise })

  console.log('🌱 Seeding themes...')

  for (const theme of defaultThemes) {
    const existing = await payload.find({
      collection: 'themes',
      where: { slug: { equals: theme.slug } },
    })

    if (existing.docs.length === 0) {
      await payload.create({
        collection: 'themes',
        data: theme,
      })
      console.log(`✅ Created theme: ${theme.name}`)
    } else {
      console.log(`⏭️  Skipped (already exists): ${theme.name}`)
    }
  }

  console.log('✨ Theme seeding complete!')
}

Migration must include:

Advanced Usage & Best Practices

Per-Client Theme Assignment

For multi-tenant SaaS, assign theme per client/tenant:

// src/collections/Clients.ts
{
  name: 'theme',
  type: 'relationship',
  relationTo: 'themes',
  required: true,
  admin: {
    description: 'Visual theme for this client (affects all their pages)',
  },
}

// Fetch client's theme in layout:
const client = await payload.findByID({
  collection: 'clients',
  id: clientId,
})

const theme = await payload.findByID({
  collection: 'themes',
  id: client.theme,
})

// Apply theme:
<html data-theme={theme.slug}>

Theme Preview Mode

Allow admins to preview any theme without publishing:

// Add ?theme=beauty to URL
const searchParams = useSearchParams()
const previewTheme = searchParams.get('theme')

useEffect(() => {
  if (previewTheme) {
    document.documentElement.setAttribute('data-theme', previewTheme)
  }
}, [previewTheme])

Component Isolation

If a component needs to ignore theme and always use base colors:

// Use explicit color instead of CSS variable:
<button style={{ background: '#00897B' }}>Always Teal</button>

// Or create a base theme class:
.base-theme {
  --teal: #00897B !important;
}

<div className="base-theme">
  <Button>Always uses base teal</Button>
</div>

Performance Optimization

  • Cache themes: Use React Server Components to fetch themes once at build time
  • Static generation: For known verticals, generate static [data-theme="x"] CSS at build
  • CSS purging: Remove unused theme CSS if only 1-2 verticals are active

Troubleshooting

  • Theme not applying: Check <html data-theme="..."> attribute in browser DevTools
  • Colors not changing: Verify CSS variables exist in <style> tag in <head>
  • Some components not themed: Search for hardcoded colors (e.g., background: #00897B instead of var(--teal))
  • Theme switching laggy: Use CSS transitions on :root for smooth color changes: 
    :root {
  transition: background-color 0.2s, color 0.2s;
}
