TypeScript src/branches/blocks/ecommerce/product/CategoryGrid/utils.ts 
import type { Category, CategoryCardData } from './types'

/**
 * Fetches categories based on the selected source mode
 */
export async function fetchCategories(
  source: 'manual' | 'top' | 'recent' | 'all',
  options: {
    categoryIds?: string[]
    maxCategories?: number
    hideEmpty?: boolean
  }
): Promise {
  const { categoryIds, maxCategories = 8, hideEmpty = true } = options

  let url = '/api/categories?'

  if (source === 'manual' && categoryIds?.length) {
    url += `ids=${categoryIds.join(',')}`
  } else if (source === 'top') {
    url += `sort=-productCount&limit=${maxCategories}`
  } else if (source === 'recent') {
    url += `sort=-createdAt&limit=${maxCategories}`
  } else if (source === 'all') {
    url += `limit=${maxCategories}`
  }

  if (hideEmpty) {
    url += '&productCount[greater_than]=0'
  }

  const response = await fetch(url)
  const data = await response.json()

  return data.docs || []
}

/**
 * Transforms category data for rendering
 */
export function transformCategoryData(category: Category): CategoryCardData {
  const thumbnail = typeof category.thumbnail === 'object' && category.thumbnail
    ? {
        url: category.thumbnail.url || '',
        alt: category.thumbnail.alt || category.name,
      }
    : null

  return {
    id: category.id,
    name: category.name,
    slug: category.slug,
    thumbnail,
    productCount: category.productCount,
  }
}

/**
 * Generates gradient backgrounds for categories without images
 */
const GRADIENT_PRESETS = [
  'linear-gradient(135deg, #00897B, #26A69A)', // teal
  'linear-gradient(135deg, #EC4899, #F472B6)', // pink
  'linear-gradient(135deg, #7C3AED, #9333EA)', // purple
  'linear-gradient(135deg, #F59E0B, #FCD34D)', // amber
  'linear-gradient(135deg, #2196F3, #64B5F6)', // blue
  'linear-gradient(135deg, #00C853, #69F0AE)', // green
  'linear-gradient(135deg, #FF6B6B, #FF9999)', // coral
  'linear-gradient(135deg, #0A1628, #121F33)', // navy
]

export function getGradientForCategory(categoryId: string): string {
  const index = parseInt(categoryId.slice(-2), 16) % GRADIENT_PRESETS.length
  return GRADIENT_PRESETS[index]
}

7 Troubleshooting Guide
Categories not appearing in grid
Cause: hideEmptyCategories is true and all selected categories have 0 products.
Fix: Either add products to categories, set hideEmptyCategories to false, or select categories with existing products.
Manual categories not showing after selection
Cause: categorySource is not set to 'manual', or relationship field is not saving correctly.
Fix: Verify categorySource is 'manual' in database. Check Payload relationship field configuration has correct relationTo: 'categories' and hasMany: true.
Grid not responsive on mobile
Cause: Tailwind breakpoint classes not applying correctly, or mobileColumns config ignored.
Fix: Ensure Tailwind JIT mode is enabled. Verify gridCols object in CategoryGridClient has correct grid-cols-* classes. Check browser DevTools for applied CSS classes at different breakpoints.
Hover scale effect too aggressive
Cause: scale(1.03) feels too pronounced on large cards.
Fix: Reduce scale to 1.02 in CategoryGridClient hover classes. Alternatively, add will-change-transform CSS property and adjust transition duration to 150ms for smoother animation.
Product count incorrect or outdated
Cause: productCount field on Category collection not updated when products added/removed.
Fix: Add afterChange hook to Products collection that increments/decrements category.productCount. Or create scheduled job to recalculate counts nightly using SQL COUNT(*) queries.
Top categories mode returns wrong order
Cause: API endpoint not sorting by productCount descending correctly.
Fix: In fetchCategories util, verify sort parameter is '-productCount' (minus sign for descending). Check API endpoint accepts sort param and applies it to database query.
Category images not loading or showing broken
Cause: Media relationship not populated with depth: 1, or image URLs missing protocol.
Fix: In server component, ensure payload.find includes depth: 1 to populate thumbnail media. In transformCategoryData, check thumbnail.url exists before returning. Add fallback gradient for null thumbnails.
Dark mode text invisible on navy background
Cause: Text color not inverting when backgroundColor is 'navy'.
Fix: In Component.tsx, ensure isDark constant correctly detects backgroundColor === 'navy'. Verify conditional classes apply 'text-white' and 'text-white/70' instead of 'text-[#0A1628]' and 'text-[#64748B]'.
