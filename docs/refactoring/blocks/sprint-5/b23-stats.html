Step 2: TypeScript Type Definitions

// src/types/blocks.ts export interface Stat { value: string label: string description?: string icon?: string } export interface StatsBlock { blockType: 'stats' id?: string title?: string description?: string columns: '2' | '3' | '4' stats: Stat[] backgroundColor: 'white' | 'grey' | 'teal' | 'navy' } // Type guard to check if a block is a StatsBlock export function isStatsBlock(block: any): block is StatsBlock { return block?.blockType === 'stats' }
Step 3: React Component Implementation

Create src/branches/blocks/shared/Stats/Component.tsx to render the block:

// src/branches/blocks/shared/Stats/Component.tsx import React from 'react' import { StatsBlock } from '@/types/blocks' import { cn } from '@/lib/utils' interface StatsProps { block: StatsBlock } export const Stats: React.FC<StatsProps> = ({ block }) => { const { title, description, columns, stats, backgroundColor } = block // Map columns to grid classes const gridClass = { '2': 'grid-cols-1 md:grid-cols-2', '3': 'grid-cols-1 md:grid-cols-2 lg:grid-cols-3', '4': 'grid-cols-1 md:grid-cols-2 lg:grid-cols-4', }[columns] // Map background to classes const bgClass = { white: 'bg-white', grey: 'bg-grey-light', teal: 'bg-gradient-to-br from-teal to-teal-dark relative overflow-hidden', navy: 'bg-gradient-to-br from-navy to-navy-light relative overflow-hidden', }[backgroundColor] const isDark = backgroundColor === 'teal' || backgroundColor === 'navy' return ( <section className={cn('py-16 md:py-20', bgClass)}> {/* Decorative glow for gradient backgrounds */} {isDark && ( <div className="absolute -top-1/2 -right-[10%] w-[400px] h-[400px] bg-radial-glow rounded-full pointer-events-none opacity-10" /> )} <div className="container relative z-10"> {/* Header */} {(title || description) && ( <div className="text-center mb-12 md:mb-16"> {title && ( <h2 className={cn( 'font-display text-4xl md:text-5xl mb-3', isDark ? 'text-white' : 'text-navy' )}> {title} </h2> )} {description && ( <p className={cn( 'text-base max-w-2xl mx-auto', isDark ? 'text-white/70' : 'text-grey-dark' )}> {description} </p> )} </div> )} {/* Stats Grid */} <div className={cn('grid gap-8 md:gap-12', gridClass)}> {stats.map((stat, index) => ( <div key={index} className="text-center"> {/* Icon */} {stat.icon && ( <div className={cn( 'w-14 h-14 mx-auto mb-4 rounded-xl flex items-center justify-center text-3xl', isDark ? 'bg-white/15' : 'bg-teal-glow' )}> {/* Render emoji or Lucide icon */} {stat.icon.length <= 2 ? ( <span>{stat.icon}</span> ) : ( <i data-lucide={stat.icon} className="w-7 h-7" /> )} </div> )} {/* Value */} <span className={cn( 'font-display text-5xl md:text-6xl block mb-2', isDark ? 'text-white' : 'text-navy' )}> {stat.value} </span> {/* Label */} <span className={cn( 'text-sm font-semibold block', isDark ? 'text-white/80' : 'text-grey-dark' )}> {stat.label} </span> {/* Description */} {stat.description && ( <span className={cn( 'text-xs mt-1.5 block max-w-[220px] mx-auto', isDark ? 'text-white/60' : 'text-grey-mid' )}> {stat.description} </span> )} </div> ))} </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Stats Block" is visible
2. All fields are editable: Can add title, description, select columns (2/3/4), and configure stats array
3. Stats array validation: Minimum 2 stats required, maximum 4 allowed
4. Background variants render: Test all 4 backgrounds (white, grey, teal gradient, navy gradient)
5. Grid layouts work: Verify 2/3/4 column grids display correctly on desktop and are responsive on mobile
6. Icons display correctly: Test both emoji icons (🏥) and Lucide icons (activity, award, etc.)
7. Typography matches design: Values use DM Serif Display font at correct size (48px desktop)
Step 6: Database Migration

⚠️ IMPORTANT: After creating the Stats block, generate a database migration:

# Generate migration for Stats block npx payload migrate:create add_stats_block # Review generated migration file in src/migrations/ # It should contain ALTER TABLE statements for pages_blocks table # Adding columns for: # - title (varchar, nullable) # - description (text, nullable) # - columns (varchar, default '3') # - background_color (varchar, default 'white') # - stats (jsonb) — stores array of stat objects # Expected schema changes: # pages_blocks.title → varchar (nullable) # pages_blocks.description → text (nullable) # pages_blocks.columns → varchar (default '3') # pages_blocks.background_color → varchar (default 'white') # pages_blocks.stats → jsonb (array of Stat objects) # Apply migration to database npx payload migrate # Verify in database (PostgreSQL example) psql -d your_database -c "SELECT column_name, data_type FROM information_schema.columns WHERE table_name = 'pages_blocks' AND column_name IN ('title', 'description', 'columns', 'background_color', 'stats');" # Expected output: # column_name | data_type # ------------------+----------- # title | varchar # description | text # columns | varchar # background_color | varchar # stats | jsonb
Step 7: Common Issues & Solutions
Issue Solution
Stats array not saving Check migration applied: npx payload migrate → verify stats column exists and is type jsonb
Grid layout not responsive Verify Tailwind breakpoints: md:grid-cols-2, lg:grid-cols-3, etc. Check Tailwind config includes these classes
Lucide icons not rendering Ensure lucide.createIcons() called after mount. Add to useEffect in layout or component
Gradient backgrounds missing glow Check pseudo-element CSS: ::before with radial-gradient applied and positioned correctly
Font not DM Serif Display Verify font imported: @import url('...') in globals.css and font-display class uses font-family: var(--font-display)
Min/max rows validation failing Check Payload config: minRows: 2, maxRows: 4 in stats array field definition
Colors not matching design Verify CSS custom properties in Tailwind config: --navy, --teal, --grey-dark, etc.
TypeScript error on block type Import correct type: import { StatsBlock } from '@/types/blocks' and ensure blockType: 'stats' matches
Step 8: Register Block in Payload Config

Add the Stats block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { Stats } from '@/branches/blocks/shared/Stats/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ Stats, // ← Add Stats block here // Hero, // Features, // CTA, // etc... ], }, ], }
