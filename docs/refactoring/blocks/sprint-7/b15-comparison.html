Usage Guidelines

  • Featured: Light teal background (rgba(0,137,123,0.04)) on featured column

STEP 2 TypeScript Type Definitions

Define TypeScript interfaces for type-safe development with the Comparison Table block.

TypeScript src/types/blocks.ts 
// src/types/blocks.ts
export interface ComparisonColumn {
  name: string
  price: string
  featured: boolean
  id?: string
}

export interface ComparisonValue {
  type: 'check' | 'x' | 'text'
  text?: string
  id?: string
}

export interface ComparisonRow {
  feature: string
  values: ComparisonValue[]
  id?: string
}

export interface ComparisonBlock {
  blockType: 'comparison'
  id?: string
  title: string
  description?: string
  columns: ComparisonColumn[]
  rows: ComparisonRow[]
}

// Type Guards
export function isComparisonBlock(block: any): block is ComparisonBlock {
  return block?.blockType === 'comparison'
}

export function validateComparisonData(block: ComparisonBlock): boolean {
  // Validate that each row has values matching number of columns
  const columnCount = block.columns.length
  return block.rows.every(row => row.values.length === columnCount)
}

// Helper Functions
export function getFeaturedColumnIndex(columns: ComparisonColumn[]): number | null {
  const index = columns.findIndex(col => col.featured)
  return index >= 0 ? index : null
}
STEP 3 React Component Implementation

Create the Next.js React component with Tailwind CSS styling and responsive layout.

TypeScript React src/branches/blocks/navigation-info/Comparison/Component.tsx 
// src/branches/blocks/navigation-info/Comparison/Component.tsx
'use client'

import React from 'react'
import { Check, X } from 'lucide-react'
import type { ComparisonBlock, ComparisonColumn, ComparisonRow, ComparisonValue } from '@/types/blocks'

interface ComparisonProps {
  block: ComparisonBlock
}

const ComparisonCell: React.FC<{ value: ComparisonValue }> = ({ value }) => {
  switch (value.type) {
    case 'check':
      return <Check className="w-5 h-5 text-green-600" />
    case 'x':
      return <X className="w-5 h-5 text-grey-mid" />
    case 'text':
      return <span className="text-sm text-navy">{value.text}</span>
    default:
      return null
  }
}

export const Comparison: React.FC<ComparisonProps> = ({ block }) => {
  const { title, description, columns, rows } = block

  // Calculate grid template columns
  const gridCols = `200px repeat(${columns.length}, 1fr)`

  return (
    <section className="py-16 md:py-20">
      <div className="container mx-auto px-6">
        {/* Header */}
        {(title || description) && (
          <div className="text-center mb-12">
            {title && <h2 className="font-serif text-3xl md:text-4xl text-navy mb-3">{title}</h2>}
            {description && <p className="text-base text-grey-dark max-w-2xl mx-auto">{description}</p>}
          </div>
        )}

        {/* Comparison Table */}
        <div className="max-w-4xl mx-auto bg-white border border-grey rounded-2xl overflow-hidden">
          {/* Header Row */}
          <div className="grid bg-grey-light border-b border-grey" style={{ gridTemplateColumns: gridCols }}>
            <div className="bg-white"></div>
            {columns.map((column, index) => (
              <div
                key={column.id || index}
                className={`p-5 text-center ${column.featured ? 'bg-teal-glow' : ''}`}
              >
                <div className={`font-bold text-base mb-1 ${column.featured ? 'text-teal' : 'text-navy'}`}>
                  {column.name}
                </div>
                <div className="text-sm text-grey-mid">{column.price}</div>
              </div>
            ))}
          </div>

          {/* Feature Rows */}
          {rows.map((row, rowIndex) => (
            <div key={row.id || rowIndex} className="grid border-t border-grey" style={{ gridTemplateColumns: gridCols }}>
              {/* Feature Label */}
              <div className="bg-grey-light px-4 py-4 flex items-center font-semibold text-sm text-navy">
                {row.feature}
              </div>

              {/* Values */}
              {row.values.map((value, valueIndex) => {
                const isFeatured = columns[valueIndex]?.featured
                return (
                  <div
                    key={value.id || valueIndex}
                    className={`px-4 py-4 flex items-center justify-center ${isFeatured ? 'bg-teal-glow' : ''}`}
                  >
                    <ComparisonCell value={value} />
                  </div>
                )
              })}
            </div>
          ))}
        </div>

        {/* Mobile View: Stack Cards */}
        <div className="lg:hidden mt-8 space-y-4">
          {columns.map((column, colIndex) => (
            <div
              key={column.id || colIndex}
              className={`bg-white border rounded-xl overflow-hidden ${column.featured ? 'border-teal shadow-md' : 'border-grey'}`}
            >
              {/* Column Header */}
              <div className={`p-4 ${column.featured ? 'bg-teal-glow' : 'bg-grey-light'}`}>
                <div className={`font-bold text-lg ${column.featured ? 'text-teal' : 'text-navy'}`}>{column.name}</div>
                <div className="text-sm text-grey-mid">{column.price}</div>
              </div>

              {/* Features */}
              <div className="divide-y divide-grey">
                {rows.map((row, rowIndex) => (
                  <div key={row.id || rowIndex} className="flex items-center justify-between p-4">
                    <span className="font-medium text-sm text-navy">{row.feature}</span>
                    <ComparisonCell value={row.values[colIndex]} />
                  </div>
                ))}
              </div>
            </div>
          ))}
        </div>
      </div>
    </section>
  )
}
Component Features
  • Responsive: Desktop shows full table, mobile stacks columns as cards
  • Featured Column: Automatically applies teal background to featured plan
  • Dynamic Grid: Grid columns adapt to 2-4 comparison columns
  • Icon Support: Lucide icons for check/x marks with proper colors

  _order INTEGER NOT NULL,
  _parent_id INTEGER NOT NULL REFERENCES pages_blocks_comparison(id) ON DELETE CASCADE,
  id SERIAL PRIMARY KEY,
  name VARCHAR NOT NULL,
  price VARCHAR NOT NULL,
  featured BOOLEAN DEFAULT false
);

-- pages_blocks_comparison_rows
CREATE TABLE pages_blocks_comparison_rows (
  _order INTEGER NOT NULL,
  _parent_id INTEGER NOT NULL REFERENCES pages_blocks_comparison(id) ON DELETE CASCADE,
  id SERIAL PRIMARY KEY,
  feature VARCHAR NOT NULL
);

-- pages_blocks_comparison_rows_values
CREATE TABLE pages_blocks_comparison_rows_values (
  _order INTEGER NOT NULL,
  _parent_id INTEGER NOT NULL REFERENCES pages_blocks_comparison_rows(id) ON DELETE CASCADE,
  id SERIAL PRIMARY KEY,
  type VARCHAR NOT NULL CHECK (type IN ('check', 'x', 'text')),
  text VARCHAR
);

CREATE INDEX idx_comparison_parent ON pages_blocks_comparison(_parent_id);

STEP 7 Troubleshooting Guide
⚠️ Common Issues & Solutions
❌ Table columns are misaligned
Solution: Ensure grid-template-columns is properly set with dynamic column count. Check that CSS uses gridTemplateColumns: `200px repeat(${columns.length}, 1fr)`.
❌ Row values don't match column count
Solution: Implement validation in admin panel. Add server-side validation using validateComparisonData() helper to ensure each row has exactly columns.length values.
❌ Featured column highlighting not working
Solution: Verify column.featured boolean is correctly saved. Check CSS class bg-teal-glow is applied to featured cells. Ensure only ONE column is marked as featured.
❌ Icons (check/x) not rendering
Solution: Verify lucide-react is installed (npm install lucide-react). Check import statement: import { Check, X } from 'lucide-react'. Ensure ComparisonCell component is rendering correctly.
❌ Mobile view not showing stacked cards
Solution: Check that mobile view div.lg:hidden is present in component. Verify Tailwind responsive classes are working. Test with browser dev tools mobile viewport.
❌ Migration fails with foreign key errors
Solution: Ensure parent tables are created before child tables. Check ON DELETE CASCADE is set. Run npx payload migrate:status to check pending migrations.
❌ Text values showing "undefined"
Solution: Verify value.text is populated when value.type === 'text'. Add admin condition to show text input only when type is 'text'. Check null/undefined handling in ComparisonCell.
❌ Table overflows on small screens
Solution: Hide desktop table view on mobile with @media (max-width: 768px). Show stacked cards instead. Consider adding horizontal scroll for tablet sizes if needed.
