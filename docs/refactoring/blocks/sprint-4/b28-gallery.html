Step 3: React Component Implementation

Create src/branches/blocks/shared/ImageGallery/Component.tsx to render the block:

// src/branches/blocks/shared/ImageGallery/Component.tsx 'use client' import React, { useState } from 'react' import Image from 'next/image' import { ImageGalleryBlock, GalleryImage } from '@/types/blocks' import { Media } from '@/types/payload-types' import { cn } from '@/lib/utils' import { X } from 'lucide-react' interface ImageGalleryProps { block: ImageGalleryBlock } export const ImageGallery: React.FC<ImageGalleryProps> = ({ block }) => { const { title, layout, columns, images, enableLightbox, spacing } = block const [lightboxIndex, setLightboxIndex] = useState<number | null>(null) // Map columns to Tailwind grid classes const columnClass = { '2': 'grid-cols-1 md:grid-cols-2', '3': 'grid-cols-1 md:grid-cols-2 lg:grid-cols-3', '4': 'grid-cols-1 md:grid-cols-2 lg:grid-cols-4', }[columns] // Map spacing to gap classes const gapClass = { tight: 'gap-2', normal: 'gap-4', loose: 'gap-6', }[spacing] // Get media URL helper const getMediaUrl = (media: string | Media): string => { if (typeof media === 'string') return media return media?.url || '' } // Get media alt helper const getMediaAlt = (media: string | Media, fallback?: string): string => { if (typeof media === 'string') return fallback || 'Gallery image' return media?.alt || fallback || 'Gallery image' } // Open lightbox const openLightbox = (index: number) => { if (enableLightbox) { setLightboxIndex(index) document.body.style.overflow = 'hidden' } } // Close lightbox const closeLightbox = () => { setLightboxIndex(null) document.body.style.overflow = '' } // Navigate lightbox const nextImage = () => { if (lightboxIndex !== null) { setLightboxIndex((lightboxIndex + 1) % images.length) } } const prevImage = () => { if (lightboxIndex !== null) { setLightboxIndex((lightboxIndex - 1 + images.length) % images.length) } } // Keyboard navigation React.useEffect(() => { const handleKeyDown = (e: KeyboardEvent) => { if (lightboxIndex === null) return if (e.key === 'Escape') closeLightbox() if (e.key === 'ArrowRight') nextImage() if (e.key === 'ArrowLeft') prevImage() } window.addEventListener('keydown', handleKeyDown) return () => window.removeEventListener('keydown', handleKeyDown) }, [lightboxIndex]) return ( <section className="py-12 md:py-16"> <div className="container"> {title && ( <h2 className="font-display text-3xl text-navy text-center mb-8"> {title} </h2> )} <div className={cn( 'grid', columnClass, gapClass, layout === 'masonry' && 'auto-rows-auto' )} > {images.map((item, index) => { const imageUrl = getMediaUrl(item.image) const imageAlt = getMediaAlt(item.image, item.alt) return ( <div key={item.id || index} className="group relative overflow-hidden rounded-xl cursor-pointer transition-all duration-200 hover:-translate-y-1 hover:shadow-md" onClick={() => openLightbox(index)} > <div className="relative aspect-[4/3] bg-grey-light"> <Image src={imageUrl} alt={imageAlt} fill className="object-cover" sizes="(max-width: 768px) 100vw, (max-width: 1024px) 50vw, 33vw" /> {item.caption && ( <div className="absolute inset-x-0 bottom-0 bg-gradient-to-t from-navy/80 to-transparent p-4 opacity-0 group-hover:opacity-100 transition-opacity"> <p className="text-white text-sm">{item.caption}</p> </div> )} </div> </div> ) })} </div> {/* Lightbox */} {enableLightbox && lightboxIndex !== null && ( <div className="fixed inset-0 z-modal bg-navy/95 backdrop-blur-md flex items-center justify-center p-10" onClick={closeLightbox} > <button onClick={closeLightbox} className="absolute top-4 right-4 w-10 h-10 bg-white/10 hover:bg-white/20 border border-white/20 rounded-lg flex items-center justify-center transition-colors" > <X className="w-5 h-5 text-white" /> </button> <div className="relative max-w-[90vw] max-h-[90vh]" onClick={(e) => e.stopPropagation()}> <Image src={getMediaUrl(images[lightboxIndex].image)} alt={getMediaAlt(images[lightboxIndex].image, images[lightboxIndex].alt)} width={1200} height={900} className="max-w-full max-h-[90vh] object-contain rounded-lg" /> {images[lightboxIndex].caption && ( <p className="text-white text-center mt-4 text-sm"> {images[lightboxIndex].caption} </p> )} </div> </div> )} </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Image Gallery" is visible
2. Image upload works: Click "Add Image" → upload test image → verify thumbnail appears
3. Multiple images can be added: Add 6+ images → verify drag-to-reorder works
4. All layout variants render: Test grid vs masonry → verify visual difference
5. Column options work: Test 2, 3, 4 columns → verify responsive behavior
6. Lightbox functionality: Click image → lightbox opens → Esc/arrows work → close button works
7. Captions display correctly: Add caption → verify shows on hover and in lightbox
8. Responsive behavior: Test on mobile → verify grid collapses to 1 column
9. Image optimization: Check Network tab → verify Next.js Image optimization active
10. Accessibility: Test with screen reader → verify alt text announced
Step 6: Database Migration

⚠️ IMPORTANT: After creating the Image Gallery block, generate a database migration:

# Generate migration for Image Gallery block npx payload migrate:create add_image_gallery_block # Review generated migration file in src/migrations/ # It should contain ALTER TABLE statements for pages_blocks table # Adding columns for gallery configuration and images array # Expected schema changes: # - pages_blocks.title (type: varchar, nullable) # - pages_blocks.layout (type: varchar) — enum: 'grid', 'masonry' # - pages_blocks.columns (type: varchar) — enum: '2', '3', '4' # - pages_blocks.enable_lightbox (type: boolean, default: true) # - pages_blocks.spacing (type: varchar) — enum: 'tight', 'normal', 'loose' # Additionally, a junction table for images array: # - pages_blocks_images (id, parent_id, image_id, caption, alt, _order) # Apply migration to database npx payload migrate # Verify in database (PostgreSQL example) psql -d your_database -c "SELECT column_name, data_type FROM information_schema.columns WHERE table_name = 'pages_blocks' AND column_name IN ('title', 'layout', 'columns', 'enable_lightbox', 'spacing');" # Expected output: # column_name | data_type # ------------------+----------- # title | varchar # layout | varchar # columns | varchar # enable_lightbox | boolean # spacing | varchar # Check images junction table psql -d your_database -c "\d pages_blocks_images" # Expected structure: # id (primary key, auto-increment) # parent_id (foreign key to pages_blocks.id) # image_id (foreign key to media.id) # caption (varchar, nullable) # alt (varchar, nullable) # _order (integer) — for array ordering
Step 7: Common Issues & Solutions
Issue Solution
Images not displaying Check relationTo: 'media' matches your Media collection slug. Verify images uploaded to correct collection.
Lightbox not opening Ensure component is marked 'use client' — lightbox requires client-side JavaScript for state management.
Grid layout not responsive Check Tailwind classes: grid-cols-1 md:grid-cols-2 lg:grid-cols-3 ensures mobile-first responsive behavior.
Images too large / slow loading Verify using Next.js <Image> component with proper sizes prop for responsive optimization.
Masonry layout not working Masonry requires CSS Grid or third-party library (e.g., react-masonry-css). Example implementation uses CSS Grid with auto-rows-auto.
Keyboard navigation not working Check useEffect dependency array includes lightboxIndex to re-register event listeners.
Caption not showing on hover Verify group and group-hover:opacity-100 classes applied. Check caption has content in admin.
TypeScript error on Media type Import Media from @/types/payload-types (auto-generated by Payload). Run payload generate:types if missing.
Database error on save Run migration: npx payload migrate → verify pages_blocks_images table exists.
Alt text not accessible Ensure alt field passed to <Image alt={} /> prop. Test with screen reader (NVDA/JAWS).
Step 8: Register Block in Payload Config

Add the Image Gallery block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { ImageGallery } from '@/branches/blocks/shared/ImageGallery/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ ImageGallery, // ← Add Image Gallery block here // Hero, // Content, // Features, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { ImageGallery } from '@/branches/blocks/shared/ImageGallery/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'gallery': return <ImageGallery key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'content': return <Content key={index} block={block} /> default: return null } })} </main> ) }
