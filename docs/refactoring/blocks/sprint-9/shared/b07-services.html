Step 3: React Component Implementation

Create src/branches/blocks/shared/Services/Component.tsx to render the block:

// src/branches/blocks/shared/Services/Component.tsx 'use client' import React from 'react' import Link from 'next/link' import { ServicesBlock, iconColorMap } from '@/types/blocks' import { cn } from '@/lib/utils' import * as LucideIcons from 'lucide-react' interface ServicesProps { block: ServicesBlock } export const Services: React.FC<ServicesProps> = ({ block }) => { const { subtitle, title, description, columns, services } = block // Map columns to Tailwind grid classes const gridClass = { '2': 'md:grid-cols-2', '3': 'md:grid-cols-3', '4': 'md:grid-cols-2 lg:grid-cols-4', }[columns] // Helper to get Lucide icon component by name const getIcon = (iconName: string) => { const IconComponent = LucideIcons[ iconName as keyof typeof LucideIcons ] as React.ComponentType<LucideIcons.LucideProps> return IconComponent || LucideIcons.Package // Fallback to Package icon } return ( <section className="py-16 md:py-20"> <div className="container"> {/* Section Header (optional) */} {(title || subtitle || description) && ( <div className="max-w-3xl mx-auto text-center mb-12 md:mb-16"> {subtitle && ( <div className="text-xs font-bold uppercase tracking-wider text-teal mb-2"> {subtitle} </div> )} {title && ( <h2 className="font-display text-3xl md:text-4xl text-navy mb-3"> {title} </h2> )} {description && ( <p className="text-base text-grey-dark leading-relaxed"> {description} </p> )} </div> )} {/* Services Grid */} <div className={cn('grid gap-5', gridClass)}> {services.map((service, index) => { const Icon = getIcon(service.icon) const colorTheme = iconColorMap[service.iconColor] return ( <div key={service.id || index} className="group bg-white border border-grey rounded-xl p-6 transition-all duration-200 hover:shadow-md hover:-translate-y-1 cursor-pointer" > {/* Icon */} <div className={cn( 'w-12 h-12 rounded-xl flex items-center justify-center mb-4', colorTheme.bg )}> <Icon className={cn('w-6 h-6', colorTheme.color)} /> </div> {/* Title */} <h3 className="text-base font-bold text-navy mb-2"> {service.title} </h3> {/* Description */} <p className="text-sm text-grey-dark leading-relaxed mb-3"> {service.description} </p> {/* Link (optional) */} {service.link && ( <Link href={service.link} className={cn( 'inline-flex items-center gap-1 text-sm font-semibold transition-all group-hover:gap-2', colorTheme.color )} > {service.linkText || 'Meer info'} <LucideIcons.ArrowRight className="w-3.5 h-3.5" /> </Link> )} </div> ) })} </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Services Block" is visible
2. All fields render correctly: Add Services Block → verify all fields present (subtitle, title, description, columns, services array)
3. Service array works: Click "+ Add Service" → fill in fields → verify all inputs working (icon, color, title, description, link)
4. Grid columns apply: Test all 3 column options (2, 3, 4) → verify CSS grid classes applied correctly on frontend
5. Lucide icons render: Try different icon names (package, code, rocket) → verify icons display correctly on frontend
6. Icon colors work: Test all 6 color options → verify background and icon colors match design system
7. Links are optional: Create service without link → verify no link displayed. Add link → verify arrow + hover animation works
8. Responsive layout: Test on mobile (< 768px) → grid collapses to 1 column, cards remain readable
9. Hover states work: Hover over service cards → shadow elevation + translateY animation triggers
10. TypeScript compilation: Run npm run build → no TypeScript errors

Step 7: Common Issues & Solutions
Issue Solution
Icons not displaying Check lucide-react installed: npm install lucide-react. Verify icon name matches lucide.dev library (case-sensitive: "package", not "Package")
Fallback icon always showing Icon name mismatch. Check getIcon() helper → ensure icon exists in Lucide library. Use browser DevTools to inspect component props.
Grid not responsive Missing Tailwind breakpoints. Verify gridClass mapping includes md: and lg: prefixes. Test with browser DevTools responsive mode.
Icon colors not applying Check iconColorMap imported correctly. Ensure Tailwind config includes custom color utilities (teal-50, blue-50, etc). Run npm run build to regenerate CSS.
Services array empty on save Database migration not applied. Run npx payload migrate → verify pages_blocks_services table exists in database.
Link hover animation not working Check transition classes on Link component. Ensure group-hover:gap-2 class present. Verify parent has group class.
TypeScript error on iconColorMap Ensure IconColor type defined in types file. Check Tailwind class strings are correct (no typos in color names).
Section header not showing Conditional render checks (title || subtitle || description) → at least one must have value. Verify fields saved in admin panel.
Step 8: Register Block in Payload Config

Add the Services block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { Services } from '@/branches/blocks/shared/Services/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ Services, // ← Add Services block here // Hero, // Features, // Content, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { Services } from '@/branches/blocks/shared/Services/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'services': return <Services key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'content': return <Content key={index} block={block} /> default: return null } })} </main> ) }
