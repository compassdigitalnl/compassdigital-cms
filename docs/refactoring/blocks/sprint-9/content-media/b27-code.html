Step 2: TypeScript Type Definitions

Create src/types/blocks.ts for type safety:

// src/types/blocks.ts export type CodeLanguage = | 'javascript' | 'typescript' | 'html' | 'css' | 'python' | 'bash' | 'json' | 'jsx' | 'tsx' | 'graphql' | 'sql' | 'markdown' | 'yaml' | 'php' | 'ruby' | 'go' | 'rust' | 'swift' | 'kotlin' | 'c' | 'cpp' | 'csharp' | 'java' | 'text' export interface CodeBlock { blockType: 'code' id?: string code: string language: CodeLanguage showLineNumbers: boolean filename?: string highlightLines?: string } // Type guard to check if a block is a CodeBlock export function isCodeBlock(block: any): block is CodeBlock { return block?.blockType === 'code' } // Helper to parse highlight lines string to array of numbers export function parseHighlightLines(highlightLines?: string): number[] { if (!highlightLines) return [] const lines: number[] = [] const parts = highlightLines.split(',') parts.forEach((part) => { const trimmed = part.trim() if (trimmed.includes('-')) { // Range (e.g., "3-5") const [start, end] = trimmed.split('-').map(Number) for (let i = start; i <= end; i++) { lines.push(i) } } else { // Single line const lineNum = Number(trimmed) if (!isNaN(lineNum)) { lines.push(lineNum) } } }) return lines }
Step 3: React Component Implementation

Create src/branches/blocks/shared/Code/Component.tsx to render the block:

// src/branches/blocks/shared/Code/Component.tsx 'use client' import React, { useState } from 'react' import { Prism as SyntaxHighlighter } from 'react-syntax-highlighter' import { oneDark } from 'react-syntax-highlighter/dist/cjs/styles/prism' import { Copy, Check } from 'lucide-react' import { CodeBlock, parseHighlightLines } from '@/types/blocks' import { cn } from '@/lib/utils' interface CodeProps { block: CodeBlock } export const Code: React.FC<CodeProps> = ({ block }) => { const { code, language, showLineNumbers, filename, highlightLines } = block const [copied, setCopied] = useState(false) const handleCopy = async () => { try { await navigator.clipboard.writeText(code) setCopied(true) setTimeout(() => setCopied(false), 2000) } catch (err) { console.error('Failed to copy code:', err) } } // Parse highlight lines const highlightedLines = parseHighlightLines(highlightLines) // Custom style with line highlighting const customStyle = { ...oneDark, 'code[class*="language-"]': { ...oneDark['code[class*="language-"]'], background: 'transparent', }, 'pre[class*="language-"]': { ...oneDark['pre[class*="language-"]'], background: '#1e1e2e', margin: 0, padding: '16px', }, } return ( <section className="py-8 md:py-12"> <div className="container"> <div className="max-w-4xl mx-auto"> <div className="rounded-xl overflow-hidden shadow-lg border border-gray-800"> {/* Header */} <div className="bg-[#181825] px-4 py-3 border-b border-white/5 flex items-center justify-between"> <div className="flex items-center gap-3"> {/* Traffic lights */} <div className="flex gap-1.5"> <div className="w-2.5 h-2.5 rounded-full bg-[#ff5f57]" /> <div className="w-2.5 h-2.5 rounded-full bg-[#febc2e]" /> <div className="w-2.5 h-2.5 rounded-full bg-[#28c840]" /> </div> {filename && ( <span className="text-xs font-mono text-gray-300 font-semibold"> {filename} </span> )} <span className="text-[9px] font-bold uppercase tracking-wider px-2 py-0.5 rounded bg-white/5 text-white/40"> {language} </span> </div> {/* Copy button */} <button onClick={handleCopy} className="px-3 py-1.5 text-xs font-semibold text-gray-300 bg-white/5 hover:bg-white/10 Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Code Block" is visible
2. Monaco editor loads: Click "Add Code Block" → code editor initializes with syntax highlighting
3. All languages work: Test switching between JavaScript, TypeScript, HTML, CSS, Python, etc. → editor updates language mode
4. Line numbers toggle works: Check/uncheck "Show Line Numbers" → preview updates accordingly
5. Copy button works: Click copy button on frontend → code copied to clipboard → button shows "Copied!" feedback
6. Highlight lines work: Enter "1,3-5" in highlightLines → lines 1, 3, 4, 5 have teal background
7. TypeScript compilation: Run npm run build → no TypeScript errors

Step 7: Common Issues & Solutions
Issue Solution
Syntax highlighter not working Check imports: react-syntax-highlighter installed? Run npm install react-syntax-highlighter and npm install -D @types/react-syntax-highlighter
Monaco editor not loading in admin Payload's code field uses Monaco by default. Verify field type is 'code' not 'textarea'. Check console for Monaco loader errors.
Copy button not working Check browser clipboard permissions. Use 'use client' directive in component. Verify navigator.clipboard.writeText is supported.
Line numbers not showing Verify showLineNumbers prop is passed to <SyntaxHighlighter>. Check CSS doesn't hide line numbers.
Highlight lines not working Check parseHighlightLines function. Verify lineProps callback in SyntaxHighlighter. Test with simple input like "1,2".
Code overflow on mobile Ensure SyntaxHighlighter parent has overflow-x: auto. Add responsive padding. Test with long lines.
Database error on save Run migration: npx payload migrate → check pages_blocks table has code, language, show_line_numbers, filename, highlight_lines columns
Unsupported language selected Verify language value matches Prism/SyntaxHighlighter supported languages. Check react-syntax-highlighter docs for full list.
Step 8: Register Block in Payload Config

Add the Code block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { Code } from '@/branches/blocks/shared/Code/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ Code, // ← Add Code block here // Hero, // Content, // Features, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { Code } from '@/branches/blocks/shared/Code/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'code': return <Code key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'content': return <Content key={index} block={block} /> default: return null } })} </main> ) }
