Step 3: React Component Implementation

Create src/branches/blocks/shared/Accordion/Component.tsx to render the block:

// src/branches/blocks/shared/Accordion/Component.tsx 'use client' import React, { useState } from 'react' import { RichText } from '@payloadcms/richtext-lexical/react' import { AccordionBlock } from '@/types/blocks' import { cn } from '@/lib/utils' import { ChevronDown } from 'lucide-react' interface AccordionProps { block: AccordionBlock } export const Accordion: React.FC<AccordionProps> = ({ block }) => { const { title, items, allowMultipleOpen } = block // Find default open item (first item with defaultOpen: true) const defaultOpenIndex = items.findIndex((item) => item.defaultOpen) // Track which items are open (use Set for multiple open support) const [openItems, setOpenItems] = useState<Set<number>>( new Set(defaultOpenIndex >= 0 ? [defaultOpenIndex] : []) ) const toggleItem = (index: number) => { setOpenItems((prev) => { const newSet = new Set(allowMultipleOpen ? prev : []) if (prev.has(index)) { // Close this item newSet.delete(index) } else { // Open this item newSet.add(index) } return newSet }) } return ( <section className="py-12 md:py-16"> <div className="container"> <div className="mx-auto max-w-3xl"> {title && ( <h2 className="font-display text-2xl md:text-3xl text-navy mb-6 text-center"> {title} </h2> )} <div className="space-y-2"> {items.map((item, index) => { const isOpen = openItems.has(index) return ( <div key={item.id || index} className={cn( 'border border-grey rounded-xl overflow-hidden transition-all duration-200', isOpen && 'bg-teal-glow border-teal', !isOpen && 'hover:border-teal-light' )} > {/* Accordion Header */} <button onClick={() => toggleItem(index)} className="w-full flex items-center justify-between px-5 py-4 text-left transition-all" aria-expanded={isOpen} aria-controls={`accordion-content-${index}`} > <span className="font-bold text-[15px] text-navy pr-4"> {item.title} </span> <ChevronDown className={cn( 'w-[18px] h-[18px] flex-shrink-0 transition-all duration-200', isOpen ? 'text-teal rotate-180' : 'text-grey-mid' )} /> </button> {/* Accordion Content */} <div id={`accordion-content-${index}`} className={cn( 'overflow-hidden transition-all duration-300 ease-out', isOpen ? 'max-h-[1000px] border-t border-grey' : 'max-h-0' )} > <div className="px-5 py-4 text-[13px] text-grey-dark leading-relaxed prose prose-sm max-w-none prose-p:mb-3 prose-ul:my-2 prose-li:my-1"> <RichText data={item.content} /> </div> </div> </div> ) })} </div> </div> </div> </section> ) }

// CSS-only Accordion (Progressive Enhancement) export const AccordionNoJS: React.FC<AccordionProps> = ({ block }) => { const { title, items } = block return ( <section className="py-12 md:py-16"> <div className="container"> <div className="mx-auto max-w-3xl"> {title && ( <h2 className="font-display text-2xl md:text-3xl text-navy mb-6 text-center"> {title} </h2> )} <div className="space-y-2"> {items.map((item, index) => ( <details key={item.id || index} open={item.defaultOpen} className="group border border-grey rounded-xl overflow-hidden transition-all hover:border-teal-light open:bg-teal-glow open:border-teal" > <summary className="flex items-center justify-between px-5 py-4 cursor-pointer list-none [&::-webkit-details-marker]:hidden"> <span className="font-bold text-[15px] text-navy pr-4"> {item.title} </span> <svg className="w-[18px] h-[18px] flex-shrink-0 text-grey-mid transition-transform duration-200 group-open:rotate-180 group-open:text-teal" fill="none" stroke="currentColor" viewBox="0 0 24 24" > <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 9l-7 7-7-7" /> </svg> </summary> <div className="px-5 py-4 border-t border-grey text-[13px] text-grey-dark leading-relaxed prose prose-sm max-w-none"> <RichText data={item.content} /> </div> </details> ))} </div> </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Accordion Block" is visible
2. Array field works: Click "Add Accordion Items" → can add/remove/reorder items
3. Rich text editor loads: Each item has functional Lexical editor with formatting
4. Default open works: Check "Open by Default" on one item → that item renders expanded on frontend
5. Single open behavior: Click item → opens, click another → first closes (accordion mode)
6. Multiple open mode: Enable "Allow Multiple Items Open" → can open multiple items simultaneously
7. Animations smooth: Verify chevron rotation and content slide transitions work smoothly
Step 6: Database Migration

⚠️ IMPORTANT: After creating the Accordion block, generate a database migration:

# Generate migration for Accordion block npx payload migrate:create add_accordion_block # Review generated migration file in src/migrations/ # It should contain ALTER TABLE statements for pages_blocks table # Adding columns for accordion data # Expected schema changes: # - pages_blocks.title (type: varchar) — optional section title # - pages_blocks.allow_multiple_open (type: boolean) — behavior flag # - New table: pages_blocks_items (for accordion items array) # - id (uuid/serial) # - parent_id (foreign key to pages_blocks) # - title (varchar) — item title # - content (jsonb) — serialized Lexical state # - default_open (boolean) # - _order (integer) — array position # Apply migration to database npx payload migrate # Verify in database (PostgreSQL example) psql -d your_database -c "SELECT column_name, data_type FROM information_schema.columns WHERE table_name = 'pages_blocks_items';" # Expected output: # column_name | data_type # ----------------+----------- # id | uuid # parent_id | uuid # title | varchar # content | jsonb # default_open | boolean # _order | integer
Step 7: Common Issues & Solutions
Issue Solution
Array items not saving Check migration: pages_blocks_items table must exist with parent_id foreign key to pages_blocks
Multiple items open at once (when shouldn't) Verify allowMultipleOpen is false in admin. Check component logic closes other items when opening new one.
Content doesn't expand/collapse Check max-height transition CSS. Ensure overflow: hidden is set on accordion content wrapper.
Chevron icon not rotating Verify icon has transition: transform 0.2s and rotate-180 class applied when open.
Default open item not working Check defaultOpen boolean saved correctly. Ensure initial state in React uses findIndex to locate default item.
Items reorder incorrectly Verify _order column in database. Payload should auto-manage this. Check array sorting in query.
CSS-only version doesn't work Ensure using <details> and <summary> elements. Check open attribute for default open.
Step 8: Register Block in Payload Config

Add the Accordion block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { Accordion } from '@/branches/blocks/shared/Accordion/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ Accordion, // ← Add Accordion block here // Hero, // Features, // FAQ, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { Accordion } from '@/branches/blocks/shared/Accordion/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'accordion': return <Accordion key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'faq': return <FAQ key={index} block={block} /> default: return null } })} </main> ) }

SEO & Accessibility Notes:

  • ✅ Use semantic HTML: <details> / <summary> for better SEO (all content indexable)
  • ✅ Add aria-expanded and aria-controls for screen readers
  • ✅ Keyboard navigation: Space/Enter to toggle, Arrow keys to move between items
  • ✅ Focus visible: Add focus ring (focus:ring-2 ring-teal-glow) to headers
  • ✅ Prefers-reduced-motion: Disable transitions for users who prefer reduced motion
