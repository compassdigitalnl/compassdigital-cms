Step 3: React Component Implementation

Create src/branches/blocks/shared/Video/Component.tsx to render the block:

// src/branches/blocks/shared/Video/Component.tsx import React from 'react' import { RichText } from '@payloadcms/richtext-lexical/react' import { VideoBlock } from '@/types/blocks' import { cn } from '@/lib/utils' import { Media } from '@/payload-types' interface VideoProps { block: VideoBlock } // Helper function to extract YouTube video ID function getYouTubeId(url: string): string | null { const regex = /(?:youtube\.com\/(?:[^\/]+\/.+\/|(?:v|e(?:mbed)?)\/|.*[?&]v=)|youtu\.be\/)([^"&?\/\s]{11})/ const match = url.match(regex) return match ? match[1] : null } // Helper function to extract Vimeo video ID function getVimeoId(url: string): string | null { const regex = /(?:vimeo\.com\/)(\d+)/ const match = url.match(regex) return match ? match[1] : null } export const Video: React.FC<VideoProps> = ({ block }) => { const { title, source, videoUrl, videoFile, thumbnail, caption, aspectRatio } = block // Map aspect ratio to CSS classes const aspectRatioClass = { '16:9': 'aspect-video', // 16:9 '4:3': 'aspect-[4/3]', // 4:3 '1:1': 'aspect-square', // 1:1 }[aspectRatio] // Render embedded YouTube video const renderYouTube = () => { if (!videoUrl) return null const videoId = getYouTubeId(videoUrl) if (!videoId) return null return ( <iframe src={`https://www.youtube.com/embed/${videoId}?rel=0&modestbranding=1`} title={title || 'YouTube video'} allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowFullScreen className="w-full h-full" /> ) } // Render embedded Vimeo video const renderVimeo = () => { if (!videoUrl) return null const videoId = getVimeoId(videoUrl) if (!videoId) return null return ( <iframe src={`https://player.vimeo.com/video/${videoId}?byline=0&portrait=0`} title={title || 'Vimeo video'} allow="autoplay; fullscreen; picture-in-picture" allowFullScreen className="w-full h-full" /> ) } // Render self-hosted video const renderUpload = () => { if (!videoFile) return null const videoMedia = typeof videoFile === 'string' ? null : videoFile as Media const videoSrc = typeof videoFile === 'string' ? videoFile : videoMedia?.url if (!videoSrc) return null const thumbnailMedia = typeof thumbnail === 'string' ? null : thumbnail as Media const posterSrc = typeof thumbnail === 'string' ? thumbnail : thumbnailMedia?.url return ( <video src={videoSrc} poster={posterSrc} controls className="w-full h-full" preload="metadata" > Your browser does not support the video tag. </video> ) } return ( <section className="py-12 md:py-16"> <div className="container"> <div className="max-w-4xl mx-auto"> {title && ( <h2 className="font-display text-2xl md:text-3xl text-navy mb-6 text-center"> {title} </h2> )} <div className={cn( 'relative rounded-xl overflow-hidden shadow-md bg-grey-light', aspectRatioClass )}> {source === 'youtube' && renderYouTube()} {source === 'vimeo' && renderVimeo()} {source === 'upload' && renderUpload()} </div> {caption && ( <div className="mt-4 text-center prose prose-sm max-w-none text-grey-dark"> <RichText data={caption} /> </div> )} </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Video Block" is visible
2. Conditional fields work: Change source → verify videoUrl shows for YouTube/Vimeo, videoFile shows for Upload
3. YouTube embed works: Paste YouTube URL → save → verify iframe embeds correctly on frontend
4. Vimeo embed works: Paste Vimeo URL → save → verify iframe embeds correctly on frontend
5. Self-hosted video works: Upload MP4 file → save → verify <video> tag renders with controls
6. Aspect ratios work: Test all 3 aspect ratios (16:9, 4:3, 1:1) → verify correct CSS applied
7. TypeScript compilation: Run npm run build → no TypeScript errors

Step 7: Common Issues & Solutions
Issue Solution
YouTube/Vimeo video not embedding Check URL format → use full URL (not shortened) → verify regex in getYouTubeId() / getVimeoId() functions
Conditional fields not showing/hiding Verify admin.condition function in config.ts → check siblingData.source value matches expected strings
Self-hosted video not playing Check video format (MP4/WebM recommended) → verify file uploaded to media collection → check browser console for CORS errors
Aspect ratio not applied Verify Tailwind classes (aspect-video, aspect-[4/3], aspect-square) → ensure aspect-ratio plugin enabled in tailwind.config
Thumbnail not showing Check poster attribute on <video> tag → verify thumbnail uploaded and URL accessible → check image format (JPG/PNG)
Validation error on save Check validate functions → ensure required fields filled based on source type → verify videoUrl for YouTube/Vimeo, videoFile for Upload
TypeScript error: Media type Import Media type from @/payload-types → run npx payload generate:types if missing
