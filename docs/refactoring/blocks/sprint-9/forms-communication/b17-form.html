3 React Client Component

Form component uses client-side React (use client directive) for form state management, validation, and submission. Integrates with reCAPTCHA v3 for spam protection.

import type { FormComponentProps, FormSubmissionData } from './types'
import { serializeLexical } from '@/utilities/serializeLexical'
import { cn } from '@/utilities/cn'

export function Form({ block }: FormComponentProps) {
  const {
    title,
    description,
    submitLabel = 'Verzenden',
    successMessage,
    submitTo,
    fields,
    enableRecaptcha = true,
    redirectUrl,
    maxWidth = 'medium',
  } = block

  const [formData, setFormData] = useState({})
  const [errors, setErrors] = useState>({})
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [isSuccess, setIsSuccess] = useState(false)
  const { executeRecaptcha } = useGoogleReCaptcha()

  const maxWidthClasses = {
    small: 'max-w-md',   // 480px
    medium: 'max-w-2xl', // 640px
    large: 'max-w-3xl',  // 800px
  }

  const handleChange = (name: string, value: string | string[] | File) => {
    setFormData((prev) => ({ ...prev, [name]: value }))
    // Clear error for this field
    if (errors[name]) {
      setErrors((prev) => {
        const newErrors = { ...prev }
        delete newErrors[name]
        return newErrors
      })
    }
  }

  const validate = (): boolean => {
    const newErrors: Record = {}

    fields.forEach((field) => {
      const value = formData[field.name]

      // Required field validation
      if (field.required && (!value || (Array.isArray(value) && value.length === 0))) {
        newErrors[field.name] = `${field.label} is verplicht`
      }

      // Email validation
      if (field.type === 'email' && value && typeof value === 'string') {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
        if (!emailRegex.test(value)) {
          newErrors[field.name] = 'Voer een geldig e-mailadres in'
        }
      }

      // Phone validation
      if (field.type === 'tel' && value && typeof value === 'string') {
        const phoneRegex = /^[\d\s\+\-\(\)]+$/
        if (!phoneRegex.test(value)) {
          newErrors[field.name] = 'Voer een geldig telefoonnummer in'
        }
      }
    })

    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()

    if (!validate()) return

    setIsSubmitting(true)

    try {
      let recaptchaToken = ''

      // Get reCAPTCHA token if enabled
      if (enableRecaptcha && executeRecaptcha) {
        recaptchaToken = await executeRecaptcha('form_submit')
      }

      // Submit form data
      const response = await fetch('/api/forms/submit', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          formData,
          submitTo,
          recaptchaToken,
        }),
      })

      const result = await response.json()

      if (result.success) {
        setIsSuccess(true)
        setFormData({})

        // Redirect if URL provided
        if (redirectUrl) {
          setTimeout(() => {
            window.location.href = redirectUrl
          }, 1500)
        }
      } else {
        setErrors(result.errors || { _form: 'Er is iets misgegaan. Probeer het opnieuw.' })
      }
    } catch (error) {
      setErrors({ _form: 'Er is een fout opgetreden. Probeer het later opnieuw.' })
    } finally {
      setIsSubmitting(false)
    }
  }

  if (isSuccess && !redirectUrl) {
    return (
