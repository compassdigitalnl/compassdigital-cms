STEP 2 TypeScript Type Definitions

Define TypeScript interfaces for type-safe development with the ContactForm block.

TypeScript src/types/blocks.ts 
// src/types/blocks.ts
export interface ContactInfo {
  phone?: string
  email?: string
  address?: string
}

export interface ContactFormBlock {
  blockType: 'contactform'
  id?: string
  title: string
  description?: string
  showPhone: boolean
  showSubject: boolean
  submitTo: string
  contactInfo?: ContactInfo
  successMessage: string
}

// Form Submission Types
export interface ContactFormData {
  name: string
  email: string
  phone?: string
  subject?: string
  message: string
}

export interface ContactFormResponse {
  success: boolean
  message: string
  errors?: {
    field: string
    message: string
  }[]
}

// Type Guards
export function isContactFormBlock(block: any): block is ContactFormBlock {
  return block?.blockType === 'contactform'
}

export function hasContactInfo(block: ContactFormBlock): boolean {
  const info = block.contactInfo
  return !!(info?.phone || info?.email || info?.address)
}

// Validation
export function validateContactFormData(data: ContactFormData): string[] {
  const errors: string[] = []

  if (!data.name || data.name.trim().length < 2) {
    errors.push('Naam is verplicht (minimaal 2 tekens)')
  }

  if (!data.email || !isValidEmail(data.email)) {
    errors.push('Geldig email adres is verplicht')
  }

  if (!data.message || data.message.trim().length < 10) {
    errors.push('Bericht is verplicht (minimaal 10 tekens)')
  }

  return errors
}

function isValidEmail(email: string): boolean {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)
}
STEP 3 React Component Implementation

Create the Next.js React component with form handling, validation, and API submission.

TypeScript React src/branches/blocks/forms-communication/ContactForm/Component.tsx 
// src/branches/blocks/forms-communication/ContactForm/Component.tsx
'use client'

import React, { useState } from 'react'
import { MapPin, Phone, Mail, Send, CheckCircle, AlertCircle } from 'lucide-react'
import type { ContactFormBlock, ContactFormData } from '@/types/blocks'

interface ContactFormProps {
  block: ContactFormBlock
}

export const ContactForm: React.FC<ContactFormProps> = ({ block }) => {
  const {
    title,
    description,
    showPhone,
    showSubject,
    submitTo,
    contactInfo,
    successMessage
  } = block

  const [formData, setFormData] = useState<ContactFormData>({
    name: '',
    email: '',
    phone: '',
    subject: '',
    message: ''
  })

  const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle')
  const [errorMessage, setErrorMessage] = useState('')

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>) => {
    setFormData(prev => ({
      ...prev,
      [e.target.name]: e.target.value
    }))
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setStatus('loading')
    setErrorMessage('')

    try {
      const response = await fetch('/api/contact', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          ...formData,
          submitTo
        })
      })

      const data = await response.json()

      if (data.success) {
        setStatus('success')
        setFormData({
          name: '',
          email: '',
          phone: '',
          subject: '',
          message: ''
        })
      } else {
        setStatus('error')
        setErrorMessage(data.message || 'Er is iets misgegaan')
      }
    } catch (error) {
      setStatus('error')
      setErrorMessage('Kan geen verbinding maken met de server')
    }
  }

  const hasContactInfo = contactInfo?.phone || contactInfo?.email || contactInfo?.address

  return (
    <section className="py-16 md:py-20">
      <div className="container mx-auto px-6">
        {/* Header */}
        {(title || description) && (
          <div className="text-center mb-12">
            {title && <h2 className="font-serif text-3xl md:text-4xl text-navy mb-3">{title}</h2>}
            {description && <p className="text-base text-grey-dark max-w-2xl mx-auto">{description}</p>}
          </div>
        )}
        {/* Success Message */}
        {status === 'success' && (
          <div className="max-w-3xl mx-auto mb-8 p-4 bg-green-light border border-green rounded-lg flex items-center gap-3">
            <CheckCircle className="w-5 h-5 text-green flex-shrink-0" />
            <p className="text-sm text-navy">{successMessage}</p>
          </div>
        )}
        {/* Error Message */}
        {status === 'error' && (
          <div className="max-w-3xl mx-auto mb-8 p-4 bg-coral-light border border-coral rounded-lg flex items-center gap-3">
            <AlertCircle className="w-5 h-5 text-coral flex-shrink-0" />
            <p className="text-sm text-navy">{errorMessage}</p>
          </div>
        )}

        {/* Form Grid */}
        <div className="grid lg:grid-cols-3 gap-8 md:gap-12 max-w-5xl mx-auto">
          {/* Contact Form */}
          <div className="lg:col-span-2">
            <form onSubmit={handleSubmit} className="space-y-5">
              {/* Name */}
              <div>
                <label htmlFor="name" className="block text-sm font-semibold text-navy mb-2">
                  Naam <span className="text-coral">*</span>
                </label>
                <input
                  type="text"
                  id="name"
                  name="name"
                  value={formData.name}
                  onChange={handleChange}
                  required
                  className="w-full px-4 py-3 text-sm border-1.5 border-grey rounded-lg focus:outline-none focus:border-teal focus:ring-3 focus:ring-teal-glow transition"
                  placeholder="Je volledige naam"
                />
              </div>

              {/* Email */}
              <div>
                <label htmlFor="email" className="block text-sm font-semibold text-navy mb-2">
                  Email <span className="text-coral">*</span>
                </label>
                <input
                  type="email"
                  id="email"
                  name="email"
                  value={formData.email}
                  onChange={handleChange}
                  required
                  className="w-full px-4 py-3 text-sm border-1.5 border-grey rounded-lg focus:outline-none focus:border-teal focus:ring-3 focus:ring-teal-glow transition"
                  placeholder="naam@voorbeeld.nl"
                />
              </div>

              {/* Phone (optional) */}
              {showPhone && (
                <div>
                  <label htmlFor="phone" className="block text-sm font-semibold text-navy mb-2">
                    Telefoon
                  </label>
                  <input
                    type="tel"
                    id="phone"
                    name="phone"
                    value={formData.phone}
                    onChange={handleChange}
                    className="w-full px-4 py-3 text-sm border-1.5 border-grey rounded-lg focus:outline-none focus:border-teal focus:ring-3 focus:ring-teal-glow transition"
                    placeholder="06 12 34 56 78"
                  />
                </div>
              )}

              {/* Subject (optional) */}
              {showSubject && (
                <div>
                  <label htmlFor="subject" className="block text-sm font-semibold text-navy mb-2">
                    Onderwerp
                  </label>
                  <input
                    type="text"
                    id="subject"
                    name="subject"
                    value={formData.subject}
                    onChange={handleChange}
                    className="w-full px-4 py-3 text-sm border-1.5 border-grey rounded-lg focus:outline-none focus:border-teal focus:ring-3 focus:ring-teal-glow transition"
                    placeholder="Waar kunnen we je mee helpen?"
                  />
                </div>
              )}

              {/* Message */}
              <div>
                <label htmlFor="message" className="block text-sm font-semibold text-navy mb-2">
                  Bericht <span className="text-coral">*</span>
                </label>
                <textarea
                  id="message"
                  name="message"
                  value={formData.message}
                  onChange={handleChange}
                  required
                  rows={5}
                  className="w-full px-4 py-3 text-sm border-1.5 border-grey rounded-lg focus:outline-none focus:border-teal focus:ring-3 focus:ring-teal-glow transition resize-vertical"
                  placeholder="Typ hier je bericht..."
                />
              </div>

              {/* Submit Button */}
              <button
                type="submit"
                disabled={status === 'loading'}
                className="inline-flex items-center gap-2 px-6 py-3 bg-teal text-white text-sm font-bold rounded-lg hover:bg-teal-dark transition disabled:opacity-50 disabled:cursor-not-allowed"
              >
                <span>{status === 'loading' ? 'Versturen...' : 'Verstuur bericht'}</span>
                <Send className="w-4 h-4" />
              </button>
            </form>
          </div>

          {/* Contact Info Sidebar */}
          {hasContactInfo && (
            <div className="bg-grey-light rounded-xl p-8">
              <h3 className="font-serif text-xl text-navy mb-6">Contactgegevens</h3>
              <div className="space-y-5">
                {/* Phone */}
                {contactInfo.phone && (
                  <div className="flex gap-3">
                    <div className="w-9 h-9 bg-white rounded-lg flex items-center justify-center flex-shrink-0">
                      <Phone className="w-4.5 h-4.5 text-teal" />
                    </div>
                    <div>
                      <div className="text-xs font-bold uppercase tracking-wide text-grey-dark mb-1">Telefoon</div>
                      <a href={`tel:${contactInfo.phone.replace(/[^0-9+]/g, '')}`} className="text-sm text-teal hover:text-teal-dark transition">
                        {contactInfo.phone}
                      </a>
                    </div>
                  </div>
                )}
                {/* Email */}
                {contactInfo.email && (
                  <div className="flex gap-3">
                    <div className="w-9 h-9 bg-white rounded-lg flex items-center justify-center flex-shrink-0">
                      <Mail className="w-4.5 h-4.5 text-teal" />
                    </div>
                    <div>
                      <div className="text-xs font-bold uppercase tracking-wide text-grey-dark mb-1">Email</div>
                      <a href={`mailto:${contactInfo.email}`} className="text-sm text-teal hover:text-teal-dark transition">
                        {contactInfo.email}
                      </a>
                    </div>
                  </div>
                )}
                {/* Address */}
                {contactInfo.address && (
                  <div className="flex gap-3">
                    <div className="w-9 h-9 bg-white rounded-lg flex items-center justify-center flex-shrink-0">
                      <MapPin className="w-4.5 h-4.5 text-teal" />
                    </div>
                    <div>
                      <div className="text-xs font-bold uppercase tracking-wide text-grey-dark mb-1">Adres</div>
                      <p className="text-sm text-navy whitespace-pre-line">{contactInfo.address}</p>
                    </div>
                  </div>
                )}
              </div>
            </div>
          )}
        </div>
      </div>
    </section>
  )
}

STEP 4 API Endpoint Implementation

Create the Next.js API route for handling contact form submissions.

TypeScript src/app/api/contact/route.ts 
// src/app/api/contact/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { validateContactFormData } from '@/types/blocks'
import { sendEmail } from '@/lib/email' // Using Resend or Nodemailer

export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    const { name, email, phone, subject, message, submitTo } = body

    // Validate input
    const errors = validateContactFormData({ name, email, phone, subject, message })
    if (errors.length > 0) {
      return NextResponse.json({
        success: false,
        message: errors.join(', '),
        errors: errors.map(err => ({ field: 'general', message: err }))
      }, { status: 400 })
    }

    // Send email
    const emailHtml = `
      
Nieuw contactformulier bericht

      
Naam: ${name}

      
Email: ${email}

      ${phone ? `
Telefoon: ${phone}
` : ''}
      ${subject ? `
Onderwerp: ${subject}
` : ''}
      
Bericht:

      
${message.replace(/\n/g, '
')}

    `

    await sendEmail({
      to: submitTo,
      replyTo: email,
      subject: subject || `Contact formulier van ${name}`,
      html: emailHtml
    })

    // Optional: Save to database
    // await payload.create({
    //   collection: 'form-submissions',
    //   data: { name, email, phone, subject, message }
    // })

    return NextResponse.json({
      success: true,
      message: 'Bedankt voor je bericht!'
    })

  } catch (error) {
    console.error('Contact form error:', error)
    return NextResponse.json({
      success: false,
      message: 'Er is een fout opgetreden bij het versturen van je bericht.'
    }, { status: 500 })
  }
}

STEP 7 Troubleshooting Guide
⚠️ Common Issues & Solutions
❌ Form submits but email not received
Solution: Check email service configuration (Resend/Nodemailer). Verify submitTo email is valid. Check spam folder. Enable email service logging to debug. Verify API key is set correctly in environment variables.
❌ Success message not clearing form
Solution: Ensure setFormData() is called with empty values after successful submission. Check that form inputs have value={formData.fieldName} bound correctly.
❌ Optional fields (phone/subject) not hiding
Solution: Verify showPhone and showSubject boolean props are being passed correctly. Check conditional rendering: {showPhone && (...)}. Ensure Payload admin checkboxes are saved correctly.
❌ Form validation not working
Solution: Ensure required attribute is set on HTML inputs. Add type="email" for email field. Implement server-side validation in API route using validateContactFormData() helper function.
❌ CORS errors when submitting form
Solution: Ensure API route is in same Next.js app (not external). If using external API, add CORS headers to API response. For Next.js internal API routes, CORS is handled automatically.
❌ Sidebar not stacking on mobile
Solution: Verify grid class: grid lg:grid-cols-3 defaults to 1 column on mobile. Form uses lg:col-span-2 on desktop. Test with browser dev tools mobile viewport.
❌ Focus ring not showing on inputs
Solution: Ensure Tailwind focus classes are applied: focus:border-teal focus:ring-3 focus:ring-teal-glow. Check outline: none is set to remove default browser outline. Verify teal-glow color is defined in Tailwind config.
❌ Duplicate form submissions
Solution: Disable submit button during submission: disabled={status === 'loading'}. Add debouncing to prevent double-clicks. Track submission state and prevent re-submission until complete.
