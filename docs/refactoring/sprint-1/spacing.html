Step 3 CSS Variable Generation in Next.js
src/app/layout.tsx

Fetch spacing data from Payload and inject CSS custom properties into the DOM

// src/app/layout.tsx import { getPayloadHMR } from '@payloadcms/next/utilities' import config from '@payload-config' export default async function RootLayout({ children }) { const payload = await getPayloadHMR({ config }) // Fetch theme global const theme = await payload.findGlobal({ slug: 'theme', }) // Generate spacing CSS variables const spacing = theme?.spacing || {} const spacingVars = ` --sp-1: ${spacing.sp1 || 4}px; --sp-2: ${spacing.sp2 || 8}px; --sp-3: ${spacing.sp3 || 12}px; --sp-4: ${spacing.sp4 || 16}px; --sp-6: ${spacing.sp6 || 24}px; --sp-8: ${spacing.sp8 || 32}px; --sp-12: ${spacing.sp12 || 48}px; --sp-16: ${spacing.sp16 || 64}px; --sp-20: ${spacing.sp20 || 80}px; ` return ( <html lang="nl"> <head> <style dangerouslySetInnerHTML={{ __html: `:root { ${spacingVars} }` }} /> </head> <body>{children}</body> </html> ) }

Step 8 Advanced Usage & ESLint Enforcement
ESLint Rule: Enforce Spacing Tokens

Problem: Developers accidentally using arbitrary spacing values (5px, 15px, 22px)
Solution: Custom ESLint rule to enforce spacing token usage

// .eslintrc.js (add to rules section) module.exports = { rules: { 'no-arbitrary-spacing': ['warn', { allowedValues: [4, 8, 12, 16, 24, 32, 48, 64, 80], message: 'Use spacing token instead: var(--sp-*)', }], }, } // Custom rule implementation (eslint-plugin-local/rules/no-arbitrary-spacing.js) module.exports = { meta: { type: 'suggestion', docs: { description: 'Enforce 4px grid spacing tokens', }, }, create(context) { const allowedValues = [4, 8, 12, 16, 24, 32, 48, 64, 80] return { TemplateLiteral(node) { const cssText = node.quasis.map(q => q.value.raw).join('') const spacingProps = ['margin', 'padding', 'gap', 'top', 'right', 'bottom', 'left'] spacingProps.forEach(prop => { const regex = new RegExp(`${prop}:\\s*(\\d+)px`, 'g') let match while ((match = regex.exec(cssText)) !== null) { const value = parseInt(match[1]) if (!allowedValues.includes(value)) { context.report({ node, message: `Arbitrary spacing "${value}px" not allowed. Use spacing token: var(--sp-*)`, }) } } }) }, } }, }
Vertical-Specific Spacing Overrides

While spacing tokens are LOCKED in Payload admin, you can create vertical-specific overrides in CSS for special cases

/* src/styles/verticals/beauty.css */ /* Beauty vertical might want tighter spacing for compact layouts */ [data-vertical="beauty"] { /* Override default card padding */ --card-padding: var(--sp-4); /* 16px instead of default 24px */ } /* src/styles/verticals/horeca.css */ /* Horeca vertical might want more spacious sections */ [data-vertical="horeca"] { --section-spacing: var(--sp-16); /* 64px instead of default 48px */ } /* Usage in components */ .card { padding: var(--card-padding, var(--sp-6)); /* Fallback to 24px */ } .section { margin-bottom: var(--section-spacing, var(--sp-12)); /* Fallback to 48px */ }
Best Practice: Only override spacing for vertical-specific layouts. The core 4px grid tokens should remain consistent across all verticals to maintain design system integrity.

Programmatic Spacing (React Components)

For dynamic spacing in React components, create a spacing utility

// src/utils/spacing.ts type SpacingToken = 1 | 2 | 3 | 4 | 6 | 8 | 12 | 16 | 20 export const getSpacing = (token: SpacingToken): string => { return `var(--sp-${token})` } export const getSpacingValue = (token: SpacingToken): number => { const values = { 1: 4, 2: 8, 3: 12, 4: 16, 6: 24, 8: 32, 12: 48, 16: 64, 20: 80 } return values[token] } // Usage in React component import { getSpacing } from '@/utils/spacing' export const Card = ({ spacing = 6 }) => { return ( <div style={{ padding: getSpacing(spacing) }}> Card content </div> ) } // TypeScript ensures only valid tokens can be passed <Card spacing={6} /> /* ✅ Valid */ <Card spacing={5} /> /* ❌ TypeScript error: 5 is not a valid SpacingToken */
