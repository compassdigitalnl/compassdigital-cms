Step 2: TypeScript Interfaces

Definieer type-safe interfaces voor de Newsletter block en API responses. 

// src/types/blocks.ts
export interface NewsletterBlock {
  blockType: 'newsletter'
  id?: string
  title: string
  description?: string
  placeholder: string
  buttonLabel: string
  backgroundColor: 'white' | 'grey' | 'teal' | 'navy'
  privacyText: string
  successMessage: string
  listId?: string
  tags?: string
}

export interface NewsletterSubscription {
  email: string
  listId?: string
  tags?: string[]
  subscribedAt: Date
  source: string // 'newsletter-block', 'footer', etc.
}

export interface NewsletterResponse {
  success: boolean
  message: string
  error?: string
}

// Validation helper
export function isValidEmail(email: string): boolean {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
  return emailRegex.test(email)
}

export function validateNewsletterEmail(email: string): string | null {
  if (!email || email.trim().length === 0) {
    return 'Email adres is verplicht'
  }
  if (!isValidEmail(email)) {
    return 'Voer een geldig email adres in'
  }
  if (email.length > 255) {
    return 'Email adres is te lang'
  }
  return null
}

Step 3: React Component

Newsletter formulier met client-side validatie, loading states, en success/error feedback. 

// src/branches/blocks/forms-communication/Newsletter/Component.tsx
'use client'

import React, { useState } from 'react'
import { Send, CheckCircle, AlertCircle } from 'lucide-react'
import type { NewsletterBlock } from '@/types/blocks'

interface NewsletterProps {
  block: NewsletterBlock
}

export const Newsletter: React.FC<NewsletterProps> = ({ block }) => {
  const {
    title,
    description,
    placeholder,
    buttonLabel,
    backgroundColor,
    privacyText,
    successMessage,
    listId,
    tags,
  } = block

  const [email, setEmail] = useState('')
  const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle')
  const [errorMessage, setErrorMessage] = useState('')

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setStatus('loading')
    setErrorMessage('')

    try {
      const response = await fetch('/api/newsletter', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          email: email.trim(),
          listId,
          tags: tags?.split(',').map(t => t.trim()),
          source: 'newsletter-block',
        }),
      })

      const data = await response.json()

      if (data.success) {
        setStatus('success')
        setEmail('')

        // Auto-hide success message after 5 seconds
        setTimeout(() => {
          setStatus('idle')
        }, 5000)
      } else {
        setStatus('error')
        setErrorMessage(data.message || 'Er is iets misgegaan')
      }
    } catch (error) {
      console.error('Newsletter subscription error:', error)
      setStatus('error')
      setErrorMessage('Kan geen verbinding maken met de server')
    }
  }

  // Background class mapping
  const bgClass = {
    white: 'bg-white border border-grey',
    grey: 'bg-grey-light',
    teal: 'bg-gradient-to-br from-teal-dark to-teal text-white',
    navy: 'bg-gradient-to-br from-navy to-navy-light text-white',
  }[backgroundColor]

  // Text color classes
  const isDark = backgroundColor === 'teal' || backgroundColor === 'navy'
  const titleClass = isDark ? 'text-white' : 'text-navy'
  const descClass = isDark ? 'text-white/80' : 'text-grey-dark'
  const privacyClass = isDark ? 'text-white/60' : 'text-grey-mid'
  const buttonClass = backgroundColor === 'navy'
    ? 'bg-white text-navy hover:bg-grey-light'
    : 'bg-teal text-white hover:bg-teal-dark'

  return (
    <section className={`py-16 md:py-20 px-8 text-center rounded-2xl ${bgClass}`}>
      <div className="max-w-2xl mx-auto">
        {/* Success Message */}
        {status === 'success' && (
          <div className="mb-6 flex items-center justify-center gap-3 bg-green-light border border-green text-green-dark rounded-xl p-4">
            <CheckCircle className="w-5 h-5 flex-shrink-0" />
            <p className="text-sm font-medium">{successMessage}</p>
          </div>
        )}
        {/* Error Message */}
        {status === 'error' && (
          <div className="mb-6 flex items-center justify-center gap-3 bg-coral-light border border-coral text-coral-dark rounded-xl p-4">
            <AlertCircle className="w-5 h-5 flex-shrink-0" />
            <p className="text-sm font-medium">{errorMessage}</p>
          </div>
        )}
        {/* Header */}
        <h2 className={`text-3xl md:text-4xl font-serif mb-4 ${titleClass}`}>{title}</h2>
        {description && <p className={`text-base mb-8 ${descClass}`}>{description}</p>}
        {/* Form */}
        <form onSubmit={handleSubmit} className="flex flex-col sm:flex-row gap-2 mb-4">
          <input
            type="email"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            placeholder={placeholder}
            required
            disabled={status === 'loading' || status === 'success'}
            className="flex-1 px-4 py-3 text-sm border-2 border-grey rounded-lg bg-white text-black placeholder-grey-mid focus:border-teal focus:ring-3 focus:ring-teal-glow focus:outline-none transition-all disabled:opacity-50"
          />
          <button
            type="submit"
            disabled={status === 'loading' || status === 'success'}
            className={`px-6 py-3 text-sm font-semibold rounded-lg transition-all whitespace-nowrap disabled:opacity-50 disabled:cursor-not-allowed hover:transform hover:-translate-y-0.5 hover:shadow-md ${buttonClass}`}
          >
            {status === 'loading' ? (
              <span className="flex items-center gap-2">
                <svg className="animate-spin h-4 w-4" viewBox="0 0 24 24">
                  <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
                  <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z" />
                </svg>
                Bezig...
              </span>
            ) : (
              <span className="flex items-center gap-2">
                <Send className="w-4 h-4" />
                {buttonLabel}
              </span>
            )}
          </button>
        </form>
        {/* Privacy Text */}
        <p className={`text-xs ${privacyClass}`}>{privacyText}</p>
      </div>
    </section>
  )
}

export default Newsletter

Step 4: API Endpoint

Next.js API route voor newsletter inschrijvingen met email service integratie (Resend/Mailchimp). 

// src/app/api/newsletter/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { validateNewsletterEmail } from '@/types/blocks'

// Example using Resend (install: npm install resend)
import { Resend } from 'resend'
const resend = new Resend(process.env.RESEND_API_KEY)

export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    const { email, listId, tags, source } = body

    // Validate email
    const validationError = validateNewsletterEmail(email)
    if (validationError) {
      return NextResponse.json(
        { success: false, message: validationError },
        { status: 400 }
      )
    }

    // Option 1: Using Resend Audiences
    if (listId) {
      await resend.contacts.create({
        email,
        audienceId: listId,
        // Optional: add custom fields
        firstName: undefined, // Can be extended to capture name
        unsubscribed: false,
      })
    }

    // Option 2: Store in database (recommended for full control)
    // const { data: subscription } = await payload.create({
    //   collection: 'newsletter-subscriptions',
    //   data: {
    //     email,
    //     listId,
    //     tags: tags || [],
    //     source,
    //     subscribedAt: new Date().toISOString(),
    //     ipAddress: request.headers.get('x-forwarded-for') || 'unknown',
    //     userAgent: request.headers.get('user-agent') || 'unknown',
    //   },
    // })

    // Option 3: Send welcome email
    await resend.emails.send({
      from: 'Newsletter <newsletter@yourdomain.com>',
      to: email,
      subject: 'Welkom bij onze nieuwsbrief! 🎉',
      html: `
        <div style="font-family: sans-serif; max-width: 600px; margin: 0 auto;">
          <h1 style="color: #00897B;">Bedankt voor je inschrijving!</h1>
          <p>Je ontvangt vanaf nu regelmatig updates, tips en exclusieve aanbiedingen.</p>
          <p style="color: #64748B; font-size: 12px; margin-top: 32px;">
            Wil je geen emails meer ontvangen?
            <a href="https://yourdomain.com/unsubscribe?email=${encodeURIComponent(email)}" style="color: #00897B;">
              Uitschrijven
            </a>
          </p>
        </div>
      `,
    })

    return NextResponse.json({
      success: true,
      message: 'Je bent succesvol ingeschreven!',
    })
  } catch (error: any) {
    console.error('Newsletter subscription error:', error)

    // Handle duplicate email error
    if (error.message?.includes('already exists') || error.message?.includes('duplicate')) {
      return NextResponse.json(
        { success: false, message: 'Dit email adres is al ingeschreven' },
        { status: 400 }
      )
    }

    return NextResponse.json(
      { success: false, message: 'Er is een fout opgetreden. Probeer het later opnieuw.' },
      { status: 500 }
    )
  }
}

CREATE TABLE pages_blocks_newsletter (

-- Optional: Newsletter subscriptions tracking table
CREATE TABLE newsletter_subscriptions (

  list_id VARCHAR,
  tags TEXT[], -- PostgreSQL array for tags
  source VARCHAR DEFAULT 'newsletter-block',
  subscribed_at TIMESTAMP DEFAULT NOW(),
  unsubscribed_at TIMESTAMP,
  ip_address VARCHAR,
  user_agent TEXT,
  confirmed BOOLEAN DEFAULT false,
  confirmation_token VARCHAR,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_newsletter_email ON newsletter_subscriptions(email);
CREATE INDEX idx_newsletter_subscribed ON newsletter_subscriptions(subscribed_at);
CREATE INDEX idx_newsletter_confirmed ON newsletter_subscriptions(confirmed);

-- Trigger for updated_at
CREATE OR REPLACE FUNCTION update_newsletter_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER newsletter_updated_at
  BEFORE UPDATE ON newsletter_subscriptions
  FOR EACH ROW
  EXECUTE FUNCTION update_newsletter_updated_at();

Step 7: Troubleshooting Guide

Oplossingen voor veelvoorkomende problemen:

1. Form submit doet niks

Oorzaak: Client component niet correct gemarkeerd

Oplossing: Controleer dat 'use client' bovenaan Component.tsx staat

2. Email wordt niet verzonden

Oorzaak: RESEND_API_KEY niet geconfigureerd

Oplossing: Check .env file en Resend dashboard voor API key

3. Success message blijft hangen

Oorzaak: Auto-hide timeout werkt niet

Oplossing: Verifieer setTimeout() in handleSubmit functie

4. Duplicate error niet friendly

Oorzaak: Email service error niet correct afgehandeld

Oplossing: Check error.message parsing in catch block

5. Background kleuren werken niet

Oorzaak: Tailwind classes niet geconfigureerd

Oplossing: Voeg custom colors toe aan tailwind.config.ts

6. Form niet responsive

Oorzaak: sm: breakpoint classes ontbreken

Oplossing: Gebruik flex-col sm:flex-row voor form container

7. Loading spinner niet zichtbaar

Oorzaak: SVG viewBox verkeerd of className ontbreekt

Oplossing: Check animate-spin class en SVG viewBox="0 0 24 24"

8. Subscriptions niet in database

Oorzaak: Payload.create() call gecommentarieerd

Oplossing: Uncommment database save in route.ts (Option 2)
