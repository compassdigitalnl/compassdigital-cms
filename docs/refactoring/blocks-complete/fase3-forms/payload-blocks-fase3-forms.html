3 React Component Usage

Voorbeeld van ContactForm component met form handling en validation:

// src/components/blocks/ContactForm.tsx
'use client'
import React, { useState } from 'react'
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'

const contactSchema = z.object({
  firstName: z.string().min(2, 'Voornaam is verplicht'),
  lastName: z.string().min(2, 'Achternaam is verplicht'),
  email: z.string().email('Ongeldig e-mailadres'),
  phone: z.string().optional(),
  company: z.string().optional(),
  subject: z.string().min(1, 'Kies een onderwerp'),
  message: z.string().min(10, 'Bericht moet minimaal 10 karakters zijn')
})

type ContactFormData = z.infer<typeof contactSchema>

export const ContactForm: React.FC<ContactFormBlock> = ({
  variant = 'standard',
  title,
  description,
  channels,
  showChannels,
  emailRecipient,
  recaptchaEnabled
}) => {
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [submitSuccess, setSubmitSuccess] = useState(false)

  const {
    register,
    handleSubmit,
    formState: { errors },
    reset
  } = useForm<ContactFormData>({
    resolver: zodResolver(contactSchema)
  })

  const onSubmit = async (data: ContactFormData) => {
    setIsSubmitting(true)

    try {
      // reCAPTCHA verification
      if (recaptchaEnabled) {
        const token = await executeRecaptcha()
        data.recaptchaToken = token
      }

      // Submit to API endpoint
      const response = await fetch('/api/contact', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ ...data, recipient: emailRecipient })
      })

      if (!response.ok) throw new Error('Submission failed')

      setSubmitSuccess(true)
      reset()
    } catch (error) {
      console.error('Contact form error:', error)
      alert('Er ging iets mis. Probeer opnieuw.')
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <div className={`contact-form variant-${variant}`}>
      {showChannels && channels && (
        <div className="channels">
          {channels.map((channel, i) => (
            <a key={i} href={channel.link} className="channel">
              {/* Channel UI */}
            </a>
          ))}
        </div>
      )}

      <div className="card">
        <h3 className="card-title">{title}</h3>
        <p className="card-desc">{description}</p>

        {submitSuccess ? (
          <div className="form-success">
            ✓ Bericht verzonden! We nemen zo snel mogelijk contact op.
          </div>
        ) : (
          <form onSubmit={handleSubmit(onSubmit)}>
            {/* Form fields with validation errors */}
            <button type="submit" disabled={isSubmitting}>
              {isSubmitting ? 'Versturen...' : 'Verstuur bericht'}
            </button>
          </form>
        )}
      </div>
    </div>
  )
}

API Route voor form submissions:

// src/app/api/contact/route.ts
import { NextResponse } from 'next/server'
import { Resend } from 'resend'

const resend = new Resend(process.env.RESEND_API_KEY)

export async function POST(request: Request) {
  try {
    const data = await request.json()

    // reCAPTCHA verification
    if (data.recaptchaToken) {
      const isValid = await verifyRecaptcha(data.recaptchaToken)
      if (!isValid) {
        return NextResponse.json(
          { error: 'reCAPTCHA verification failed' },
          { status: 400 }
        )
      }
    }

    // Send email via Resend
    await resend.emails.send({
      from: 'contact@yourdomain.com',
      to: data.recipient,
      subject: `Contact formulier: ${data.subject}`,
      html: generateEmailTemplate(data)
    })

    // Store in database (optional)
    await payload.create({
      collection: 'contact-submissions',
      data: {
        ...data,
        submittedAt: new Date()
      }
    })

    return NextResponse.json({ success: true })
  } catch (error) {
    return NextResponse.json(
      { error: 'Submission failed' },
      { status: 500 }
    )
  }
}

5 Testing Checklist

Alle form blocks moeten deze tests doorstaan:

  • Form Validation: Test required fields, email format, phone format, min/max lengths
  • Spam Protection: Verify reCAPTCHA works (test keys vs production)
  • Email Delivery: Confirm emails arrive at correct recipient with proper formatting
  • Database Storage: Check submissions are stored in Payload CMS
  • Error Handling: Test API failures, network errors, validation errors
  • Success States: Verify success message displays, form resets after submit
  • Mobile Responsive: Test all forms on mobile (touch targets, keyboard behavior)
  • Accessibility: Screen reader support, keyboard navigation, ARIA labels
  • GDPR Compliance: Privacy checkboxes, terms links, data retention policies
  • Performance: Check form submit speed, loading states, debouncing

export const ContactSubmissions: CollectionConfig = {
  slug: 'contact-submissions',

    useAsTitle: 'email',

7 Troubleshooting

Veelvoorkomende problemen en oplossingen:

Probleem Oorzaak Oplossing
Form submit fails zonder error reCAPTCHA key mismatch Check NEXT_PUBLIC_RECAPTCHA_SITE_KEY en RECAPTCHA_SECRET_KEY
Emails komen niet aan Resend API key onjuist Verify RESEND_API_KEY in .env, check Resend dashboard
Validation errors niet zichtbaar Zod schema mismatch Ensure all form fields have matching Zod schema definitions
Calendar dates niet interactief JavaScript niet geladen Verify Lucide icons loaded, check browser console
File upload fails Max size exceeded Increase Next.js body size limit in next.config.js
