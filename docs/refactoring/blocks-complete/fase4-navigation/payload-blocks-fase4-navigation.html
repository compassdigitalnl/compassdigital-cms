import React, { useState } from 'react'
import { Info, CheckCircle, AlertTriangle, AlertCircle, X } from 'lucide-react'

const iconComponents = {
  'info': Info,
  'check-circle': CheckCircle,
  'alert-triangle': AlertTriangle,
  'alert-circle': AlertCircle,
}

export const InfoBox: React.FC<{ block: InfoBoxBlock }> = ({ block }) => {
  const [dismissed, setDismissed] = useState(false)
  const { variant, title, description, icon, dismissible } = block

  if (dismissed) return null

  // Auto icon selection
  const iconName = icon === 'auto' ? iconMap[variant] : icon
  const IconComponent = iconComponents[iconName]

  return (
    <div className={`info-box ${variant}`}>
      <IconComponent className="icon" />

      <div className="info-box-content">
        <div className="info-box-title">{title}</div>
        <div className="info-box-description">{description}</div>
      </div>

      {dismissible && (
        <button
          className="info-box-close"
          onClick={() => setDismissed(true)}
          aria-label="Dismiss"
        >
          <X size={16} />
        </button>
      )}
    </div>
  )
}

import React, { useState, useEffect } from 'react'
import { Megaphone, Gift, AlertTriangle, X } from 'lucide-react'
import { isBefore } from 'date-fns'

const iconMap = {
  'megaphone': Megaphone,
  'gift': Gift,
  'alert-triangle': AlertTriangle,
}

export const Banner: React.FC<{ block: BannerBlock }> = ({ block }) => {
  const [dismissed, setDismissed] = useState(false)
  const { variant, title, description, icon, ctaText, ctaLink, dismissible, expiryDate } = block

  // Check expiry
  useEffect(() => {
    if (expiryDate && isBefore(new Date(expiryDate), new Date())) {
      setDismissed(true)
    }
  }, [expiryDate])

  // Check localStorage for dismissed state
  useEffect(() => {
    const isDismissed = localStorage.getItem(`banner-${title}`)
    if (isDismissed) setDismissed(true)
  }, [title])

  const handleDismiss = () => {
    setDismissed(true)
    localStorage.setItem(`banner-${title}`, 'true')
  }

  if (dismissed) return null

  const IconComponent = iconMap[icon]

  return (
    <div className={`banner ${variant}`}>
      {IconComponent && <IconComponent className="icon" />}

      <div className="banner-content">
        <div className="banner-title">{title}</div>
        {description && <div className="banner-description">{description}</div>}
      </div>

      {variant === 'promo' && ctaText && (
        <div className="banner-actions">
          <a href={ctaLink} className="banner-link">{ctaText} →</a>
        </div>
      )}

      {dismissible && (
        <button className="banner-close" onClick={handleDismiss} aria-label="Close banner">
          <X className="icon" />
        </button>
      )}
    </div>
  )
}

import React from 'react'
import { ChevronRight } from 'lucide-react'
import { usePathname } from 'next/navigation'

export const Breadcrumbs: React.FC<BreadcrumbsProps> = ({
  items,
  homeLabel = 'Home',
  separator = 'chevron'
}) => {
  const pathname = usePathname()

  // Auto-generate breadcrumbs from URL if not provided
  const breadcrumbs = items || generateBreadcrumbs(pathname, homeLabel)

  const separatorMap = {
    chevron: <ChevronRight size={14} />,
    slash: <span>/</span>,
    dash: <span>-</span>
  }

  return (
    <nav className="breadcrumbs" aria-label="Breadcrumb">
      {breadcrumbs.map((crumb, index) => (
        <React.Fragment key={index}>
          <div className="breadcrumb-item">
            {crumb.current ? (
              <span className="breadcrumb-current" aria-current="page">
                {crumb.label}
              </span>
            ) : (
              <a href={crumb.href} className="breadcrumb-link">
                {crumb.label}
              </a>
            )}
          </div>

          {index < breadcrumbs.length - 1 && (
            <div className="breadcrumb-separator" aria-hidden="true">
              {separatorMap[separator]}
            </div>
          )}
        </React.Fragment>
      ))}
    </nav>
  )
}

// Auto-generate from URL
function generateBreadcrumbs(pathname: string, homeLabel: string): Breadcrumb[] {
  const paths = pathname.split('/').filter(Boolean)

  const breadcrumbs: Breadcrumb[] = [
    { label: homeLabel, href: '/' }
  ]

  let currentPath = ''
  paths.forEach((path, index) => {
    currentPath += `/${path}`
    const isLast = index === paths.length - 1

    breadcrumbs.push({
      label: formatLabel(path),
      href: isLast ? undefined : currentPath,
      current: isLast
    })
  })

  return breadcrumbs
}

function formatLabel(slug: string): string {
  return slug
    .split('-')
    .map(word => word.charAt(0).toUpperCase() + word.slice(1))
    .join(' ')
}

STEP 3: React Component

import React from 'react'
import { ChevronLeft, ChevronRight } from 'lucide-react'

export const Pagination: React.FC<PaginationProps> = ({
  currentPage,
  totalPages,
  totalItems,
  itemsPerPage,
  onPageChange,
  showInfo = true,
  maxButtons = 7
}) => {
  // Generate page numbers with ellipsis
  const getPageNumbers = () => {
    const pages: (number | 'ellipsis')[] = []
    const halfMax = Math.floor(maxButtons / 2)

    if (totalPages <= maxButtons) {
      // Show all pages
      for (let i = 1; i <= totalPages; i++) {
        pages.push(i)
      }
    } else {
      // Always show first page
      pages.push(1)

      let start = Math.max(2, currentPage - halfMax)
      let end = Math.min(totalPages - 1, currentPage + halfMax)

      // Add ellipsis after first page if needed
      if (start > 2) {
        pages.push('ellipsis')
      }

      // Add middle pages
      for (let i = start; i <= end; i++) {
        pages.push(i)
      }

      // Add ellipsis before last page if needed
      if (end < totalPages - 1) {
        pages.push('ellipsis')
      }

      // Always show last page
      pages.push(totalPages)
    }

    return pages
  }

  const startItem = (currentPage - 1) * itemsPerPage + 1
  const endItem = Math.min(currentPage * itemsPerPage, totalItems)

  return (
    <>
      <nav className="pagination" aria-label="Pagination">
        <button
          className="pagination-button"
          onClick={() => onPageChange(currentPage - 1)}
          disabled={currentPage === 1}
          aria-label="Previous page"
        >
          <ChevronLeft className="icon" />
        </button>

        {getPageNumbers().map((page, index) => (
          page === 'ellipsis' ? (
            <span key={`ellipsis-${index}`} className="pagination-ellipsis">
              ...
            </span>
          ) : (
            <button
              key={page}
              className={`pagination-button ${page === currentPage ? 'active' : ''}`}
              onClick={() => onPageChange(page)}
              aria-label={`Page ${page}`}
              aria-current={page === currentPage ? 'page' : undefined}
            >
              {page}
            </button>
          )
        ))}

        <button
          className="pagination-button"
          onClick={() => onPageChange(currentPage + 1)}
          disabled={currentPage === totalPages}
          aria-label="Next page"
        >
          <ChevronRight className="icon" />
        </button>
      </nav>

      {showInfo && (
        <div className="pagination-info">
          Toont {startItem}-{endItem} van {totalItems} {totalItems === 1 ? 'item' : 'items'}
        </div>
      )}
    </>
  )
}
