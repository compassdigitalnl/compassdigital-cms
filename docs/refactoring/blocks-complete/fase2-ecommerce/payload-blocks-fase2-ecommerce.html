interface ProductGridBlock {
  blockType: 'productGrid'
  products: Product[]
  columns: '3' | '4' | '5'
  showRatings: boolean
  showStockStatus: boolean
  showQuickAdd: boolean
}

interface Product {
  id: string
  name: string
  brand: string
  price: number
  originalPrice?: number
  image: Media
  rating?: number
  reviewCount?: number
  stockStatus: 'ok' | 'low' | 'out'
  badges?: Array<'new' | 'sale' | 'pro'>
}
3

React Component

export function ProductGrid({ block }: { block: ProductGridBlock }) {
  return (
    <div className={`product-grid cols-${block.columns}`}>
      {block.products.map((product) => (
        <ProductCard
          key={product.id}
          product={product}
          showRating={block.showRatings}
          showStock={block.showStockStatus}
          showQuickAdd={block.showQuickAdd}
        />
      ))}
    </div>
  )
}

export function CategoryGrid({ block }: { block: CategoryGridBlock }) {

        <Link
          key={category.id}
          href={`/shop/${category.slug}`}

          }`}
        >
          {block.variant === 'icon' ? (
            <div className="category-icon">
              <Icon name={category.icon} />
            </div>
          ) : (
            <div className="category-image">
              <Image src={category.image.url} alt={category.name} />
            </div>
          )}

            <div className="category-name">{category.name}</div>
            {block.showProductCount && (
              <div className="category-count">
                {category.productCount} producten
              </div>
            )}
          </div>
        </Link>

    </div>
  )
}

export function SubscriptionPricing({ block }: Props) {

export function StaffelPricing({ block }: Props) {
  const [quantity, setQuantity] = useState(1)

  const activeTier = block.tiers.find(tier =>
    quantity >= tier.minQuantity &&
    (!tier.maxQuantity || quantity <= tier.maxQuantity)
  )

  const total = activeTier ? activeTier.price * quantity : 0

  return (
    <div>
      <table className="staffel-table">
        <thead>
          <tr>
            <th>Aantal</th>
            <th>Prijs per stuk</th>
            <th>Korting</th>
            {block.showTotalRange && <th>Totaal</th>}
          </tr>
        </thead>
        <tbody>
          {block.tiers.map((tier, index) => (
            <tr
              key={index}
              className={activeTier === tier ? 'active' : ''}
            >
              <td>
                {tier.minQuantity} - {tier.maxQuantity || '∞'} stuks
              </td>
              <td>€{tier.price.toFixed(2)}</td>
              <td>
                {tier.discount && (
                  <span className="staffel-discount">
                    {tier.discount}%
                  </span>
                )}
              </td>
              {block.showTotalRange && (
                <td>{calculateTotalRange(tier)}</td>
              )}
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  )
}
