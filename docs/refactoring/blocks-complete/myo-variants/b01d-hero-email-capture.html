File: src/types/email.ts

export interface EmailValidationResult {
  valid: boolean
  error?: string
}

export function validateEmail(email: string): EmailValidationResult {
  const trimmed = email.trim()
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/

  if (!trimmed) {
    return { valid: false, error: 'Email is required' }
  }

  if (!regex.test(trimmed)) {
    return { valid: false, error: 'Invalid email format' }
  }

  return { valid: true }
}

STEP 4: Email Signup API Route

Create the API endpoint to handle email signups with Mailchimp/Resend integration.

File: src/app/api/email-signup/route.ts

import { NextRequest, NextResponse } from 'next/server'

export async function POST(req: NextRequest) {
  try {
    const body = await req.json()
    const { email, listId, source } = body

    // Validate email
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
    if (!email || !emailRegex.test(email)) {
      return NextResponse.json(
        { success: false, error: 'Invalid email address' },
        { status: 400 }
      )
    }

    // Option 1: Mailchimp Integration
    if (process.env.MAILCHIMP_API_KEY && process.env.MAILCHIMP_LIST_ID) {
      const DATACENTER = process.env.MAILCHIMP_API_KEY.split('-')[1]
      const url = `https://${DATACENTER}.api.mailchimp.com/3.0/lists/${listId || process.env.MAILCHIMP_LIST_ID}/members`

      const response = await fetch(url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${process.env.MAILCHIMP_API_KEY}`,
        },
        body: JSON.stringify({
          email_address: email,
          status: 'subscribed',
          tags: [source],
        }),
      })

      const data = await response.json()

      if (response.ok) {
        return NextResponse.json({ success: true, message: 'Subscription successful' })
      } else {
        if (data.title === 'Member Exists') {
          return NextResponse.json(
            { success: false, error: 'This email is already subscribed' },
            { status: 409 }
          )
        }
        throw new Error(data.detail || 'Mailchimp error')
      }
    }

    // Option 2: Resend Integration
    if (process.env.RESEND_API_KEY && process.env.RESEND_AUDIENCE_ID) {
      const response = await fetch('https://api.resend.com/audiences/${process.env.RESEND_AUDIENCE_ID}/contacts', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${process.env.RESEND_API_KEY}`,
        },
        body: JSON.stringify({ email }),
      })

      const data = await response.json()

      if (response.ok) {
        return NextResponse.json({ success: true, message: 'Subscription successful' })
      } else {
        throw new Error(data.message || 'Resend error')
      }
    }

    // Option 3: Save to Payload CMS (custom collection)
    // Requires creating an "EmailSubscribers" collection in Payload
    // This is a fallback if no email service is configured

    return NextResponse.json({ success: true, message: 'Email saved' })
  } catch (error) {
    console.error('[Email Signup Error]', error)
    return NextResponse.json(
      { success: false, error: 'Subscription failed. Please try again.' },
      { status: 500 }
    )
  }
}
