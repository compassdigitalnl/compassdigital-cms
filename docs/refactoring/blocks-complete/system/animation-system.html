Animation System - Block Animations

Opt-in scroll animations via Payload CMS settings • Intersection Observer based • Accessibility first

📦 System Overview: This animation system provides smooth, performant scroll-triggered animations for all Payload CMS blocks. Animations are opt-in per block, respect user preferences (prefers-reduced-motion), and use Intersection Observer for optimal performance.

STEP 4: React Hook - useBlockAnimation

Custom hook to handle Intersection Observer setup and return animation class names.

File: src/hooks/useBlockAnimation.ts

'use client'

import { useEffect, useRef, useState } from 'react'
import { AnimationConfig } from '@/types/animation'

interface UseBlockAnimationOptions {
  // Optional override for global settings
  threshold?: number
  rootMargin?: string
}

export function useBlockAnimation(
  config?: AnimationConfig,
  options?: UseBlockAnimationOptions
) {
  const elementRef = useRef<HTMLElement | null>(null)
  const [isVisible, setIsVisible] = useState(false)

  useEffect(() => {
    // Skip if animations disabled or prefers-reduced-motion
    if (!config?.enabled) return
    if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
      setIsVisible(true)
      return
    }

    const element = elementRef.current
    if (!element) return

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting && !isVisible) {
            setIsVisible(true)
            // Optional: unobserve after animation triggers (one-time animation)
            // observer.unobserve(entry.target)
          }
        })
      },
      {
        threshold: options?.threshold ?? 0.1,
        rootMargin: options?.rootMargin ?? '0px 0px -100px 0px',
      }
    )

    observer.observe(element)

    return () => {
      if (element) {
        observer.unobserve(element)
      }
    }
  }, [config, isVisible, options])

  // Build class names
  const classes: string[] = []

  if (config?.enabled) {
    classes.push('animate-on-scroll')

    // Animation type
    if (config.type) {
      classes.push(`animate-${config.type}`)
    }

    // Duration
    if (config.duration) {
      classes.push(`animate-duration-${config.duration}`)
    }

    // Visible state
    if (isVisible) {
      classes.push('is-visible')
    }
  }

  return {
    ref: elementRef,
    className: classes.join(' '),
    isVisible,
  }
}

Alternative: Class-based Hook (returns only classes)

'use client'

import { useEffect, useRef } from 'react'
import { AnimationConfig } from '@/types/animation'

export function useAnimationClasses(config?: AnimationConfig): string {
  if (!config?.enabled) return ''

  const classes: string[] = ['animate-on-scroll']

  if (config.type) classes.push(`animate-${config.type}`)
  if (config.duration) classes.push(`animate-duration-${config.duration}`)

  return classes.join(' ')
}

// Separate observer hook
export function useScrollObserver(
  ref: React.RefObject<HTMLElement>,
  enabled: boolean = true
) {
  useEffect(() => {
    if (!enabled) return
    if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return

    const element = ref.current
    if (!element) return

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            entry.target.classList.add('is-visible')
          }
        })
      },
      {
        threshold: 0.1,
        rootMargin: '0px 0px -100px 0px',
      }
    )

    observer.observe(element)

    return () => observer.unobserve(element)
  }, [ref, enabled])
}

Example 3: Global Observer Component (Automatic)

// File: src/components/AnimationObserver.tsx

'use client'

import { useEffect } from 'react'

interface AnimationObserverProps {
  threshold?: number
  rootMargin?: string
}

export const AnimationObserver: React.FC<AnimationObserverProps> = ({
  threshold = 0.1,
  rootMargin = '0px 0px -100px 0px',
}) => {
  useEffect(() => {
    // Check prefers-reduced-motion
    if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
      // Force all animations visible
      const elements = document.querySelectorAll('.animate-on-scroll')
      elements.forEach((el) => el.classList.add('is-visible'))
      return
    }

    // Create observer
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            entry.target.classList.add('is-visible')
          }
        })
      },
      { threshold, rootMargin }
    )

    // Observe all animated elements
    const elements = document.querySelectorAll('.animate-on-scroll')
    elements.forEach((el) => observer.observe(el))

    return () => {
      elements.forEach((el) => observer.unobserve(el))
    }
  }, [threshold, rootMargin])

  return null // This is a utility component, renders nothing
}

// Usage in layout:
// <AnimationObserver threshold={0.1} rootMargin="0px 0px -100px 0px" />

STEP 8: Troubleshooting

Common issues and solutions when implementing the animation system.

Problem Cause Solution
Animations not triggering CSS classes missing or Intersection Observer not initialized 1. Check that CSS animation classes are in globals.css
2. Verify useBlockAnimation hook is called
3. Check browser console for errors
4. Ensure animation.enabled is true
Content invisible on page load CSS hides content before JS initializes Add fallback: Show content after 2s if JS hasn't loaded:
setTimeout(() => { el.classList.add('is-visible') }, 2000)
Animations play multiple times Observer not unobserving after trigger In observer callback, add:
observer.unobserve(entry.target)
after adding 'is-visible' class
Stagger delays not working data-delay attribute missing or CSS not targeting it 1. Ensure elements have data-delay="1" attribute
2. Check CSS selector: .animate-on-scroll[data-delay="1"]
3. Verify stagger: true in config
Two-column animation not symmetric Left/right transforms swapped Check CSS:
.animate-fade-right { transform: translateX(-32px) } (left → right)
.animate-fade-left { transform: translateX(32px) } (right → left)
Jerky/laggy animations Too many elements animating at once or heavy JS on main thread 1. Use will-change: transform, opacity sparingly
2. Reduce stagger count (max 5-6 elements)
3. Use contain: layout on animated containers
4. Profile with Chrome DevTools Performance tab
Animations disabled for all users Global animation setting is off or prefers-reduced-motion detected incorrectly 1. Check Payload global settings: animations.enabled
2. Test media query: window.matchMedia('(prefers-reduced-motion: reduce)').matches
3. Clear browser settings (may be cached)
Hook causing infinite re-renders useEffect dependencies not stable Memoize config object:

2. Import types: import { AnimationConfig } from '@/types/animation'
3. Extend block interfaces with AnimatedBlock

npm install intersection-observer
