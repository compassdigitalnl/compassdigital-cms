Step 2: TypeScript Type Definitions

Create src/types/blocks.ts for type safety:

// src/types/blocks.ts import { Media } from './payload-types' export type SocialPlatform = | 'linkedin' | 'twitter' | 'facebook' | 'instagram' | 'github' | 'dribbble' export interface SocialLink { platform: SocialPlatform url: string } export interface TeamMember { photo: string | Media name: string role: string bio?: string email?: string social?: SocialLink[] } export interface TeamBlock { blockType: 'team' id?: string title?: string description?: string columns: '2' | '3' | '4' photoVariant: 'square' | 'circle' members: TeamMember[] } // Type guard to check if a block is a TeamBlock export function isTeamBlock(block: any): block is TeamBlock { return block?.blockType === 'team' } // Helper to get icon name from platform export function getSocialIcon(platform: SocialPlatform): string { const icons = { linkedin: 'linkedin', twitter: 'twitter', facebook: 'facebook', instagram: 'instagram', github: 'github', dribbble: 'dribbble', } return icons[platform] }
Step 3: React Component Implementation

Create src/branches/blocks/shared/Team/Component.tsx to render the block:

// src/branches/blocks/shared/Team/Component.tsx import React from 'react' import Image from 'next/image' import { TeamBlock, getSocialIcon, Media } from '@/types/blocks' import { cn } from '@/lib/utils' import { Mail, Linkedin, Twitter, Facebook, Instagram, Github, Dribbble } from 'lucide-react' interface TeamProps { block: TeamBlock } const iconMap = { linkedin: Linkedin, twitter: Twitter, facebook: Facebook, instagram: Instagram, github: Github, dribbble: Dribbble, } export const Team: React.FC<TeamProps> = ({ block }) => { const { title, description, columns, photoVariant, members } = block // Map columns to Tailwind grid classes const gridClass = { '2': 'md:grid-cols-2', '3': 'md:grid-cols-2 lg:grid-cols-3', '4': 'md:grid-cols-2 lg:grid-cols-4', }[columns] return ( <section className="py-16 md:py-24"> <div className="container"> {/* Section Header */} {(title || description) && ( <div className="text-center mb-12 md:mb-16"> {title && ( <h2 className="font-display text-3xl md:text-4xl text-navy mb-3"> {title} </h2> )} {description && ( <p className="text-grey-dark text-base max-w-2xl mx-auto"> {description} </p> )} </div> )} {/* Team Grid */} <div className={cn('grid gap-6', gridClass)}> {members.map((member, index) => { const photoUrl = typeof member.photo === 'string' ? member.photo : (member.photo as Media)?.url return ( <div key={index} className="group bg-white border border-grey rounded-xl overflow-hidden shadow-sm hover:shadow-lg hover:-translate-y-1 transition-all duration-200" > {/* Photo */} <div className={cn( "relative aspect-square bg-grey-light overflow-hidden", photoVariant === 'circle' && "m-6 rounded-full" )}> {photoUrl ? ( <Image src={photoUrl} alt={member.name} fill className="object-cover" /> ) : ( <div className="w-full h-full flex items-center justify-center text-6xl"> 👤 </div> )} </div> {/* Info */} <div className="p-5"> <h3 className="font-display text-xl font-bold text-navy mb-1"> {member.name} </h3> <div className="text-xs font-bold uppercase tracking-wide text-teal mb-3"> {member.role} </div> {member.bio && ( <p className="text-sm text-grey-dark leading-relaxed mb-4"> {member.bio} </p> )} {/* Contact & Social */} {(member.email || member.social?.length > 0) && ( <div className="flex items-center gap-2 pt-3 border-t border-grey-light"> {member.email && ( <a href={`mailto:${member.email}`} className="flex items-center gap-1.5 text-xs text-grey-dark hover:text-teal transition-colors flex-1" > <Mail className="w-3.5 h-3.5 text-teal" /> <span className="truncate">{member.email}</span> </a> )} {member.social && member.social.length > 0 && ( <div className="flex gap-1.5 ml-auto"> {member.social.map((link, idx) => { const Icon = iconMap[link.platform] return ( <a key={idx} href={link.url} target="_blank" rel="noopener noreferrer" className="w-8 h-8 rounded-lg bg-grey-light flex items-center justify-center text-grey-dark hover:bg-teal hover:text-white hover:-translate-y-0.5 transition-all" title={link.platform} > <Icon className="w-4 h-4" /> </a> ) })} </div> )} </div> )} </div> </div> ) })} </div> </div> </section> ) }

Step 5: Testing Checklist
1. Block appears in page builder: Navigate to Pages → Edit → Blocks → Add Block → "Team Block" is visible
2. All fields load correctly: Title, description, columns, photoVariant, and members array all render in admin
3. Photo upload works: Upload team member photo → verify image appears in admin and frontend
4. Grid layouts render correctly: Test 2, 3, and 4 column layouts → verify responsive behavior on mobile
5. Photo variants work: Test square vs circle photo shapes → verify CSS classes applied correctly
6. Social links functional: Add LinkedIn, Twitter, etc. → verify icons display and links open correctly
7. Email validation works: Enter invalid email → verify validation error shows
8. Hover effects work: Hover over team cards and social icons → verify transitions smooth
9. TypeScript compilation: Run npm run build → no TypeScript errors

Step 7: Common Issues & Solutions
Issue Solution
Photos not displaying on frontend Check photo URL extraction: typeof member.photo === 'string' ? member.photo : member.photo?.url. Ensure Media type properly imported.
Grid layout broken on mobile Verify responsive classes: grid md:grid-cols-2 lg:grid-cols-3. Check Tailwind config includes breakpoints.
Social icons not showing Ensure lucide-react installed: npm install lucide-react. Check iconMap includes all platforms.
Circle photos not round Check className: photoVariant === 'circle' && 'rounded-full'. Verify parent container has m-6 for padding.
Email validation not working Check validate function in config.ts → ensure regex /^https?:\/\/.+/ tests URL correctly.
Database error on save Run migration: npx payload migrate → verify pages_blocks_members and pages_blocks_members_social tables exist.
TypeScript error: Media type not found Import from correct path: import { Media } from '@/types/payload-types' (auto-generated by Payload).
Hover effects not working Check Tailwind classes: hover:shadow-lg hover:-translate-y-1 transition-all. Ensure group class on parent for nested hovers.
Step 8: Register Block in Payload Config

Add the Team block to your Pages collection's blocks field:

// src/collections/Pages.ts import { CollectionConfig } from 'payload/types' import { Team } from '@/branches/blocks/shared/Team/config' // Import other blocks... export const Pages: CollectionConfig = { slug: 'pages', fields: [ { name: 'title', type: 'text', required: true, }, { name: 'slug', type: 'text', required: true, unique: true, }, { name: 'blocks', type: 'blocks', label: 'Page Blocks', blocks: [ Team, // ← Add Team block here // Hero, // Features, // CTA, // etc... ], }, ], }

Next Step: Add the block component to your page renderer:

// app/(frontend)/[...slug]/page.tsx import { Team } from '@/branches/blocks/shared/Team/Component' export default function Page({ page }) { return ( <main> {page.blocks?.map((block, index) => { switch (block.blockType) { case 'team': return <Team key={index} block={block} /> // Other blocks... case 'hero': return <Hero key={index} block={block} /> case 'features': return <Features key={index} block={block} /> default: return null } })} </main> ) }
